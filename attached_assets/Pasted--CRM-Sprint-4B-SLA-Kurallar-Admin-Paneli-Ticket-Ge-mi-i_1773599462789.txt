```typescript
export const slaRules = pgTable('sla_rules', {
  id: serial('id').primaryKey(),
  department: varchar('department', { length: 50 }).notNull(), 
  // teknik, lojistik, muhasebe, marketing, trainer, hr
  priority: varchar('priority', { length: 20 }).notNull(),
  // kritik, yuksek, normal, dusuk
  hoursLimit: integer('hours_limit').notNull(), // SLA deadline in hours
  isActive: boolean('is_active').notNull().default(true),
  updatedBy: varchar('updated_by').references(() => users.id),
  updatedAt: timestamp('updated_at').defaultNow().notNull(),
  createdAt: timestamp('created_at').defaultNow().notNull(),
});

export type SlaRule = typeof slaRules.$inferSelect;
export type InsertSlaRule = typeof slaRules.$inferInsert;
```

## TASK 2 — SLA Rules API

Add to `server/routes/crm-iletisim.ts`:

```typescript
// GET /api/iletisim/sla-rules — get all SLA rules (authenticated)
router.get('/sla-rules', requireAuth, async (req, res) => {
  try {
    const rules = await db
      .select()
      .from(slaRules)
      .orderBy(asc(slaRules.department), asc(slaRules.priority));
    res.json(rules);
  } catch (err) {
    res.status(500).json({ error: 'SLA kuralları alınamadı' });
  }
});

// PATCH /api/iletisim/sla-rules/:id — update a rule (admin/ceo only)
router.patch('/sla-rules/:id', requireAuth, requireRole(['admin', 'ceo']), async (req, res) => {
  try {
    const id = parseInt(req.params.id);
    const { hoursLimit } = req.body;
    const userId = (req as any).user?.id;

    if (!hoursLimit || hoursLimit < 1 || hoursLimit > 720) {
      return res.status(400).json({ error: 'hoursLimit must be between 1 and 720' });
    }

    const [updated] = await db
      .update(slaRules)
      .set({ hoursLimit, updatedBy: userId, updatedAt: new Date() })
      .where(eq(slaRules.id, id))
      .returning();

    if (!updated) return res.status(404).json({ error: 'Kural bulunamadı' });
    res.json(updated);
  } catch (err) {
    res.status(500).json({ error: 'SLA kuralı güncellenemedi' });
  }
});

// POST /api/iletisim/sla-rules/reset — reset all rules to defaults (admin only)
router.post('/sla-rules/reset', requireAuth, requireRole(['admin']), async (req, res) => {
  try {
    const DEFAULTS = [
      { department: 'teknik',    priority: 'kritik', hoursLimit: 4 },
      { department: 'teknik',    priority: 'yuksek', hoursLimit: 8 },
      { department: 'teknik',    priority: 'normal',  hoursLimit: 24 },
      { department: 'teknik',    priority: 'dusuk',   hoursLimit: 48 },
      { department: 'lojistik',  priority: 'kritik', hoursLimit: 8 },
      { department: 'lojistik',  priority: 'yuksek', hoursLimit: 12 },
      { department: 'lojistik',  priority: 'normal',  hoursLimit: 24 },
      { department: 'lojistik',  priority: 'dusuk',   hoursLimit: 48 },
      { department: 'muhasebe',  priority: 'kritik', hoursLimit: 12 },
      { department: 'muhasebe',  priority: 'yuksek', hoursLimit: 24 },
      { department: 'muhasebe',  priority: 'normal',  hoursLimit: 48 },
      { department: 'muhasebe',  priority: 'dusuk',   hoursLimit: 72 },
      { department: 'marketing', priority: 'kritik', hoursLimit: 24 },
      { department: 'marketing', priority: 'yuksek', hoursLimit: 48 },
      { department: 'marketing', priority: 'normal',  hoursLimit: 72 },
      { department: 'marketing', priority: 'dusuk',   hoursLimit: 96 },
      { department: 'trainer',   priority: 'kritik', hoursLimit: 12 },
      { department: 'trainer',   priority: 'yuksek', hoursLimit: 24 },
      { department: 'trainer',   priority: 'normal',  hoursLimit: 48 },
```

Update the `calculateSlaDeadline` function (or wherever slaDeadline is set) 
to query the sla_rules table:

```typescript
import { db } from '../db';
import { slaRules } from '../../shared/schema';
import { and, eq } from 'drizzle-orm';

async function getSlaHours(department: string, priority: string): Promise<number> {
  try {
    const [rule] = await db
      .select()
      .from(slaRules)
      .where(
        and(
          eq(slaRules.department, department),
          eq(slaRules.priority, priority),
          eq(slaRules.isActive, true)
        )
      );
    return rule?.hoursLimit ?? 24; // fallback to 24h if rule not found
  } catch {
    return 24; // safe fallback
  }
}
```

## TASK 3 — SLA Kuralları Admin UI

Find the "SLA Kuralları" menu item in the CRM nav 
(`client/src/pages/iletisim-merkezi/crm-nav.tsx` — key: 'sla').

**Create:** `client/src/pages/iletisim-merkezi/sla-rules-panel.tsx`

```tsx
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { useState } from 'react';
import { cn } from '@/lib/utils';

const DEPT_LABELS: Record<string, string> = {
  teknik:    '🔧 Teknik Destek',
  lojistik:  '🚚 Lojistik',
  muhasebe:  '💰 Muhasebe',
  marketing: '📣 Marketing',
  trainer:   '🎓 Trainer',
  hr:        '👥 İK',
};

const PRIORITY_LABELS: Record<string, { label: string; color: string }> = {
  kritik: { label: 'Kritik',  color: '#dc2626' },
  yuksek: { label: 'Yüksek',  color: '#f97316' },
  normal: { label: 'Normal',  color: '#f59e0b' },
  dusuk:  { label: 'Düşük',   color: '#94a3b8' },
};

interface SlaRule {
  id: number;
  department: string;
  priority: string;
  hoursLimit: number;
  updatedAt: string;
}

export function SlaRulesPanel({ isAdmin }: { isAdmin: boolean }) {
  const qc = useQueryClient();
  const [editingId, setEditingId] = useState<number | null>(null);
  const [editValue, setEditValue] = useState<string>('');

  const { data: rules = [], isLoading } = useQuery<SlaRule[]>({
    queryKey: ['/api/iletisim/sla-rules'],
    queryFn: () => fetch('/api/iletisim/sla-rules').then(r => r.json()),
  });

  const updateMutation = useMutation({
    mutationFn: ({ id, hoursLimit }: { id: number; hoursLimit: number }) =>
      fetch(`/api/iletisim/sla-rules/${id}`, {
        method: 'PATCH',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ hoursLimit }),
      }).then(r => r.json()),
    onSuccess: () => {
      qc.invalidateQueries({ queryKey: ['/api/iletisim/sla-rules'] });
      setEditingId(null);
    },
  });

  const resetMutation = useMutation({
    mutationFn: () =>
      fetch('/api/iletisim/sla-rules/reset', { method: 'POST' }).then(r => r.json()),
    onSuccess: () => qc.invalidateQueries({ queryKey: ['/api/iletisim/sla-rules'] }),
  });

  // Group rules by department
  const grouped = rules.reduce((acc, rule) => {
    if (!acc[rule.department]) acc[rule.department] = [];
    acc[rule.department].push(rule);
    return acc;
  }, {} as Record<string, SlaRule[]>);

  const PRIORITY_ORDER = ['kritik', 'yuksek', 'normal', 'dusuk'];

  if (isLoading) return <div className="p-6 text-sm text-muted-foreground">Yükleniyor...</div>;

  return (
    <div className="flex-1 overflow-y-auto p-6">
      {/* Header */}
      <div className="flex items-center justify-between mb-6">
        <div>
          <h2 className="text-lg font-extrabold text-foreground">SLA Kuralları</h2>
          <p className="text-xs text-muted-foreground mt-1">
            Departman ve öncelik bazlı yanıt süreleri (saat cinsinden)
          </p>
        </div>
        {isAdmin && (
          <button
            onClick={() => resetMutation.mutate()}
            disabled={resetMutation.isPending}
            className="text-xs px-3 py-2 rounded-lg border border-border bg-background text-muted-foreground hover:bg-accent transition-colors disabled:opacity-50"
          >
            ↺ Varsayılana Sıfırla
          </button>
        )}
      </div>

      {/* Rules table per department */}
      {Object.entries(DEPT_LABELS).map(([dept, deptLabel]) => {
        const deptRules = grouped[dept] ?? [];
        if (!deptRules.length) return null;

        return (
          <div key={dept} className="mb-6 bg-card border border-border rounded-xl overflow-hidden">
            <div className="px-4 py-3 border-b border-border bg-muted/30">
              <span className="text-sm font-bold text-foreground">{deptLabel}</span>
            </div>
            <table className="w-full">
              <thead>
                <tr className="border-b border-border">
                  <th className="text-left px-4 py-2 text-xs font-semibold text-muted-foreground">Öncelik</th>
                  <th className="text-left px-4 py-2 text-xs font-semibold text-muted-foreground">SLA Süresi</th>
                  <th className="text-left px-4 py-2 text-xs font-semibold text-muted-foreground">Son Güncelleme</th>
                  {isAdmin && <th className="px-4 py-2"></th>}
                </tr>
              </thead>
              <tbody>
                {PRIORITY_ORDER.map(priority => {
                  const rule = deptRules.find(r => r.priority === priority);
                  if (!rule) return null;
                  const pInfo = PRIORITY_LABELS[priority];
                  const isEditing = editingId === rule.id;

                  return (
                    <tr key={rule.id} className="border-b border-border last:border-0 hover:bg-muted/20 transition-colors">
                      <td className="px-4 py-3">
                        <span className="text-xs font-semibold" style={{ color: pInfo.color }}>
                          {pInfo.label}
                        </span>
                      </td>
                      <td className="px-4 py-3">
                        {isEditing ? (
                          <div className="flex items-center gap-2">
                            <input
                              type="number"
                              value={editValue}
                              onChange={e => setEditValue(e.target.value)}
                              min={1}
                              max={720}
                              className="w-16 text-xs px-2 py-1 rounded border border-border bg-background text-foreground outline-none focus:border-[#122549]"
                              autoFocus
                            />
                            <span className="text-xs text-muted-foreground">saat</span>
                            <button
                              onClick={() => updateMutation.mutate({ id: rule.id, hoursLimit: parseInt(editValue) })}
                              disabled={updateMutation.isPending || !editValue}
                              className="text-xs px-2 py-1 rounded bg-[#122549] text-white disabled:opacity-50"
                            >
                              Kaydet
                            </button>
                            <button
                              onClick={() => setEditingId(null)}
                              className="text-xs px-2 py-1 rounded border border-border text-muted-foreground"
                            >
                              İptal
                            </button>
                          </div>
                        ) : (
                          <span className="text-sm font-bold text-foreground">
                            {rule.hoursLimit}s
                            <span className="text-xs font-normal text-muted-foreground ml-1">
                              ({rule.hoursLimit >= 24
                                ? `${Math.floor(rule.hoursLimit / 24)} gün`
                                : `${rule.hoursLimit} saat`})
                            </span>
                          </span>
                        )}
                      </td>
                      <td className="px-4 py-3 text-xs text-muted-foreground">
                        {new Date(rule.updatedAt).toLocaleDateString('tr-TR')}
                      </td>
                      {isAdmin && (
                        <td className="px-4 py-3 text-right">
                          {!isEditing && (
                            <button
                              onClick={() => {
                                setEditingId(rule.id);
                                setEditValue(rule.hoursLimit.toString());
                              }}
                              className="text-xs px-2 py-1 rounded border border-border text-muted-foreground hover:bg-accent transition-colors"
                            >
                              Düzenle
                            </button>
                          )}
                        </td>
                      )}
                    </tr>
                  );
                })}
              </tbody>
            </table>
          </div>
        );
      })}
    </div>
  );
}
```
