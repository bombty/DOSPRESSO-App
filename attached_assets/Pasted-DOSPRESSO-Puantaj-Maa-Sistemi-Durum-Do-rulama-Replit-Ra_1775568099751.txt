BÖLÜM 1: VERİTABANI DURUM KONTROLÜ
Aşağıdaki SQL sorgularını çalıştır ve sonuçları raporla:
sql-- 1a. PDKS kayıtları
SELECT COUNT(*) as total, 
       COUNT(DISTINCT user_id) as unique_users,
       MIN(record_date) as first_date,
       MAX(record_date) as last_date,
       record_type, COUNT(*) as type_count
FROM pdks_records 
GROUP BY record_type;

-- 1b. Aylık bordro kayıtları (basit motor)
SELECT year, month, COUNT(*) as records, 
       status, SUM(net_pay) as total_net
FROM monthly_payroll 
GROUP BY year, month, status
ORDER BY year, month;

-- 1c. Aylık bordro kayıtları (detaylı motor - monthlyPayrolls)
SELECT year, month, COUNT(*) as records, status
FROM monthly_payrolls 
GROUP BY year, month, status
ORDER BY year, month;

-- 1d. Pozisyon maaşları
SELECT position_code, position_name, total_salary, base_salary, bonus, 
       effective_from, effective_to
FROM position_salaries 
ORDER BY position_code;

-- 1e. Bordro parametreleri
SELECT year, is_active, minimum_wage_gross, minimum_wage_net,
       working_days_per_month, working_hours_per_day, overtime_multiplier,
       meal_allowance_tax_exempt_daily
FROM payroll_parameters 
ORDER BY year DESC;

-- 1f. Resmi tatiller
SELECT * FROM public_holidays WHERE year = 2026 ORDER BY date;
-- Eğer boşsa: SELECT * FROM public_holidays ORDER BY date;

-- 1g. Scheduled offs
SELECT COUNT(*) as total, 
       COUNT(DISTINCT user_id) as unique_users,
       off_type, COUNT(*) as type_count
FROM scheduled_offs 
GROUP BY off_type;

-- 1h. Vardiya planları
SELECT COUNT(*) as total,
       shift_type, COUNT(*) as type_count,
       MIN(shift_date) as first_date,
       MAX(shift_date) as last_date
FROM shifts WHERE deleted_at IS NULL
GROUP BY shift_type;

-- 1i. Shift attendance kayıtları
SELECT COUNT(*) as total,
       status, COUNT(*) as status_count
FROM shift_attendance
GROUP BY status;

-- 1j. Leave requests
SELECT COUNT(*) as total,
       leave_type, status, COUNT(*) as count
FROM leave_requests
GROUP BY leave_type, status;

-- 1k. Salary deduction types
SELECT code, name, calculation_type, is_automatic, is_active
FROM salary_deduction_types
ORDER BY code;

-- 1l. Monthly attendance summaries
SELECT COUNT(*) as total FROM monthly_attendance_summaries;
