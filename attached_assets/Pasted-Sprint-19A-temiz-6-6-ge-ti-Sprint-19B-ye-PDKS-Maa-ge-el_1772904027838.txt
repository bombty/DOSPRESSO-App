## TASK: Sprint 19B — PDKS + Maaş Hesaplama Modülü

### ÖZET

DOSPRESSO şubelerinde personel giriş/çıkış takibi (PDKS) ve aylık maaş/hakediş hesaplaması yapılacak. Şu an Excel'de manuel yapılıyor. Bu sprint sisteme entegre otomatik PDKS + maaş hesaplama modülü kuruyor.

**KRİTİK İŞ KURALLARI (DOSPRESSO'ya özel — değiştirme):**

1. **Günlük ücret = Toplam Maaş ÷ 30** (İş Kanunu Md.49 — Şubat 28 gün olsa bile bölen DAİMA 30)
2. **Eksik gün kesintisi = (Eksik Gün + 1) × Günlük Ücret** (+1 ceza günü)
3. **Prim kesintisi = (Ücretsiz İzin Günü ÷ 30) × Prim** (sadece prim kesilir, taban korunur)
4. **FM = (FM Dakika ÷ 60) × Saatlik Ücret × 1.5** (Saatlik = Toplam Maaş ÷ 240)
5. **Kapanış vardiyası:** 00:00-05:59 arası TEK damga = önceki gecenin kapanış çıkışı → o gün OFF
6. **Off limiti:** Kişi başı max 4 off/ay
7. **NET = Toplam Maaş − Gün Kesintisi − Prim Kesintisi + FM Tutarı**
8. **İşten ayrılan:** NET = Çalışılan Gün × Günlük Ücret + FM Tutarı

### T003 — Gün Sınıflandırma Motoru (~2-3 saat)

**KRİTİK — DOSPRESSO'nun özel kapanış vardiyası mantığı**

```typescript
// server/lib/pdks-engine.ts (yeni)

interface DayClassification {
  date: string;
  status: 'worked' | 'program_off' | 'kapanish_off' | 'absent' | 'unpaid_leave' | 'sick_leave';
  records: PdksRecord[];
  workedMinutes: number;
  overtimeMinutes: number;
}

function classifyDay(
  date: Date,
  records: PdksRecord[],
  scheduledOffs: ScheduledOff[],
  leaves: LeaveRequest[]
): DayClassification {
  
  // 1. İzinli mi kontrol et
  const leave = leaves.find(l => l.date === dateStr);
  if (leave) {
    if (leave.type === 'ucretsiz_izin') return { status: 'unpaid_leave', ... };
    if (leave.type === 'rapor') return { status: 'sick_leave', ... };
  }
  
  // 2. PDKS kaydı var mı?
  const dayRecords = records.filter(r => r.record_date === dateStr);
  
  if (dayRecords.length === 0) {
    // Hiç kayıt yok
    const isScheduledOff = scheduledOffs.find(o => o.off_date === dateStr);
    if (isScheduledOff) return { status: 'program_off', ... };
    return { status: 'absent', ... }; // DEVAMSIZ
  }
  
  // 3. Kapanış off kontrolü
  // 00:00-05:59 arası TEK damga varsa = kapanış off
  const hasOnlyNightStamp = dayRecords.every(r => {
    const hour = parseInt(r.record_time.split(':')[0]);
    return hour >= 0 && hour < 6;
  }) && dayRecords.length === 1;
  
  if (hasOnlyNightStamp) {
    return { status: 'kapanish_off', ... };
  }
  
  // 4. 06:00+ kaydı var = ÇALIŞTI
  const hasWorkStamp = dayRecords.some(r => {
    const hour = parseInt(r.record_time.split(':')[0]);
    return hour >= 6;
  });
  
  if (hasWorkStamp) {
    // Çalışma süresi hesapla
    const workedMinutes = calculateWorkedMinutes(dayRecords);
    const overtimeMinutes = Math.max(0, workedMinutes - 480); // 480dk = 8 saat
    return { status: 'worked', workedMinutes, overtimeMinutes, ... };
  }
  
  return { status: 'absent', ... };
}

function calculateWorkedMinutes(records: PdksRecord[]): number {
  // Giriş-çıkış çiftlerini bul
  const sorted = records.sort((a, b) => a.record_time.localeCompare(b.record_time));
  let total = 0;
  
  for (let i = 0; i < sorted.length - 1; i += 2) {
    if (sorted[i].type === 'giris' && sorted[i+1]?.type === 'cikis') {
      const diff = timeDiffMinutes(sorted[i].record_time, sorted[i+1].record_time);
      total += diff;
    }
  }
  
  return total;
}
```

### T004 — Maaş Hesaplama Motoru (~2-3 saat)

```typescript
// server/lib/payroll-engine.ts (yeni)

interface PayrollCalculation {
  userId: string;
  year: number;
  month: number;
  
  // Devam
  totalCalendarDays: number;
  workedDays: number;
  offDays: number;
  absentDays: number;
  unpaidLeaveDays: number;
  sickLeaveDays: number;
  overtimeMinutes: number;
  
  // Maaş
  totalSalary: number; // kuruş
  baseSalary: number;
  bonus: number;
  dailyRate: number; // totalSalary / 30
  
  // Kesintiler
  absenceDeduction: number; // (absent + 1) × dailyRate (absent > 0 ise)
  bonusDeduction: number; // (unpaidLeave / 30) × bonus
  overtimePay: number; // (minutes / 60) × (totalSalary / 240) × 1.5
  
  // Net
  netPay: number;
}

async function calculatePayroll(userId: string, year: number, month: number): Promise<PayrollCalculation> {
  // 1. Kullanıcı pozisyonunu ve maaşını al
  const user = await getUser(userId);
  const salary = await getPositionSalary(user.role, year, month);
  
  // 2. Ayın tüm günlerini sınıflandır
  const daysInMonth = new Date(year, month, 0).getDate();
  const records = await getPdksRecords(userId, year, month);
  const offs = await getScheduledOffs(userId, year, month);
  const leaves = await getLeaves(userId, year, month);
  
  const classifications: DayClassification[] = [];
  for (let day = 1; day <= daysInMonth; day++) {
    const date = new Date(year, month - 1, day);
    classifications.push(classifyDay(date, records, offs, leaves));
  }
  
  // 3. Sayıları topla
  const workedDays = classifications.filter(d => d.status === 'worked').length;
  const offDays = classifications.filter(d => ['program_off', 'kapanish_off'].includes(d.status)).length;
  const absentDays = classifications.filter(d => d.status === 'absent').length;
  const unpaidLeaveDays = classifications.filter(d => d.status === 'unpaid_leave').length;
  const sickLeaveDays = classifications.filter(d => d.status === 'sick_leave').length;
  const totalOvertimeMinutes = classifications.reduce((sum, d) => sum + d.overtimeMinutes, 0);
  
  // 4. Off limiti kontrolü (max 4)
  const effectiveOffDays = Math.min(offDays, 4);
  
  // 5. Maaş hesapla
  const dailyRate = Math.round(salary.totalSalary / 30);
  
  // Gün kesintisi
  let absenceDeduction = 0;
  if (absentDays > 0) {
    absenceDeduction = (absentDays + 1) * dailyRate; // +1 ceza
  }
  
  // Prim kesintisi
  let bonusDeduction = 0;
  if (unpaidLeaveDays > 0) {
    bonusDeduction = Math.round((unpaidLeaveDays / 30) * salary.bonus);
  }
  
  // FM tutarı
  const hourlyRate = Math.round(salary.totalSalary / 240); // 30 gün × 8 saat
  const overtimePay = Math.round((totalOvertimeMinutes / 60) * hourlyRate * 1.5);
  
  // NET
  const netPay = salary.totalSalary - absenceDeduction - bonusDeduction + overtimePay;
  
  return {
    totalCalendarDays: daysInMonth,
    workedDays, offDays: effectiveOffDays, absentDays,
    unpaidLeaveDays, sickLeaveDays, overtimeMinutes: totalOvertimeMinutes,
    totalSalary: salary.totalSalary, baseSalary: salary.baseSalary,
    bonus: salary.bonus, dailyRate,
    absenceDeduction, bonusDeduction, overtimePay,
    netPay
  };
}
```
