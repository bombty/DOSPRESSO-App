ADIM 2: VERİTABANI SORGULARI
Aşağıdaki sorguları çalıştır ve sonuçları raporla:
sql-- A1. Görev durumları
SELECT status, COUNT(*) as cnt FROM tasks GROUP BY status ORDER BY cnt DESC;

-- A2. Görev kaynağı (nereden geliyor?)
SELECT source_type, COUNT(*) as cnt FROM tasks GROUP BY source_type ORDER BY cnt DESC;

-- A3. Tekrarlayan görev patlaması — aynı açıklamayla kaç görev var?
SELECT description, COUNT(*) as cnt FROM tasks 
WHERE status IN ('beklemede', 'gecikmiş') 
GROUP BY description HAVING COUNT(*) > 3 ORDER BY cnt DESC LIMIT 15;

-- A4. CRM ticket durumu
SELECT status, department, COUNT(*) FROM support_tickets GROUP BY status, department ORDER BY department;

-- A5. Eski ticket sistemi
SELECT COUNT(*) as old_tickets FROM hq_support_tickets;

-- A6. Misafir şikayetleri
SELECT status, complaint_source, COUNT(*) FROM guest_complaints GROUP BY status, complaint_source;

-- A7. Müşteri feedback
SELECT COUNT(*) as total, AVG(overall_rating) as avg_rating FROM customer_feedback;

-- A8. Ürün şikayetleri
SELECT status, COUNT(*) FROM product_complaints GROUP BY status;

-- A9. Kiosk ayarları olan şubeler
SELECT b.id, b.name, b.is_active,
  CASE WHEN EXISTS(
    SELECT 1 FROM shifts s WHERE s.branch_id = b.id AND s.deleted_at IS NULL LIMIT 1
  ) THEN '✅' ELSE '❌' END as has_shifts
FROM branches b WHERE b.is_active = true ORDER BY b.name;

-- A10. Tekrarlayan görev şablonları (recurring)
SELECT id, description, recurrence_type, recurrence_interval, 
       last_recurred_at, next_run_at, status
FROM tasks WHERE is_recurring = true ORDER BY next_run_at;
