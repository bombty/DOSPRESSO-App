```tsx
            >
              ×
            </button>
          </div>
        ))}

          +
        </div>
      </div>
    )}

  <input
    ref={fileInputRef}
    type="file"
    multiple
    accept="image/*"
    className="hidden"
    onChange={(e) => {
```

### 4B: Upload photos before creating ticket

```tsx
async function uploadPhotos(files: File[]): Promise<string[]> {
  const urls: string[] = [];
  for (const file of files) {
    const formData = new FormData();
    formData.append('file', file);
    const res = await fetch(`/api/iletisim/tickets/temp-upload`, {
      method: 'POST',
      body: formData,
    });
    if (res.ok) {
      const { url } = await res.json();
      urls.push(url);
    }
  }
  return urls;
}
```

### 4C: Backend — temp upload endpoint

Add to `server/routes/crm-iletisim.ts`:

```typescript
import multer from 'multer';
import path from 'path';
import fs from 'fs';

// Temp upload for ticket creation (before ticket ID exists)
const tempUploadDir = path.join(process.cwd(), 'uploads', 'tickets', 'temp');
if (!fs.existsSync(tempUploadDir)) fs.mkdirSync(tempUploadDir, { recursive: true });

const tempUpload = multer({
  storage: multer.diskStorage({
    destination: tempUploadDir,
    filename: (req, file, cb) => {
      const unique = `${Date.now()}-${Math.random().toString(36).slice(2)}`;
      cb(null, `${unique}${path.extname(file.originalname)}`);
    },
  }),
  limits: { fileSize: 10 * 1024 * 1024 }, // 10MB
  fileFilter: (req, file, cb) => {
    if (file.mimetype.startsWith('image/')) cb(null, true);
    else cb(new Error('Sadece resim dosyaları'));
  },
});

// POST /api/iletisim/tickets/temp-upload
router.post('/tickets/temp-upload', requireAuth, tempUpload.single('file'), (req, res) => {
  if (!req.file) return res.status(400).json({ error: 'Dosya gerekli' });
  // Return a URL that can be used to serve the file
  const url = `/api/iletisim/attachments/${req.file.filename}`;
  res.json({ url, filename: req.file.filename });
});
```

## TASK 5 — Backend Security

**File:** `server/routes/crm-iletisim.ts`

### 5A: Equipment endpoint branch validation

```typescript
// GET /api/equipment?branchId=X
// For branch roles: enforce branchId = user's own branchId
// For HQ roles: allow any branchId

router.get('/equipment-for-ticket', requireAuth, async (req, res) => {
  try {
    const userRole = (req as any).user?.role;
    const userBranchId = (req as any).user?.branchId;
    const requestedBranchId = parseInt(req.query.branchId as string);

    const isBranchRole = ['mudur','supervisor','barista','bar_buddy','stajyer'].includes(userRole);

    // Branch users can only see their own branch's equipment
    if (isBranchRole && requestedBranchId !== userBranchId) {
      return res.status(403).json({ error: 'Sadece kendi şubenizin ekipmanlarına erişebilirsiniz' });
    }

    // Fetch equipment for this branch
    // Use actual equipment table name from schema
    // grep -n "equipment\|ekipman" shared/schema.ts | grep "pgTable" | head -5
    const equipment = await db
      .select()
      .from(branchEquipment) // actual table name
      .where(eq(branchEquipment.branchId, requestedBranchId));

    res.json(equipment);
  } catch (err) {
    res.status(500).json({ error: 'Ekipman listesi alınamadı' });
  }
});
```

### 5B: Equipment ID validation in ticket creation

When a branch user submits a ticket with `relatedEquipmentId`:
```typescript
// Validate equipment belongs to user's branch:
if (relatedEquipmentId && isBranchRole) {
  const [equipment] = await db
    .select()
    .from(branchEquipment)
    .where(
      and(
        eq(branchEquipment.id, relatedEquipmentId),
        eq(branchEquipment.branchId, userBranchId)
      )
    );
  if (!equipment) {
    return res.status(403).json({ error: 'Geçersiz ekipman seçimi' });
  }
}
```
