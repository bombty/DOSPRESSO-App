### 1C: API endpoints

Add to an appropriate route file (misc.ts or create new dashboard-layout.ts):

```typescript
// GET /api/me/dashboard-layout
// Returns effective layout: user pref > system default > 'classic'
router.get("/api/me/dashboard-layout", isAuthenticated, async (req, res) => {
  try {
    const user = req.user as any;
    let layout = user.dashboardLayout || 'default';
    
    if (layout === 'default') {
      const config = await db.select().from(aiSystemConfig)
        .where(eq(aiSystemConfig.key, 'default_dashboard_layout')).limit(1);
      layout = config[0]?.value || 'classic';
    }
    
    res.json({ 
      layout, 
      userPreference: user.dashboardLayout || 'default',
      available: ['classic', 'mission-control']
    });
  } catch (error) {
    res.json({ layout: 'classic', userPreference: 'default', available: ['classic', 'mission-control'] });
  }
});

// PATCH /api/me/dashboard-layout
router.patch("/api/me/dashboard-layout", isAuthenticated, async (req, res) => {
  try {
    const user = req.user as any;
    const { layout } = req.body;
    
    if (!['default', 'classic', 'mission-control'].includes(layout)) {
      return res.status(400).json({ error: "Geçersiz layout" });
    }
    
    await db.update(users).set({ dashboardLayout: layout }).where(eq(users.id, user.id));
    res.json({ success: true, layout });
  } catch (error) {
    res.status(500).json({ error: "Layout güncellenemedi" });
  }
});

// PATCH /api/admin/default-dashboard-layout (admin only)
router.patch("/api/admin/default-dashboard-layout", isAuthenticated, async (req, res) => {
  const user = req.user as any;
  if (!['admin', 'ceo', 'cgo'].includes(user.role)) {
    return res.status(403).json({ error: "Yetkiniz yok" });
  }
  
  const { layout } = req.body;
  if (!['classic', 'mission-control'].includes(layout)) {
    return res.status(400).json({ error: "Geçersiz layout" });
  }
  
  await db.insert(aiSystemConfig)
    .values({ key: 'default_dashboard_layout', value: layout })
    .onConflictDoUpdate({ target: aiSystemConfig.key, set: { value: layout } });
  
  res.json({ success: true, defaultLayout: layout });
});
```

### 1D: Frontend hook — useDashboardLayout

Create `client/src/hooks/useDashboardLayout.ts`:

```typescript
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { apiRequest } from '@/lib/queryClient';

export function useDashboardLayout() {
  const queryClient = useQueryClient();
  
  const { data, isLoading } = useQuery({
    queryKey: ['/api/me/dashboard-layout'],
    staleTime: 300000, // 5 min cache — layout doesn't change often
  });
  
  const mutation = useMutation({
    mutationFn: (layout: string) => 
      apiRequest('PATCH', '/api/me/dashboard-layout', { layout }),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['/api/me/dashboard-layout'] });
    },
  });
  
  return {
    layout: (data as any)?.layout || 'classic',
    userPreference: (data as any)?.userPreference || 'default',
    available: (data as any)?.available || ['classic', 'mission-control'],
    setLayout: (layout: string) => mutation.mutate(layout),
    isLoading: isLoading || mutation.isPending,
  };
}
```
