# Delegasyon-Sprint-1 — Modül Delegasyon Sistemi (Backend + Frontend)

## Context
DOSPRESSO franchise management platform. Stack: React 18 + TypeScript + Tailwind CSS, Shadcn/ui, TanStack Query v5, Drizzle ORM, PostgreSQL, Wouter routing.

**What is module delegation?**
HQ staff count sometimes decreases (vacation, resignation, etc.).
Admin can temporarily or permanently delegate a module's responsibility 
from one role to another — e.g. "Teknik Destek" from "Teknik Sorumlu" to "CGO".
The receiving role sees the module in their dashboard and CRM nav with a 🔄 badge.
No data is lost. Both roles can see ticket history. 
Temporary delegations auto-expire on a set date.

Do NOT touch: auth system, existing page contents (except dashboard + CRM nav), 
dark/light mode system, factory/academy modules.

## BACKEND — TASK 1: Database Schema

**File:** `shared/schema.ts`

Add the following table at the end of the schema file:

```typescript
// Module Delegations
export const moduleDelegations = pgTable('module_delegations', {
  id: serial('id').primaryKey(),
  
  // Which module is being delegated
  moduleKey: varchar('module_key', { length: 100 }).notNull(),
  moduleName: varchar('module_name', { length: 200 }).notNull(),
  
  // From role → To role
  fromRole: varchar('from_role', { length: 100 }).notNull(),
  toRole: varchar('to_role', { length: 100 }).notNull(),
  
  // Type: 'gecici' (temporary) or 'kalici' (permanent)
  delegationType: varchar('delegation_type', { length: 20 }).notNull().default('gecici'),
  
  // Active toggle — admin can pause without deleting
  isActive: boolean('is_active').notNull().default(true),
  
  // Optional expiry for temporary delegations
  expiresAt: timestamp('expires_at'),
  
  // Reason / note
  note: varchar('note', { length: 500 }),
  
  // Who created this delegation
  createdBy: varchar('created_by', { length: 100 }).notNull(), // users.id (VARCHAR)
  
  // Timestamps
  createdAt: timestamp('created_at').defaultNow().notNull(),
  updatedAt: timestamp('updated_at').defaultNow().notNull(),
});

export type ModuleDelegation = typeof moduleDelegations.$inferSelect;
export type InsertModuleDelegation = typeof moduleDelegations.$inferInsert;
```

After adding, run: `npm run db:push` to create the table.

## BACKEND — TASK 2: API Routes

**Create new file:** `server/routes/delegation-routes.ts`

```typescript
import { Router } from 'express';
import { db } from '../db';
import { moduleDelegations } from '../../shared/schema';
import { eq, and } from 'drizzle-orm';
import { requireAuth, requireRole } from '../middleware/auth'; // use actual middleware paths

const router = Router();

// GET /api/delegations — list all delegations (admin only)
router.get('/', requireAuth, requireRole(['admin', 'ceo']), async (req, res) => {
  try {
    const delegations = await db.select().from(moduleDelegations).orderBy(moduleDelegations.createdAt);
    res.json(delegations);
  } catch (err) {
    res.status(500).json({ error: 'Failed to fetch delegations' });
  }
});

// GET /api/delegations/active — list active delegations for current user's role
router.get('/active', requireAuth, async (req, res) => {
  try {
    const userRole = (req as any).user?.role;
    if (!userRole) return res.json([]);
    
    const now = new Date();
    const active = await db
      .select()
      .from(moduleDelegations)
      .where(
        and(
          eq(moduleDelegations.toRole, userRole),
          eq(moduleDelegations.isActive, true)
        )
      );
    
    // Filter out expired temporary delegations
    const valid = active.filter(d => {
      if (d.delegationType === 'kalici') return true;
      if (!d.expiresAt) return true;
      return d.expiresAt > now;
    });
    
    res.json(valid);
  } catch (err) {
    res.status(500).json({ error: 'Failed to fetch active delegations' });
  }
});

// POST /api/delegations — create new delegation (admin only)
router.post('/', requireAuth, requireRole(['admin', 'ceo']), async (req, res) => {
  try {
    const { moduleKey, moduleName, fromRole, toRole, delegationType, expiresAt, note } = req.body;
    const userId = (req as any).user?.id;
    
    if (!moduleKey || !fromRole || !toRole || !delegationType) {
      return res.status(400).json({ error: 'moduleKey, fromRole, toRole, delegationType required' });
    }
    
    // Check for duplicate active delegation on same module
    const existing = await db
      .select()
      .from(moduleDelegations)
      .where(
        and(
          eq(moduleDelegations.moduleKey, moduleKey),
          eq(moduleDelegations.isActive, true)
        )
      );
    
    if (existing.length > 0) {
      return res.status(409).json({ 
        error: 'Bu modül zaten başka bir role devredilmiş. Önce mevcut delegasyonu kaldırın.' 
      });
    }
    
    const [created] = await db.insert(moduleDelegations).values({
      moduleKey,
      moduleName: moduleName || moduleKey,
      fromRole,
      toRole,
      delegationType,
      expiresAt: expiresAt ? new Date(expiresAt) : null,
      note: note || null,
      createdBy: userId,
    }).returning();
    
    res.status(201).json(created);
  } catch (err) {
    res.status(500).json({ error: 'Failed to create delegation' });
  }
});

// PATCH /api/delegations/:id — update delegation (admin only)
router.patch('/:id', requireAuth, requireRole(['admin', 'ceo']), async (req, res) => {
  try {
    const id = parseInt(req.params.id);
    const { isActive, expiresAt, note, toRole } = req.body;
    
    const updateData: any = { updatedAt: new Date() };
    if (typeof isActive === 'boolean') updateData.isActive = isActive;
    if (expiresAt !== undefined) updateData.expiresAt = expiresAt ? new Date(expiresAt) : null;
    if (note !== undefined) updateData.note = note;
    if (toRole) updateData.toRole = toRole;
    
    const [updated] = await db
      .update(moduleDelegations)
      .set(updateData)
      .where(eq(moduleDelegations.id, id))
      .returning();
    
    if (!updated) return res.status(404).json({ error: 'Delegation not found' });
    res.json(updated);
  } catch (err) {
    res.status(500).json({ error: 'Failed to update delegation' });
  }
});

// DELETE /api/delegations/:id — delete delegation (admin only)
router.delete('/:id', requireAuth, requireRole(['admin', 'ceo']), async (req, res) => {
  try {
    const id = parseInt(req.params.id);
    await db.delete(moduleDelegations).where(eq(moduleDelegations.id, id));
    res.json({ success: true });
  } catch (err) {
    res.status(500).json({ error: 'Failed to delete delegation' });
  }
});

export default router;
```

    crm_teknik:    '/iletisim-merkezi',
    crm_lojistik:  '/iletisim-merkezi',
