echo "SELECT fri.id, fri.name, fri.recipe_id, fr.code as recipe_code
FROM factory_recipe_ingredients fri 
JOIN factory_recipes fr ON fr.id = fri.recipe_id
WHERE fri.raw_material_id IS NULL AND fri.ingredient_type != 'keyblend'
ORDER BY fri.name;" | psql $DATABASE_URL


Her eşleşmeyen malzeme için inventory’de ara:

for name in "LABNE" "TAZE PEYNİR" "KREMA" "TEREYAĞ" "BİTTER ÇİKOLATA" "SÜTLÜ ÇİKOLATA" "BEYAZ ÇİKOLATA" "KAKAO" "KARAMEL" "OREO" "BADEM" "ESMER ŞEKER" "SIVI YAĞ" "ZEYTİN" "KABARTMA" "KARBONAT" "HPMC" "KARBON" "VİZYON" "MOCHA" "ESPRESSO" "MUSKAT" "AROMA"; do
  result=$(echo "SELECT id, code, name FROM inventory WHERE LOWER(name) LIKE LOWER('%${name}%') LIMIT 2;" | psql $DATABASE_URL -t 2>/dev/null)
  if [ -n "$(echo "$result" | tr -d '[:space:]')" ]; then
    echo "✅ $name → $result"
  else
    echo "❌ $name → YOK"
  fi
done
