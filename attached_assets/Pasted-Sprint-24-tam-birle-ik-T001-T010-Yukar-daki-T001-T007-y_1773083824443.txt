### T003 — Route Actions to Correct Role (~2 hours)

**Create routing function** in `server/agent/routing.ts`:

```typescript
async function routeAgentAction(action: { category, subcategory, targetUserId?, branchId? }) {
  // 1. Find matching rule
  const rule = await db.select().from(agentRoutingRules)
    .where(and(
      eq(agentRoutingRules.category, action.category),
      eq(agentRoutingRules.subcategory, action.subcategory),
      eq(agentRoutingRules.isActive, true)
    )).limit(1);
  
  // 2. Fallback to CGO if no rule
  if (!rule.length) return { targetRole: 'cgo', targetUserId: null };
  
  const r = rule[0];
  
  // 3. Check rejection patterns — skip if recently rejected for same person+category
  if (action.targetUserId) {
    const recentRejection = await db.select().from(agentRejectionPatterns)
      .where(and(
        eq(agentRejectionPatterns.targetUserId, action.targetUserId),
        eq(agentRejectionPatterns.category, action.category),
        gt(agentRejectionPatterns.expiresAt, new Date())
      )).limit(1);
    
    if (recentRejection.length) return null; // Skip — previously rejected
  }
  
  // 4. Find recipient user
  let recipientId;
  if (['supervisor', 'mudur'].includes(r.primaryRole) && action.branchId) {
    recipientId = await findUserByRoleAndBranch(r.primaryRole, action.branchId);
  } else {
    recipientId = await findActiveUserByRole(r.primaryRole);
  }
  
  // 5. Notify secondary role (info only)
  if (r.secondaryRole) {
    const secondaryUser = await findActiveUserByRole(r.secondaryRole);
    if (secondaryUser) {
      await createNotification({
        userId: secondaryUser.id,
        title: `[Bilgi] ${action.title}`,
        message: action.description,
        type: 'agent_info'
      });
    }
  }
  
  // 6. Set escalation
  const escalationDate = r.escalationDays 
    ? new Date(Date.now() + r.escalationDays * 86400000) 
    : null;
  
  return {
    targetRole: r.primaryRole,
    targetUserId: recipientId?.id,
    escalationDate,
    escalationRole: r.escalationRole
  };
}
```

**Call routeAgentAction()** wherever agent skills insert pending actions — replace hardcoded CGO targeting.

### T004 — Escalation Background Job (~1 hour)

**Add hourly job** in server scheduler:

```typescript
// Every hour: check for overdue actions that need escalation
async function checkEscalations() {
  const overdue = await db.select().from(agentPendingActions)
    .where(and(
      eq(status, 'pending'),
      isNotNull(escalationDate),
      lt(escalationDate, new Date()),
      isNotNull(escalationRole)
    ));
  
  for (const action of overdue) {
    const escalationUser = await findActiveUserByRole(action.escalationRole);
    if (escalationUser) {
      await createNotification({
        userId: escalationUser.id,
        title: `⚠️ Escalation: ${action.title}`,
        message: `Bu konu ${action.escalationRole === 'cgo' ? 'CGO' : action.escalationRole} seviyesine yükseltildi. ${action.targetRole} tarafından ${Math.ceil((Date.now() - action.createdAt) / 86400000)} gündür çözülmedi.`,
        type: 'agent_escalation',
        priority: 'high',
        link: '/agent-center'
      });
    }
    // Mark as escalated
    await db.update(agentPendingActions)
      .set({ status: 'escalated', escalatedAt: new Date() })
      .where(eq(agentPendingActions.id, action.id));
  }
}
```

### T007 — Outcome Tracking Background Job (~1 hour)

**Daily job (runs at 08:00):**

```typescript
async function checkActionOutcomes() {
  // Find outcomes that need follow-up
  const pending = await db.select().from(agentActionOutcomes)
    .where(and(
      isNull(agentActionOutcomes.outcome),
      lte(agentActionOutcomes.followUpDate, new Date())
    ));
  
  for (const o of pending) {
    // Get the original action
    const action = await getAgentAction(o.actionId);
    if (!action?.targetUserId) continue;
    
    // Get current score
    const currentScore = await getLatestCompositeScore(action.targetUserId);
    
    // Determine outcome
    let outcome;
    if (currentScore === null) outcome = 'no_data';
    else if (o.initialScore === null) outcome = 'no_baseline';
    else if (currentScore > o.initialScore + 5) outcome = 'improved';
    else if (currentScore < o.initialScore - 5) outcome = 'declined';
    else outcome = 'unchanged';
    
    // Save outcome
    await db.update(agentActionOutcomes)
      .set({ 
        followUpScore: currentScore, 
        outcome, 
        completedAt: new Date() 
      })
      .where(eq(agentActionOutcomes.id, o.id));
    
    // Notify the original approver
    await createNotification({
      userId: action.approvedBy,
      title: outcome === 'improved' 
        ? `✅ Öneri Başarılı: ${action.metadata?.personName}`
        : `⚠️ Öneri Sonucu: ${action.metadata?.personName}`,
      message: outcome === 'improved'
        ? `Skor ${o.initialScore} → ${currentScore} yükseldi. Uygulanan strateji etkili oldu.`
        : outcome === 'declined'
        ? `Skor ${o.initialScore} → ${currentScore} düştü. Farklı yaklaşım gerekebilir.`
        : `Skor değişmedi (${currentScore}/100). Durumu izlemeye devam edin.`,
      type: 'agent_outcome'
    });
  }
}
```

Register to run daily at 08:00.
