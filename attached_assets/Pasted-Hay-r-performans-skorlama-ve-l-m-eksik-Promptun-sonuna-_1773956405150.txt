### STEP 16: PRODUCTION EFFICIENCY METRICS
```bash
echo "═══ STEP 16: Production Efficiency ═══"

# Station-level efficiency
echo "--- Production per station (last 30 days) ---"
echo "SELECT fs.name as station, count(pr.id) as runs, sum(pr.quantity) as total_qty, 
  avg(EXTRACT(EPOCH FROM (pr.ended_at - pr.started_at))/60) as avg_minutes
FROM factory_production_runs pr
JOIN factory_stations fs ON pr.station_id = fs.id
WHERE pr.started_at > NOW() - INTERVAL '30 days'
GROUP BY fs.name
ORDER BY total_qty DESC;" | psql 2>/dev/null

# Daily production trends
echo "--- Daily production (last 7 days) ---"
echo "SELECT DATE(started_at) as day, count(*) as runs, sum(quantity) as total_qty
FROM factory_production_runs
WHERE started_at > NOW() - INTERVAL '7 days'
GROUP BY DATE(started_at)
ORDER BY day DESC;" | psql 2>/dev/null

# Product-level stats
echo "--- Production by product (last 30 days) ---"
echo "SELECT fp.name as product, count(pr.id) as runs, sum(pr.quantity) as total_qty
FROM factory_production_runs pr
JOIN factory_products fp ON pr.product_id = fp.id
WHERE pr.started_at > NOW() - INTERVAL '30 days'
GROUP BY fp.name
ORDER BY total_qty DESC
LIMIT 10;" | psql 2>/dev/null
```
