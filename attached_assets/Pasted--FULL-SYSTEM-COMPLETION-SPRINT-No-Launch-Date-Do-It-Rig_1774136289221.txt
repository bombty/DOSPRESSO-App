```typescript
try {
  // Generate lot number: LOT-YYYYMMDD-XXX
  const dateStr = new Date().toISOString().split('T')[0].replace(/-/g, '');
  const prefix = `LOT-${dateStr}-`;
  const existing = await db.select({ lotNumber: productionLots.lotNumber })
    .from(productionLots)
    .where(like(productionLots.lotNumber, `${prefix}%`))
    .orderBy(desc(productionLots.lotNumber))
    .limit(1);
  const nextNum = existing.length > 0 ? parseInt(existing[0].lotNumber.split('-').pop() || '0') + 1 : 1;
  const lotNumber = `${prefix}${nextNum.toString().padStart(3, '0')}`;
  
  await db.insert(productionLots).values({
    lotNumber,
    productionDate: new Date().toISOString().split('T')[0],
    productId: output.productId,
```

```typescript
export interface PayrollInput {

  performanceBonus: number;

export interface PayrollResult {
  baseSalary: number;
  overtimePay: number;
  offDayPay: number;

  totalBonuses: number;

  unemploymentEmployer: number;

}

export async function calculatePayroll(db: any, input: PayrollInput): Promise<PayrollResult> {
  // 1. Get active payroll parameters from DB (2026)
  // 2. Calculate: daily = gross / 30, hourly = daily / 7.5
  // 3. Overtime: hours × hourly × 1.5
  // 4. OFF day: hours × hourly × 1.5
  // 5. Holiday: hours × hourly × 2.0
  // 6. Deficit deduction: absent days × daily
  // 7. Gross total = base + overtime + off + holiday + bonuses - deficit
  // 8. SGK employee = gross × rate (check ceiling)
  // 9. Unemployment = gross × rate
  // 10. Taxable = gross - SGK - unemployment
  // 11. Income tax = cumulative bracket calculation
  // 12. Stamp tax = gross × 0.00759
  // 13. AGI (simplified or from params)
  // 14. Net = gross - all deductions + AGI
  // 15. Employer: SGK employer + unemployment employer
  
  // ALL RATES from payroll_parameters table — decode integer values:
  // sgk_worker_rate 140 → 0.14 (divide by 1000)
  // income_tax_bracket1 in kuruş → TL (divide by 100)
  // stamp_tax_rate 759 → 0.00759 (divide by 100000)
}
```

```typescript
import { PDFDocument, rgb, StandardFonts } from 'pdf-lib';

export async function generatePayrollPDF(data: any): Promise<Buffer> {
  const doc = await PDFDocument.create();
  const font = await doc.embedFont(StandardFonts.Helvetica);
  const boldFont = await doc.embedFont(StandardFonts.HelveticaBold);
  
  // Page 1: Summary
  let page = doc.addPage([595, 842]); // A4
  const { width, height } = page.getSize();
  
  // Header
  page.drawText('DOSPRESSO — Bordro Raporu', { x: 50, y: height - 50, size: 16, font: boldFont });
  page.drawText(`${data.month}/${data.year}`, { x: 50, y: height - 70, size: 12, font });
  
  // Summary table
  let y = height - 110;
  page.drawText('Toplam Çalışan: ' + data.employees.length, { x: 50, y, size: 10, font });
  y -= 20;
  page.drawText('Toplam Brüt: ' + formatMoney(data.totalGross), { x: 50, y, size: 10, font });
  y -= 20;
  page.drawText('Toplam Net: ' + formatMoney(data.totalNet), { x: 50, y, size: 10, font });
  
  // Per employee pages
  for (const emp of data.employees) {
    page = doc.addPage([595, 842]);
    y = height - 50;
    page.drawText(`${emp.firstName} ${emp.lastName}`, { x: 50, y, size: 14, font: boldFont });
    y -= 20;
    page.drawText(`${emp.position} — ${emp.branch}`, { x: 50, y, size: 10, font });
    y -= 30;
    
    // Earnings
    page.drawText('KAZANÇLAR', { x: 50, y, size: 10, font: boldFont });
    y -= 18;
    page.drawText(`Taban Maaş: ${formatMoney(emp.baseSalary)}`, { x: 60, y, size: 9, font });
    y -= 15;
    page.drawText(`Fazla Mesai: ${formatMoney(emp.overtimePay)}`, { x: 60, y, size: 9, font });
    y -= 15;
    page.drawText(`Primler: ${formatMoney(emp.totalBonuses)}`, { x: 60, y, size: 9, font });
    y -= 15;
    page.drawText(`BRÜT TOPLAM: ${formatMoney(emp.grossTotal)}`, { x: 60, y, size: 10, font: boldFont });
    y -= 25;
    
    // Deductions
    page.drawText('KESİNTİLER', { x: 50, y, size: 10, font: boldFont });
    y -= 18;
    page.drawText(`SGK İşçi (%14): ${formatMoney(emp.sgkEmployee)}`, { x: 60, y, size: 9, font });
    y -= 15;
    page.drawText(`İşsizlik (%1): ${formatMoney(emp.unemploymentEmployee)}`, { x: 60, y, size: 9, font });
    y -= 15;
    page.drawText(`Gelir Vergisi: ${formatMoney(emp.incomeTax)}`, { x: 60, y, size: 9, font });
    y -= 15;
    page.drawText(`Damga Vergisi: ${formatMoney(emp.stampTax)}`, { x: 60, y, size: 9, font });
    y -= 15;
    page.drawText(`TOPLAM KESİNTİ: ${formatMoney(emp.totalDeductions)}`, { x: 60, y, size: 10, font: boldFont });
    y -= 30;
    
    // Net
    page.drawText(`NET MAAŞ: ${formatMoney(emp.netSalary)} TL`, { x: 50, y, size: 14, font: boldFont });
  }
  
  return Buffer.from(await doc.save());
}

function formatMoney(amount: number): string {
  return new Intl.NumberFormat('tr-TR', { minimumFractionDigits: 2, maximumFractionDigits: 2 }).format(amount);
}
```
