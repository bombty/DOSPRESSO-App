### T001 — HTTP Güvenlik Header'ları (~1 saat)

**Problem:** Mevcut sistemde CSP, HSTS, X-Frame-Options gibi güvenlik header'ları eksik olabilir.

**Araştırma:**
```bash
# Mevcut güvenlik header'ları
grep -rn "helmet\|csp\|Content-Security-Policy\|X-Frame\|X-Content-Type\|Strict-Transport\|HSTS" server/ --include="*.ts" | head -15

# Express middleware sırası
grep -rn "app.use" server/index.ts | head -30
```

**Yapılacak:**

1. `helmet` paketi yoksa ekle (veya manuel header'lar):
```typescript
// Güvenlik header'ları
app.use((req, res, next) => {
  // XSS koruması
  res.setHeader('X-Content-Type-Options', 'nosniff');
  res.setHeader('X-XSS-Protection', '1; mode=block');
  
  // Clickjacking koruması
  res.setHeader('X-Frame-Options', 'DENY');
  
  // HTTPS zorlama (production'da)
  if (process.env.NODE_ENV === 'production') {
    res.setHeader('Strict-Transport-Security', 'max-age=31536000; includeSubDomains');
  }
  
  // Referrer politikası
  res.setHeader('Referrer-Policy', 'strict-origin-when-cross-origin');
  
  // Permissions Policy
  res.setHeader('Permissions-Policy', 'camera=(), microphone=(), geolocation=()');
  
  next();
});
```

2. CSP (Content Security Policy) — dikkatli tanımla, Vite HMR'ı ve mevcut CDN'leri kırma:
```typescript
// Development'ta gevşek, production'da sıkı
const cspPolicy = process.env.NODE_ENV === 'production'
  ? "default-src 'self'; script-src 'self'; style-src 'self' 'unsafe-inline'; img-src 'self' data: blob:; connect-src 'self' https://api.openai.com; font-src 'self';"
  : "default-src 'self' 'unsafe-inline' 'unsafe-eval'; connect-src *;";

res.setHeader('Content-Security-Policy', cspPolicy);
```

**ÖNEMLİ:** CSP eklerken mevcut fonksiyonelliği test et — Vite dev server, CDN scriptleri, OpenAI API çağrıları çalışmaya devam etmeli.

```typescript

async function logAudit(params: {
```

Mevcut service worker'a push event handler ekle:

```javascript
self.addEventListener('push', (event) => {
  const data = event.data?.json() || {};
  const options = {
    body: data.message || 'Yeni bildirim',
    icon: '/icon-192.png',
    badge: '/badge-72.png',
```

5. Push gönderme fonksiyonu:
```typescript
// server/lib/push-service.ts (yeni)
import webPush from 'web-push';

webPush.setVapidDetails(
  'mailto:info@dospresso.com',
  process.env.VAPID_PUBLIC_KEY!,
  process.env.VAPID_PRIVATE_KEY!
);

async function sendPushNotification(userId: string, payload: {
  title: string;
  message: string;
  link?: string;
  tag?: string;
}) {
  const subscriptions = await db.select()
    .from(pushSubscriptions)
    .where(and(
      eq(pushSubscriptions.userId, userId),
      eq(pushSubscriptions.isActive, true)
    ));
  
  for (const sub of subscriptions) {
    try {
      await webPush.sendNotification(
        { endpoint: sub.endpoint, keys: { p256dh: sub.p256dh, auth: sub.auth } },
        JSON.stringify(payload)
      );
    } catch (error: any) {
      if (error.statusCode === 410) {
        // Subscription expired — deaktif et
        await db.update(pushSubscriptions)
          .set({ isActive: false })
          .where(eq(pushSubscriptions.id, sub.id));
      }
    }
  }
}
```

### T007 — Frontend Push Subscription (~1-2 saat)

**Yapılacak:**

1. Push izni isteme component'i:
```typescript
// client/src/components/push-permission.tsx

function PushPermissionBanner() {
  const [showBanner, setShowBanner] = useState(false);
  
  useEffect(() => {
    // Service Worker destekliyor mu?
    if (!('serviceWorker' in navigator) || !('PushManager' in window)) return;
    
    // Zaten izin verilmiş mi?
    if (Notification.permission === 'granted') return;
    if (Notification.permission === 'denied') return;
    
    // İlk giriş sonrası 5 saniye bekle, sonra sor
    const timer = setTimeout(() => setShowBanner(true), 5000);
    return () => clearTimeout(timer);
  }, []);
  
  const handleAllow = async () => {
    const permission = await Notification.requestPermission();
    if (permission === 'granted') {
      await subscribeToPush();
    }
    setShowBanner(false);
  };
  
  if (!showBanner) return null;
  
  return (
    <div className="fixed bottom-20 left-4 right-4 bg-card border rounded-lg p-4 shadow-lg z-50">
      <p className="text-sm font-medium">🔔 Bildirimleri telefonunuza almak ister misiniz?</p>
      <p className="text-xs text-muted-foreground mt-1">Görev hatırlatmaları, eğitim bildirimleri ve önemli uyarılar</p>
      <div className="flex gap-2 mt-3">
        <Button size="sm" onClick={handleAllow}>İzin Ver</Button>
        <Button size="sm" variant="ghost" onClick={() => setShowBanner(false)}>Şimdi Değil</Button>
      </div>
    </div>
  );
}
```

2. Push subscription fonksiyonu:
```typescript
async function subscribeToPush() {
  const registration = await navigator.serviceWorker.ready;
  
  // VAPID key al
  const response = await fetch('/api/push/vapid-key');
  const { publicKey } = await response.json();
  
  const subscription = await registration.pushManager.subscribe({
    userVisibleOnly: true,
    applicationServerKey: urlBase64ToUint8Array(publicKey)
  });
  
  // Backend'e gönder
  await fetch('/api/push/subscribe', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      endpoint: subscription.endpoint,
      p256dh: btoa(String.fromCharCode(...new Uint8Array(subscription.getKey('p256dh')!))),
      auth: btoa(String.fromCharCode(...new Uint8Array(subscription.getKey('auth')!))),
      deviceInfo: navigator.userAgent.substring(0, 200)
    })
  });
}
```
