## 1. PDKS EXCEL IMPORT — DB TABLOLARI OLUŞTUR

```sql
-- 5 yeni tablo (schema-12.ts'e eklendi)

CREATE TABLE IF NOT EXISTS pdks_excel_imports (
  id SERIAL PRIMARY KEY,
  branch_id INTEGER NOT NULL REFERENCES branches(id),
  month INTEGER NOT NULL,
  year INTEGER NOT NULL,
  file_name TEXT NOT NULL,
  import_type TEXT NOT NULL DEFAULT 'historical',
  status TEXT DEFAULT 'processing',
  total_records INTEGER DEFAULT 0,
  matched_records INTEGER DEFAULT 0,
  unmatched_records INTEGER DEFAULT 0,
  warnings JSONB DEFAULT '[]',
  imported_by VARCHAR REFERENCES users(id),
  is_finalized BOOLEAN DEFAULT FALSE,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  finalized_at TIMESTAMPTZ,
  finalized_by VARCHAR REFERENCES users(id)
);

CREATE TABLE IF NOT EXISTS pdks_excel_records (
  id SERIAL PRIMARY KEY,
  import_id INTEGER NOT NULL REFERENCES pdks_excel_imports(id) ON DELETE CASCADE,
  source_row_no INTEGER,
  source_code TEXT,
  source_name TEXT,
  swipe_time TIMESTAMPTZ NOT NULL,
  matched_user_id VARCHAR REFERENCES users(id),
  match_method TEXT,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS pdks_daily_summary (
  id SERIAL PRIMARY KEY,
  import_id INTEGER NOT NULL REFERENCES pdks_excel_imports(id) ON DELETE CASCADE,
  user_id VARCHAR NOT NULL REFERENCES users(id),
  branch_id INTEGER NOT NULL REFERENCES branches(id),
  work_date TIMESTAMPTZ NOT NULL,
  first_swipe TIMESTAMPTZ,
  last_swipe TIMESTAMPTZ,
  total_swipes INTEGER DEFAULT 0,
  gross_minutes INTEGER DEFAULT 0,
  break_minutes INTEGER DEFAULT 0,
  net_minutes INTEGER DEFAULT 0,
  overtime_minutes INTEGER DEFAULT 0,
  is_off_day BOOLEAN DEFAULT FALSE,
  is_holiday BOOLEAN DEFAULT FALSE,
  is_historical BOOLEAN DEFAULT FALSE,
  warnings JSONB DEFAULT '[]',
  UNIQUE(import_id, user_id, work_date)
);

CREATE TABLE IF NOT EXISTS pdks_monthly_stats (
  id SERIAL PRIMARY KEY,
  user_id VARCHAR NOT NULL REFERENCES users(id),
  branch_id INTEGER NOT NULL REFERENCES branches(id),
  month INTEGER NOT NULL,
  year INTEGER NOT NULL,
  total_work_days INTEGER DEFAULT 0,
  total_off_days INTEGER DEFAULT 0,
  total_absent_days INTEGER DEFAULT 0,
  avg_daily_minutes INTEGER DEFAULT 0,
  total_overtime_minutes INTEGER DEFAULT 0,
  total_late_count INTEGER DEFAULT 0,
  total_early_leave_count INTEGER DEFAULT 0,
  compliance_score INTEGER DEFAULT 0,
  is_historical BOOLEAN DEFAULT FALSE,
  source_import_id INTEGER REFERENCES pdks_excel_imports(id),
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  UNIQUE(user_id, branch_id, month, year)
);

CREATE TABLE IF NOT EXISTS pdks_employee_mappings (
  id SERIAL PRIMARY KEY,
  branch_id INTEGER NOT NULL REFERENCES branches(id),
  pdks_code TEXT NOT NULL,
  pdks_name TEXT NOT NULL,
  user_id VARCHAR NOT NULL REFERENCES users(id),
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  created_by VARCHAR REFERENCES users(id),
  UNIQUE(branch_id, pdks_code)
);
```
