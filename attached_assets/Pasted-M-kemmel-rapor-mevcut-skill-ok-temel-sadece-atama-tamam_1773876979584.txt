### STEP 2: Expand generateActions() function

The generateActions function must handle all new insight types and create appropriate actions:

```typescript
async generateActions(insights: SkillInsight[]): Promise<SkillAction[]> {
  const actions: SkillAction[] = [];
  
  for (const insight of insights) {
    switch (insight.type) {
      // Existing types
      case "overall_completion_rate":
      case "low_completion_modules":
      case "high_completion_modules":
        // Keep existing action generation
        break;
        
      // New types
      case "hardest_quiz_questions":
        actions.push({
          actionType: "report",
          title: "Quiz Soru Analizi",
          description: insight.description,
          deepLink: "/akademi-hq",
          severity: insight.severity,
          category: "training",
          subcategory: "quiz_analysis",
          targetRoleScope: ["trainer", "coach"],
        });
        break;
        
      case "quiz_score_trends":
        actions.push({
          actionType: "report",
          title: "Quiz Puan Trendi",
          description: insight.description,
          deepLink: "/akademi-hq",
          severity: insight.severity,
          category: "training",
          subcategory: "quiz_trends",
          targetRoleScope: ["trainer", "coach"],
        });
        break;
        
      case "branch_training_comparison":
        actions.push({
          actionType: "report",
          title: "Şube Eğitim Karşılaştırması",
          description: insight.description,
          deepLink: "/akademi-hq",
          severity: insight.severity,
          category: "training",
          subcategory: "branch_comparison",
          targetRoleScope: ["coach", "ceo", "cgo"],
        });
        break;
        
      case "personal_training_recommendations":
        // Generate individual actions for each employee
        if (insight.metadata?.employees) {
          for (const emp of insight.metadata.employees) {
            actions.push({
              actionType: "task",
              title: `Eğitim Takibi: ${emp.name}`,
              description: emp.recommendation,
              deepLink: `/akademi-hq`,
              severity: "medium",
              category: "training",
              subcategory: "personal_recommendation",
              targetRoleScope: ["coach", "trainer"],
              branchId: emp.branchId,
              metadata: { employeeId: emp.userId, employeeName: emp.name },
            });
          }
        }
        break;
        
      case "onboarding_status":
        actions.push({
          actionType: "report",
          title: "Onboarding Durumu",
          description: insight.description,
          deepLink: "/akademi-hq",
          severity: "high",
          category: "training",
          subcategory: "onboarding",
          targetRoleScope: ["coach", "trainer"],
        });
        break;
        
      case "usage_report":
        actions.push({
          actionType: "report",
          title: "Haftalık Eğitim Kullanım Raporu",
          description: insight.description,
          deepLink: "/akademi-hq",
          severity: "low",
          category: "training",
          subcategory: "usage",
          targetRoleScope: ["coach", "trainer", "ceo", "cgo"],
        });
        break;
        
      case "certification_pipeline":
        actions.push({
          actionType: "report",
          title: "Sertifika ve Sınav Durumu",
          description: insight.description,
          deepLink: "/akademi-hq",
          severity: insight.severity,
          category: "training",
          subcategory: "certification",
          targetRoleScope: ["trainer", "coach"],
        });
        break;
        
      case "quiz_gap_detection":
        actions.push({
          actionType: "report",
          title: "Quiz Eksiklik Tespiti",
          description: insight.description,
          deepLink: "/akademi-hq",
          severity: "medium",
          category: "training",
          subcategory: "quiz_gaps",
          targetRoleScope: ["trainer"],
        });
        break;
    }
  }
  
  return actions;
}
```
