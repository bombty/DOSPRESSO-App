### Step 7: Submit QC inspection

```typescript
async function submitQCInspection(overallResult: 'approved' | 'conditional' | 'rejected') {
  try {
    await kioskFetch('/api/factory/qc/inspections', {
      method: 'POST',
      body: JSON.stringify({
        lotId: selectedLot.id,
        ...qcChecks,
        temperatureValue: qcChecks.temperatureValue ? parseFloat(qcChecks.temperatureValue) : null,
        overallResult,
      }),
    });
    
    setStep('qc-result');
    setQcResult(overallResult);
  } catch (err) {
    console.error("[QC] Inspection submit failed:", err);
    // Show error toast
  }
}
```
