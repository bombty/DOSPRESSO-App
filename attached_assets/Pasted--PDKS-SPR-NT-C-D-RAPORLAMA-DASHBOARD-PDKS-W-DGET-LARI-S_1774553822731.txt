## C1.1: /api/pdks/dashboard-summary endpoint oluştur

Tüm dashboard widget'larının kullanacağı tek birleştirilmiş PDKS özet endpoint'i.
Rol ve branchId'ye göre farklı veri döndürür:

```typescript
// GET /api/pdks/dashboard-summary?scope=branch&branchId=5
// GET /api/pdks/dashboard-summary?scope=hq  (HQ rolleri için — sadece 5,23,24)
// GET /api/pdks/dashboard-summary?scope=all  (CEO/Coach için — tüm şubeler)

router.get('/pdks/dashboard-summary', requireAuth, async (req, res) => {
  const user = req.user;
  const scope = req.query.scope || 'branch';
  const branchId = req.query.branchId ? parseInt(req.query.branchId) : user.branchId;
  
  const today = getIstanbulDate(); // Sprint B'deki timezone helper
  const monthStart = today.substring(0, 7) + '-01';
  
  let branchIds: number[];
  if (scope === 'hq') {
    branchIds = [5, 23, 24]; // Merkez + Fabrika + Işıklar
  } else if (scope === 'all') {
    branchIds = await getAllActiveBranchIds();
  } else {
    branchIds = [branchId];
  }
  
  // 1. Bu ay PDKS özeti
  const monthlyRecords = await db.select({
    branchId: pdksRecords.branchId,
    recordType: pdksRecords.recordType,
    count: sql<number>`count(*)`,
  })
  .from(pdksRecords)
  .where(and(
    inArray(pdksRecords.branchId, branchIds),
    gte(pdksRecords.recordDate, monthStart),
  ))
  .groupBy(pdksRecords.branchId, pdksRecords.recordType);
  
  // 2. Bugün gelen/gelmeyen
  const todayPresent = await db.select({
    branchId: pdksRecords.branchId,
    count: sql<number>`count(DISTINCT ${pdksRecords.userId})`,
  })
  .from(pdksRecords)
  .where(and(
    inArray(pdksRecords.branchId, branchIds),
    eq(pdksRecords.recordDate, today),
    eq(pdksRecords.recordType, 'giris'),
  ))
  .groupBy(pdksRecords.branchId);
  
  // 3. Geç kalma sayısı (bu ay)
  const lateCount = await db.select({
    branchId: attendancePenalties.branchId,
    count: sql<number>`count(*)`,
    totalMinutes: sql<number>`sum(${attendancePenalties.lateMinutes})`,
  })
  .from(attendancePenalties)
  .where(and(
    inArray(attendancePenalties.branchId, branchIds),
    gte(attendancePenalties.penaltyDate, monthStart),
    eq(attendancePenalties.penaltyType, 'late_arrival'),
  ))
  .groupBy(attendancePenalties.branchId);
  
  // 4. Haftalık uyum (en son hafta)
  const weeklyCompliance = await db.select()
    .from(branchWeeklyAttendanceSummary)
    .where(inArray(branchWeeklyAttendanceSummary.branchId, branchIds))
    .orderBy(desc(branchWeeklyAttendanceSummary.weekStartDate))
    .limit(branchIds.length);
  
  // 5. Scheduled offs (bu ay)
  const offDays = await db.select({
    count: sql<number>`count(DISTINCT ${scheduledOffs.userId} || '-' || ${scheduledOffs.offDate})`,
  })
  .from(scheduledOffs)
  .where(and(
    gte(scheduledOffs.offDate, monthStart),
    lte(scheduledOffs.offDate, today),
  ));
  
  // 6. Bordro durumu (geçen ay)
  const lastMonth = new Date();
  lastMonth.setMonth(lastMonth.getMonth() - 1);
  const payrollCount = await db.select({
    count: sql<number>`count(*)`,
  })
  .from(monthlyPayroll)
  .where(and(
    eq(monthlyPayroll.month, lastMonth.getMonth() + 1),
    eq(monthlyPayroll.year, lastMonth.getFullYear()),
  ));
  
  res.json({
    scope,
    branchIds,
    today: {
      present: todayPresent.reduce((sum, r) => sum + r.count, 0),
      // totalStaff ayrı sorgu ile bulunabilir
    },
    thisMonth: {
      totalEntries: monthlyRecords.filter(r => r.recordType === 'giris').reduce((sum, r) => sum + r.count, 0),
      totalExits: monthlyRecords.filter(r => r.recordType === 'cikis').reduce((sum, r) => sum + r.count, 0),
      lateArrivals: lateCount.reduce((sum, r) => sum + r.count, 0),
      totalLateMinutes: lateCount.reduce((sum, r) => sum + (r.totalMinutes || 0), 0),
      scheduledOffs: offDays[0]?.count || 0,
    },
    weeklyCompliance: weeklyCompliance.map(w => ({
      branchId: w.branchId,
      plannedMinutes: w.plannedTotalMinutes,
      actualMinutes: w.actualTotalMinutes,
      complianceScore: w.weeklyComplianceScore,
    })),
    payroll: {
      lastMonthCalculated: payrollCount[0]?.count || 0,
    },
  });
});
```

## C1.2: /api/pdks/branch-attendance endpoint (şube bazlı puantaj)

Supervisor ve Yatırımcı için şube bazlı detaylı puantaj:

```typescript
// GET /api/pdks/branch-attendance?branchId=5&month=3&year=2026
// Personel bazlı: kaç gün çalıştı, kaç gün izinli, kaç gün devamsız, geç kalma

router.get('/pdks/branch-attendance', requireAuth, async (req, res) => {
  const { branchId, month, year } = req.query;
  
  // Bu şubedeki aktif personel
  const staff = await db.select()
    .from(users)
    .where(and(
      eq(users.branchId, parseInt(branchId)),
      eq(users.isActive, true),
    ));
  
  const results = [];
  for (const person of staff) {
    // PDKS motorunu kullan (mevcut fonksiyon)
    const classification = await getMonthlyClassification(person.id, parseInt(month), parseInt(year));
    
    // Geç kalma sayısı
    const penalties = await db.select()
      .from(attendancePenalties)
      .where(and(
        eq(attendancePenalties.userId, person.id),
        sql`EXTRACT(MONTH FROM ${attendancePenalties.penaltyDate}) = ${month}`,
        sql`EXTRACT(YEAR FROM ${attendancePenalties.penaltyDate}) = ${year}`,
      ));
    
    results.push({
      userId: person.id,
      name: `${person.firstName} ${person.lastName}`,
      role: person.role,
      employmentType: person.employmentType,
      workedDays: classification?.workedDays || 0,
      offDays: classification?.offDays || 0,
      absentDays: classification?.absentDays || 0,
      lateArrivals: penalties.filter(p => p.penaltyType === 'late_arrival').length,
      totalLateMinutes: penalties.reduce((sum, p) => sum + (p.lateMinutes || 0), 0),
      overtimeMinutes: classification?.totalOvertimeMinutes || 0,
    });
  }
  
  res.json({
    branchId: parseInt(branchId),
    month: parseInt(month),
    year: parseInt(year),
    staffCount: staff.length,
    summary: {
      totalWorkedDays: results.reduce((s, r) => s + r.workedDays, 0),
      totalAbsentDays: results.reduce((s, r) => s + r.absentDays, 0),
      totalLateArrivals: results.reduce((s, r) => s + r.lateArrivals, 0),
      totalOvertimeMinutes: results.reduce((s, r) => s + r.overtimeMinutes, 0),
    },
    staff: results,
  });
});
```
