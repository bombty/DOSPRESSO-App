Fix 1: Transaction Guard (Core Fix — 15 dk)

// YENİ: Transaction içine al
try {
  await db.transaction(async (tx) => {
    await tx.insert(pdksRecords).values({...});
    await tx.insert(shiftAttendance).values({...});  // Eğer PDKS fail ederse shift_attendance da rollback
  });
} catch (err) {
  console.error("KIOSK check-in atomic fail:", err);
  throw err;  // Kullanıcıya hata göster — SESSIZ YUTMA YOK
}

INSERT INTO pdks_records (user_id, branch_id, record_date, record_time, record_type, source, created_at)

  sa.user_id,
  s.branch_id,
  s.shift_date,
  TO_CHAR(sa.check_in_time AT TIME ZONE 'Europe/Istanbul', 'HH24:MI:SS')::time,
  'giris',
  'kiosk_backfill',
  NOW()
FROM shift_attendance sa
INNER JOIN shifts s ON sa.shift_id = s.id
WHERE sa.user_id IN (
  '[basri_sen_id]',
  '[busra_dogmus_id]'
)
AND s.shift_date IN ('2026-03-21', '2026-03-29', '2026-04-02')
AND sa.check_in_time IS NOT NULL;
