function LiveClock() {

}

### PIN handler — auto-submit at 4 digits:
```typescript
const [pin, setPin] = useState('');
const [pinError, setPinError] = useState('');

function handlePinDigit(digit: string) {
  if (pin.length >= 4) return;
  const newPin = pin + digit;
  setPin(newPin);
  if (newPin.length === 4) {
    handlePinSubmit(newPin);
  }
}

async function handlePinSubmit(pinCode: string) {
  try {
    setPinError('');
    const res = await kioskFetch('/api/factory/kiosk/login-by-username', {
      method: 'POST',
      body: JSON.stringify({ username: selectedWorker.username, pin: pinCode }),
    });
    const data = await res.json();
    
    if (!res.ok) {
      setPinError(data.error || 'PIN hatalı');
      setPin('');
      return;
    }
    
    // Store token
    if (data.kioskToken) {
      localStorage.setItem("factory-kiosk-token", data.kioskToken);
    }
    
    // Auto-start shift (no station required)
    try {
      const shiftRes = await kioskFetch('/api/factory/kiosk/start-shift', {
        method: 'POST',
        body: JSON.stringify({}),
      });
      const shiftData = await shiftRes.json();
      if (shiftData.sessionId) {
        setActiveSessionId(shiftData.sessionId);
      }
      if (shiftData.existingSession || shiftData.resumed) {
        // Resuming existing shift
        setActiveSessionId(shiftData.sessionId || shiftData.existingSession?.id);
        if (shiftData.existingSession?.stationId) {
          const st = (stations || []).find((s: any) => s.id === shiftData.existingSession.stationId);
          if (st) setCurrentStation(st);
        }
      }
    } catch (e) {
      console.error('[Kiosk] Auto shift start:', e);
    }
    
    setShiftStartTime(new Date().toLocaleTimeString('tr-TR', {hour:'2-digit', minute:'2-digit'}));
    setPin('');
    setPinError('');
    setStep('worker-home');
  } catch (err) {
    setPinError('Bağlantı hatası');
    setPin('');
  }
}
```

## FIX 5: Inactivity Auto-Return (5 seconds)

On worker-home and end-shift-summary, if no touch for 5 seconds, return to select-user:

```typescript
const inactivityRef = useRef<NodeJS.Timeout | null>(null);
const INACTIVITY_MS = 5000; // 5 seconds
const AUTO_RETURN_STEPS: KioskStep[] = ['worker-home', 'end-shift-summary'];

useEffect(() => {
  function resetTimer() {
    if (inactivityRef.current) clearTimeout(inactivityRef.current);
    if (AUTO_RETURN_STEPS.includes(step) && selectedWorker) {
      inactivityRef.current = setTimeout(() => resetWorker(), INACTIVITY_MS);
    }
  }
  
  const events = ['touchstart', 'mousedown', 'keydown'];
  events.forEach(e => document.addEventListener(e, resetTimer));
  resetTimer();
  
  return () => {
    if (inactivityRef.current) clearTimeout(inactivityRef.current);
    events.forEach(e => document.removeEventListener(e, resetTimer));
  };
}, [step, selectedWorker]);
```

IMPORTANT: Do NOT auto-return during working, on-break, select-station, production-entry, log-production, fault-report, stop-options, enter-pin.

## FIX 6: Break Flow — Simple and Reliable

```typescript
async function startBreak(reason: string) {
  try {
    const res = await kioskFetch('/api/factory/kiosk/log-break', {
      method: 'POST',
      body: JSON.stringify({
        eventType: reason,
        stationId: currentStation?.id || null,
        sessionId: activeSessionId,
      }),
    });
    const data = await res.json();
    if (data.breakLogId) {
      setCurrentBreakLogId(data.breakLogId);
    }
    setBreakReason(reason);
    setStep('on-break');
  } catch (err) {
    console.error('[Kiosk] Start break error:', err);
    // Go to break screen anyway
    setBreakReason(reason);
    setStep('on-break');
  }
}

function returnFromBreak() {
  // End break (fire-and-forget — don't block UI)
  if (currentBreakLogId) {
    kioskFetch('/api/factory/kiosk/end-break', {
      method: 'POST',
      body: JSON.stringify({ breakLogId: currentBreakLogId }),
    }).catch(err => console.error('[Kiosk] End break:', err));
  }
  setCurrentBreakLogId(null);
  setBreakReason(null);
  setStep('worker-home'); // Always return to worker-home
}
```
