# Modül-İçerik-Sprint — Subtopic Ekleme/Silme (Mega Modül Editörü)

## Context
DOSPRESSO franchise platform. Stack: React 18 + TypeScript + Tailwind, Shadcn/ui, TanStack Query v5, Drizzle ORM.

The existing "Mega Modül Düzenleme" page (`client/src/pages/admin/yetkilendirme.tsx`) 
already has drag-and-drop module sorting. This sprint adds subtopic (alt konu) 
management to each module item — without touching the existing drag-and-drop logic.

The subtopic system is for the delegation module's content map 
(showing which topics each module covers). It's stored separately 
from the existing flat module structure.

Do NOT touch: existing drag-and-drop logic, existing module/mega-module tables,
auth system, any non-admin pages.

## BACKEND — TASK 1: Schema

**File:** `shared/schema.ts`

Add two new tables:

```typescript
// Module departments (e.g. "Teknik Destek" inside CRM)
export const moduleDepartments = pgTable('module_departments', {
  id: serial('id').primaryKey(),
  moduleKey: varchar('module_key', { length: 100 }).notNull(),
  name: varchar('name', { length: 200 }).notNull(),
  icon: varchar('icon', { length: 10 }).default('📌'),
  sortOrder: integer('sort_order').default(0),
  createdAt: timestamp('created_at').defaultNow().notNull(),
});

// Topics within each department
export const moduleDepartmentTopics = pgTable('module_department_topics', {
  id: serial('id').primaryKey(),
  departmentId: integer('department_id')
    .notNull()
    .references(() => moduleDepartments.id, { onDelete: 'cascade' }),
  label: varchar('label', { length: 200 }).notNull(),
  sortOrder: integer('sort_order').default(0),
  createdAt: timestamp('created_at').defaultNow().notNull(),
});

export type ModuleDepartment = typeof moduleDepartments.$inferSelect;
export type ModuleDepartmentTopic = typeof moduleDepartmentTopics.$inferSelect;
```

Run `npm run db:push` after adding.

## BACKEND — TASK 2: API Routes

**Create:** `server/routes/module-content-routes.ts`

```typescript
import { Router } from 'express';
import { db } from '../db';
import { moduleDepartments, moduleDepartmentTopics } from '../../shared/schema';
import { eq, asc } from 'drizzle-orm';
import { requireAuth, requireRole } from '../middleware/auth'; // actual path

const router = Router();

// GET /api/module-content/:moduleKey — get full content map for a module
router.get('/:moduleKey', requireAuth, async (req, res) => {
  try {
    const { moduleKey } = req.params;
    const depts = await db
      .select()
      .from(moduleDepartments)
      .where(eq(moduleDepartments.moduleKey, moduleKey))
      .orderBy(asc(moduleDepartments.sortOrder));

    const result = await Promise.all(
      depts.map(async (dept) => {
        const topics = await db
          .select()
          .from(moduleDepartmentTopics)
          .where(eq(moduleDepartmentTopics.departmentId, dept.id))
          .orderBy(asc(moduleDepartmentTopics.sortOrder));
        return { ...dept, topics };
      })
    );

    res.json(result);
  } catch (err) {
    res.status(500).json({ error: 'Failed to fetch module content' });
  }
});

// POST /api/module-content/:moduleKey/departments — add department
router.post('/:moduleKey/departments', requireAuth, requireRole(['admin', 'ceo']), async (req, res) => {
  try {
    const { moduleKey } = req.params;
    const { name, icon } = req.body;
    if (!name) return res.status(400).json({ error: 'name required' });

    const [created] = await db.insert(moduleDepartments)
      .values({ moduleKey, name, icon: icon || '📌' })
      .returning();
    res.status(201).json(created);
  } catch (err) {
    res.status(500).json({ error: 'Failed to create department' });
  }
});

// DELETE /api/module-content/departments/:id — delete department (cascades topics)
router.delete('/departments/:id', requireAuth, requireRole(['admin', 'ceo']), async (req, res) => {
  try {
    await db.delete(moduleDepartments).where(eq(moduleDepartments.id, parseInt(req.params.id)));
    res.json({ success: true });
  } catch (err) {
    res.status(500).json({ error: 'Failed to delete department' });
  }
});

// POST /api/module-content/departments/:id/topics — add topic
router.post('/departments/:id/topics', requireAuth, requireRole(['admin', 'ceo']), async (req, res) => {
  try {
    const departmentId = parseInt(req.params.id);
    const { label } = req.body;
    if (!label) return res.status(400).json({ error: 'label required' });

    const [created] = await db.insert(moduleDepartmentTopics)
      .values({ departmentId, label })
      .returning();
    res.status(201).json(created);
  } catch (err) {
    res.status(500).json({ error: 'Failed to create topic' });
  }
});

// DELETE /api/module-content/topics/:id — delete topic
router.delete('/topics/:id', requireAuth, requireRole(['admin', 'ceo']), async (req, res) => {
  try {
    await db.delete(moduleDepartmentTopics)
      .where(eq(moduleDepartmentTopics.id, parseInt(req.params.id)));
    res.json({ success: true });
  } catch (err) {
    res.status(500).json({ error: 'Failed to delete topic' });
  }
});

export default router;
```

Register in main server file:
```typescript
import moduleContentRouter from './routes/module-content-routes';
app.use('/api/module-content', moduleContentRouter);
```

## FRONTEND — TASK 3: ModuleContentEditor Component

**Create:** `client/src/components/module-content-editor.tsx`

This is a self-contained accordion editor for one module's content.
It will be embedded inside the existing Mega Modül Düzenleme page.

```tsx
import { useState } from 'react';
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { ChevronDown, ChevronUp, Plus, Trash2, GripVertical } from 'lucide-react';
import { cn } from '@/lib/utils';

interface Topic { id: number; label: string; sortOrder: number; }
interface Department { id: number; name: string; icon: string; topics: Topic[]; }

interface Props {
  moduleKey: string;
  moduleName: string;
}

export function ModuleContentEditor({ moduleKey, moduleName }: Props) {
  const [isOpen, setIsOpen] = useState(false);
  const [newDeptName, setNewDeptName] = useState('');
  const [newTopics, setNewTopics] = useState<Record<number, string>>({});
  const qc = useQueryClient();

  const { data: departments = [], isLoading } = useQuery<Department[]>({
    queryKey: ['/api/module-content', moduleKey],
    queryFn: () => fetch(`/api/module-content/${moduleKey}`).then(r => r.json()),
    enabled: isOpen, // only fetch when expanded
  });

  const addDept = useMutation({
    mutationFn: (name: string) =>
      fetch(`/api/module-content/${moduleKey}/departments`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ name }),
      }).then(r => r.json()),
    onSuccess: () => {
      qc.invalidateQueries({ queryKey: ['/api/module-content', moduleKey] });
      setNewDeptName('');
    },
  });

  const deleteDept = useMutation({
    mutationFn: (id: number) =>
      fetch(`/api/module-content/departments/${id}`, { method: 'DELETE' }),
    onSuccess: () => qc.invalidateQueries({ queryKey: ['/api/module-content', moduleKey] }),
  });

  const addTopic = useMutation({
    mutationFn: ({ deptId, label }: { deptId: number; label: string }) =>
      fetch(`/api/module-content/departments/${deptId}/topics`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ label }),
      }).then(r => r.json()),
    onSuccess: (_, { deptId }) => {
      qc.invalidateQueries({ queryKey: ['/api/module-content', moduleKey] });
      setNewTopics(prev => ({ ...prev, [deptId]: '' }));
    },
  });

  const deleteTopic = useMutation({
    mutationFn: (id: number) =>
      fetch(`/api/module-content/topics/${id}`, { method: 'DELETE' }),
    onSuccess: () => qc.invalidateQueries({ queryKey: ['/api/module-content', moduleKey] }),
  });

  return (
    <div className="border border-border rounded-xl overflow-hidden mb-3">
      {/* Module header — click to expand */}
      <button
        onClick={() => setIsOpen(!isOpen)}
        className="w-full flex items-center gap-3 px-4 py-3 bg-card hover:bg-accent transition-colors text-left"
      >
        <div className="flex-1">
          <div className="text-[12px] font-bold text-foreground">{moduleName}</div>
          <div className="text-[9px] text-muted-foreground mt-0.5">
            {isOpen ? `${departments.length} departman` : 'İçerik haritasını düzenle'}
          </div>
        </div>
        {isOpen ? (
          <ChevronUp className="w-4 h-4 text-muted-foreground" />
        ) : (
          <ChevronDown className="w-4 h-4 text-muted-foreground" />
        )}
      </button>

      {/* Expanded content */}
      {isOpen && (
        <div className="px-4 py-3 bg-muted/20 border-t border-border">
          {isLoading ? (
            <div className="text-[11px] text-muted-foreground py-2">Yükleniyor...</div>
          ) : (
            <>
              {/* Department list */}
              {departments.map((dept) => (
                <div
                  key={dept.id}
                  className="mb-3 bg-card rounded-lg border border-border p-3"
                >
                  {/* Dept header */}
                  <div className="flex items-center gap-2 mb-2">
                    <GripVertical className="w-3.5 h-3.5 text-muted-foreground/40 cursor-grab" />
                    <span className="text-[13px]">{dept.icon}</span>
                    <span className="text-[11px] font-bold text-foreground flex-1">{dept.name}</span>
                    <button
                      onClick={() => deleteDept.mutate(dept.id)}
                      className="text-[8px] font-semibold px-2 py-1 rounded bg-red-50 border border-red-200 text-red-600 dark:bg-red-950 dark:border-red-800 dark:text-red-400 hover:bg-red-100 transition-colors"
                    >
                      Sil
                    </button>
                  </div>

                  {/* Topics */}
                  <div className="flex flex-wrap gap-1.5 mb-2">
                    {dept.topics.map((topic) => (
                      <span
                        key={topic.id}
                        className="inline-flex items-center gap-1 px-2 py-1 rounded-full text-[8.5px] font-medium bg-muted border border-border text-muted-foreground"
                      >
                        <GripVertical className="w-2.5 h-2.5 opacity-40 cursor-grab" />
                        {topic.label}
                        <button
                          onClick={() => deleteTopic.mutate(topic.id)}
                          className="text-muted-foreground/60 hover:text-red-500 transition-colors ml-0.5 font-bold"
                        >
                          ×
                        </button>
                      </span>
                    ))}

                    {/* Inline add topic */}
                    <div className="inline-flex items-center gap-1">
                      <input
                        value={newTopics[dept.id] ?? ''}
                        onChange={(e) => setNewTopics(prev => ({ ...prev, [dept.id]: e.target.value }))}
                        onKeyDown={(e) => {
                          if (e.key === 'Enter' && newTopics[dept.id]?.trim()) {
                            addTopic.mutate({ deptId: dept.id, label: newTopics[dept.id].trim() });
                          }
                        }}
                        placeholder="Konu ekle..."
                        className="text-[8.5px] px-2 py-1 rounded-full border border-dashed border-muted-foreground/30 bg-transparent text-muted-foreground outline-none w-24 focus:border-[#122549] focus:w-32 transition-all"
                      />
                      <button
                        onClick={() => {
                          const label = newTopics[dept.id]?.trim();
                          if (label) addTopic.mutate({ deptId: dept.id, label });
                        }}
                        disabled={!newTopics[dept.id]?.trim()}
                        className="w-5 h-5 rounded-full bg-[#122549] text-white flex items-center justify-center disabled:opacity-30 hover:bg-[#0e1e3a] transition-colors"
                      >
                        <Plus className="w-2.5 h-2.5" />
                      </button>
                    </div>
                  </div>
                </div>
              ))}

              {/* Add department */}
              <div className="flex items-center gap-2 mt-2">
                <input
                  value={newDeptName}
                  onChange={(e) => setNewDeptName(e.target.value)}
                  onKeyDown={(e) => {
                    if (e.key === 'Enter' && newDeptName.trim()) {
                      addDept.mutate(newDeptName.trim());
                    }
                  }}
                  placeholder="Yeni departman adı..."
                  className="flex-1 text-[10px] px-3 py-1.5 rounded-lg border border-dashed border-muted-foreground/30 bg-transparent text-foreground outline-none focus:border-[#122549]"
                />
                <button
                  onClick={() => { if (newDeptName.trim()) addDept.mutate(newDeptName.trim()); }}
                  disabled={!newDeptName.trim()}
                  className="flex items-center gap-1.5 text-[9px] font-semibold px-3 py-1.5 rounded-lg border border-dashed border-muted-foreground/30 text-muted-foreground hover:border-[#122549] hover:text-[#122549] transition-colors disabled:opacity-30"
                >
                  <Plus className="w-3 h-3" />
                  Departman Ekle
                </button>
              </div>
            </>
          )}
        </div>
      )}
    </div>
  );
}
```
