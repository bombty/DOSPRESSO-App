DB MİGRASYON:
ALTER TABLE announcements ADD COLUMN IF NOT EXISTS quiz_questions TEXT;
ALTER TABLE announcements ADD COLUMN IF NOT EXISTS quiz_pass_score INTEGER DEFAULT 80;
ALTER TABLE announcements ADD COLUMN IF NOT EXISTS quiz_required BOOLEAN DEFAULT false;
CREATE TABLE IF NOT EXISTS announcement_quiz_results (
id SERIAL PRIMARY KEY,
announcement_id INTEGER NOT NULL REFERENCES announcements(id) ON DELETE CASCADE,
user_id VARCHAR NOT NULL REFERENCES users(id) ON DELETE CASCADE,
score INTEGER NOT NULL,
passed BOOLEAN NOT NULL,
total_questions INTEGER NOT NULL,
correct_answers INTEGER NOT NULL,
answers TEXT,
attempt_number INTEGER DEFAULT 1,
attempted_at TIMESTAMP DEFAULT NOW()
);
CREATE INDEX IF NOT EXISTS aqr_announcement_idx ON announcement_quiz_results(announcement_id);
CREATE INDEX IF NOT EXISTS aqr_user_idx ON announcement_quiz_results(user_id);
