4. YENİ DB TABLOLARI OLUŞTUR (5 tablo — psql ile):

   CREATE TABLE IF NOT EXISTS dobody_scopes (
     id SERIAL PRIMARY KEY,
     role VARCHAR(50) NOT NULL UNIQUE,
     allowed_modules TEXT[],
     blocked_keywords TEXT[],
     branch_scope VARCHAR(20) NOT NULL DEFAULT 'own',
     max_detail_level VARCHAR(20) NOT NULL DEFAULT 'summary',
     created_at TIMESTAMP DEFAULT NOW()
   );

   CREATE TABLE IF NOT EXISTS dobody_proposals (
     id SERIAL PRIMARY KEY,
     workflow_type VARCHAR(30) NOT NULL,
     role_target VARCHAR(50) NOT NULL,
     user_id VARCHAR REFERENCES users(id),
     branch_id INTEGER REFERENCES branches(id),
     proposal_type VARCHAR(20) NOT NULL DEFAULT 'action',
     priority VARCHAR(20) NOT NULL DEFAULT 'onemli',
     title VARCHAR(300) NOT NULL,
     description TEXT,
     source_module VARCHAR(50),
     related_entity_type VARCHAR(50),
     related_entity_id INTEGER,
     suggested_action_type VARCHAR(50),
     suggested_action_data JSONB,
     status VARCHAR(20) NOT NULL DEFAULT 'pending',
     expires_at TIMESTAMP,
     approved_by VARCHAR REFERENCES users(id),
     approved_at TIMESTAMP,
     rejected_reason TEXT,
     created_at TIMESTAMP DEFAULT NOW()
   );
   CREATE INDEX IF NOT EXISTS dobody_prop_status_idx ON dobody_proposals(status);
   CREATE INDEX IF NOT EXISTS dobody_prop_role_idx ON dobody_proposals(role_target);

   CREATE TABLE IF NOT EXISTS dobody_events (
     id SERIAL PRIMARY KEY,
     event_type VARCHAR(50) NOT NULL,
     source_module VARCHAR(50) NOT NULL,
     entity_type VARCHAR(50),
     entity_id INTEGER,
     event_data JSONB,
     proposals_generated INTEGER DEFAULT 0,
     processed_at TIMESTAMP DEFAULT NOW()
   );

   CREATE TABLE IF NOT EXISTS dobody_learning (
     id SERIAL PRIMARY KEY,
     workflow_type VARCHAR(30) NOT NULL,
     proposal_id INTEGER NOT NULL REFERENCES dobody_proposals(id) ON DELETE CASCADE,
     outcome VARCHAR(20) NOT NULL,
     rejection_reason TEXT,
     result_positive BOOLEAN,
     confidence_delta NUMERIC(5,2),
     created_at TIMESTAMP DEFAULT NOW()
   );

   CREATE TABLE IF NOT EXISTS dobody_workflow_confidence (
     id SERIAL PRIMARY KEY,
     workflow_type VARCHAR(30) NOT NULL,
     role VARCHAR(50) NOT NULL,
     confidence_score NUMERIC(5,2) NOT NULL DEFAULT 50,
     total_proposals INTEGER NOT NULL DEFAULT 0,
     approved_count INTEGER NOT NULL DEFAULT 0,
     rejected_count INTEGER NOT NULL DEFAULT 0,
     auto_apply_enabled BOOLEAN NOT NULL DEFAULT false,
     updated_at TIMESTAMP DEFAULT NOW()
   );
