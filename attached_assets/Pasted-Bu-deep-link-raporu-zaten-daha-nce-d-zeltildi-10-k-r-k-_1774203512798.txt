**Araştır — değişiklik yapma:**

```bash
echo "=== 1. LAYOUT YAPISI ==="
find client/src -name "*layout*" -o -name "*Layout*" | sort
find client/src -name "*header*" -o -name "*Header*" | sort
find client/src -name "*sidebar*" -o -name "*nav-rail*" | sort

echo ""
echo "=== 2. ÇİFT HEADER NEDEN? ==="
grep -n "Header\|header\|Topbar\|breadcrumb" client/src/App.tsx | head -15
grep -n "Header\|header" client/src/components/layout/*.tsx 2>/dev/null | head -15

echo ""
echo "=== 3. SIDEBAR DESKTOP DAVRANIŞI ==="
grep -n "collapsed\|expanded\|hidden\|lg:\|md:\|width\|44px\|240px" client/src/components/nav-rail*.tsx client/src/components/sidebar*.tsx 2>/dev/null | head -20

echo ""
echo "=== 4. MİSSİON CONTROL TOGGLE NEREDE ==="
grep -rn "DashboardModeToggle\|LayoutToggle\|mission-control" client/src/components/ client/src/pages/ --include="*.tsx" | head -15

echo ""
echo "=== 5. DASHBOARD ROL MAPPING ==="
head -80 client/src/pages/dashboard.tsx

echo ""
echo "=== 6. COACH SAYFASI LAYOUT ==="
find client/src/pages -name "*coach*" -o -name "*kocluk*" | sort
grep -n "grid\|Grid\|flex\|col-span\|grid-cols" client/src/pages/*coach* client/src/pages/*kocluk* 2>/dev/null | head -15

echo ""
echo "=== 7. HAMBURGER vs SIDEBAR ==="
grep -n "hamburger\|isOpen\|menuOpen" client/src/components/hamburger-menu.tsx | head -10
grep -n "collapsed\|isOpen\|mobile" client/src/components/nav-rail*.tsx 2>/dev/null | head -10
```

1. Çift header neden oluyor
2. Desktop sidebar neden sadece ikon
3. Mission Control toggle nerede — admin nasıl bulacak
4. Her rolün dashboard mapping'i
5. İçerik layout önerisi (list → grid)
6. Çözüm önerisi + süre tahmini
