Test C: Tatil Mesai
sql-- Mart 2026'da tatil günlerinde çalışan personel
SELECT mp.user_id, u.first_name, u.last_name,
       mp.holiday_worked_days, mp.holiday_pay, mp.net_pay
FROM monthly_payroll mp
JOIN users u ON mp.user_id = u.id
WHERE mp.year = 2026 AND mp.month = 3 AND mp.holiday_worked_days > 0;
Test D: Yemek Bedeli (Stajyer)
sql-- Stajyer rolündeki personel meal_allowance kontrolü
SELECT mp.user_id, u.first_name, u.role, mp.position_code,
       mp.worked_days, mp.meal_allowance,
       mp.worked_days * 33000 as expected_meal
FROM monthly_payroll mp
JOIN users u ON mp.user_id = u.id
WHERE mp.year = 2026 AND mp.month = 3 AND u.role = 'stajyer';
-- meal_allowance = worked_days × 33000 olmalı
