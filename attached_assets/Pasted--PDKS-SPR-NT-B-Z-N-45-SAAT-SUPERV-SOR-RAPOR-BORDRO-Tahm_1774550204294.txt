## B1.2: Fabrika/Merkez için otomatik hafta sonu program_off

Fabrika ve merkez personeli hafta içi 5 gün çalışır — cumartesi/pazar otomatik izin olmalı.

```typescript
// Scheduler: Her ayın 1'inde → gelecek ay için fabrika+HQ personelinin
// cumartesi/pazar günlerini scheduled_offs tablosuna yaz

schedule('pdks-auto-schedule-offs', '0 1 1 * *', async () => {
  // Fabrika (branchId=24) + HQ (branchId=23) + Işıklar (branchId=5) personeli
  const hqBranchIds = [23, 24, 5]; // Merkeze ait 3 birim
  
  const staff = await db.select()
    .from(users)
    .where(and(
      inArray(users.branchId, hqBranchIds),
      eq(users.isActive, true),
      eq(users.employmentType, 'fulltime')
    ));
  
  // Gelecek ayın cumartesi/pazar günlerini bul
  const nextMonth = new Date();
  nextMonth.setMonth(nextMonth.getMonth() + 1);
  const year = nextMonth.getFullYear();
  const month = nextMonth.getMonth();
  const weekendDays = [];
  
  for (let d = 1; d <= new Date(year, month + 1, 0).getDate(); d++) {
    const date = new Date(year, month, d);
    if (date.getDay() === 0 || date.getDay() === 6) {
      weekendDays.push(date.toISOString().split('T')[0]);
    }
  }
  
  // Her personel için scheduled_offs oluştur
  for (const user of staff) {
    for (const day of weekendDays) {
      await db.insert(scheduledOffs).values({
        userId: user.id,
        offDate: day,
        offType: 'program_off',
        reason: 'Hafta sonu',
      }).onConflict(/* userId + offDate */ ).doNothing();
    }
  }
  
  console.log(`[PDKS] ${staff.length} personel × ${weekendDays.length} gün = otomatik izin oluşturuldu`);
});
```

## B2.1: Haftalık özet hesaplama scheduler

```typescript
// Her pazar gece 23:00 — geçen haftanın çalışma saatlerini hesapla
schedule('pdks-weekly-summary', '0 23 * * 0', async () => {
  // Geçen haftanın tarih aralığı (Pazartesi-Pazar)
  const today = new Date();
  const weekStart = new Date(today);
  weekStart.setDate(today.getDate() - 7); // Geçen Pazartesi
  const weekEnd = new Date(today);
  weekEnd.setDate(today.getDate() - 1); // Geçen Pazar
  
  // Tüm aktif personel
  const allStaff = await db.select().from(users).where(eq(users.isActive, true));
  
  for (const user of allStaff) {
    // Bu hafta PDKS kayıtlarından çalışma dakikasını hesapla
    const weekRecords = await db.select()
      .from(pdksRecords)
      .where(and(
        eq(pdksRecords.userId, user.id.toString()),
        gte(pdksRecords.recordDate, weekStart.toISOString().split('T')[0]),
        lte(pdksRecords.recordDate, weekEnd.toISOString().split('T')[0]),
      ));
    
    // Giriş-çıkış eşleştirme → toplam dakika
    const totalMinutes = calculateWeeklyMinutes(weekRecords);
    const plannedMinutes = (user.weeklyHours || 45) * 60; // 2700dk
    const overtimeMinutes = Math.max(0, totalMinutes - plannedMinutes);
    const missingMinutes = Math.max(0, plannedMinutes - totalMinutes);
    
    await db.insert(branchWeeklyAttendanceSummary).values({
      userId: user.id,
      branchId: user.branchId,
      weekStartDate: weekStart.toISOString().split('T')[0],
      weekEndDate: weekEnd.toISOString().split('T')[0],
      plannedTotalMinutes: plannedMinutes,
      actualTotalMinutes: totalMinutes,
      overtimeMinutes,
      missingMinutes,
      weeklyComplianceScore: Math.min(100, Math.round((totalMinutes / plannedMinutes) * 100)),
    }).onConflictDoNothing();
  }
});
```

```typescript
schedule('pdks-weekly-notification', '30 8 * * 1', async () => {

  const deficits = await db.select()
    .from(branchWeeklyAttendanceSummary)
    .where(and(
```

```typescript
// Her ayın 1'i saat 04:00 — geçen ayın bordrosunu hesapla
schedule('pdks-monthly-payroll', '0 4 1 * *', async () => {
  const now = new Date();
  const lastMonth = now.getMonth() === 0 ? 12 : now.getMonth();
  const year = now.getMonth() === 0 ? now.getFullYear() - 1 : now.getFullYear();
  
  // Mevcut bordro hesaplama endpoint'ini programatik çağır
  // VEYA payroll-engine.ts'teki fonksiyonu doğrudan kullan
  
  // HQ kapsamı: Sadece branchId IN (23, 24, 5) — Merkez + Fabrika + Işıklar
  const hqBranchIds = [23, 24, 5];
  const hqStaff = await db.select()
    .from(users)
    .where(and(
      inArray(users.branchId, hqBranchIds),
      eq(users.isActive, true),
    ));
  
  let calculated = 0;
  for (const user of hqStaff) {
    try {
      // Mevcut calculatePayroll fonksiyonunu çağır
      await calculateAndSavePayroll(user.id, lastMonth, year);
      calculated++;
    } catch (e) {
      console.error(`Bordro hatası: ${user.username}`, e);
    }
  }
  
  // Muhasebe'ye bildirim
  const muhasebeUsers = await db.select()
    .from(users)
    .where(and(
      inArray(users.role, ['muhasebe_ik', 'muhasebe']),
      eq(users.isActive, true),
    ));
  
  for (const muh of muhasebeUsers) {
    await createNotification({
      userId: muh.id,
      type: 'payroll_ready',
      title: `${lastMonth}/${year} Bordro Hazır`,
      message: `${calculated} personelin bordrosu hesaplandı. Kontrol edin.`,
      priority: 'high',
      category: 'finance',
    });
  }
});
```
