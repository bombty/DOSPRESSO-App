DB MİGRASYON (yeni tablolar):
CREATE TABLE IF NOT EXISTS announcement_dismissals (
id SERIAL PRIMARY KEY,
announcement_id INTEGER NOT NULL REFERENCES announcements(id) ON DELETE CASCADE,
user_id VARCHAR NOT NULL REFERENCES users(id) ON DELETE CASCADE,
dismissed_at TIMESTAMP DEFAULT NOW(),
show_again_after TIMESTAMP,
UNIQUE(announcement_id, user_id)
);
CREATE INDEX IF NOT EXISTS dismissal_user_idx ON announcement_dismissals(user_id);
CREATE TABLE IF NOT EXISTS personnel_audit_scores (
id SERIAL PRIMARY KEY,
user_id VARCHAR NOT NULL REFERENCES users(id) ON DELETE CASCADE,
branch_id INTEGER REFERENCES branches(id),
period_type VARCHAR(20) NOT NULL,
period_start DATE NOT NULL,
period_end DATE NOT NULL,
audit_count INTEGER DEFAULT 0,
guler_yuz_avg INTEGER,
urun_bilgisi_avg INTEGER,
dress_code_avg INTEGER,
hijyen_avg INTEGER,
takim_ruhu_avg INTEGER,
gelisim_avg INTEGER,
overall_score INTEGER,
previous_score INTEGER,
trend VARCHAR(10),
created_at TIMESTAMP DEFAULT NOW(),
updated_at TIMESTAMP DEFAULT NOW()
);
CREATE TABLE IF NOT EXISTS audit_personnel_feedback (
id SERIAL PRIMARY KEY,
audit_instance_id INTEGER NOT NULL,
personnel_id VARCHAR NOT NULL REFERENCES users(id) ON DELETE CASCADE,
auditor_id VARCHAR NOT NULL REFERENCES users(id) ON DELETE CASCADE,
feedback TEXT NOT NULL,
category VARCHAR(30),
severity VARCHAR(10) DEFAULT 'info',
is_read_by_personnel BOOLEAN DEFAULT false,
read_at TIMESTAMP,
personnel_response TEXT,
responded_at TIMESTAMP,
created_at TIMESTAMP DEFAULT NOW()
);
