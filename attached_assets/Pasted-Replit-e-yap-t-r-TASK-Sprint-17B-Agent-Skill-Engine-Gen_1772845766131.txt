### T001 — Skill Registry & Tanımlama (~2 saat)

**Problem:** Mevcut agent engine tek bir monolitik analiz fonksiyonu çalıştırıyor. Skill bazlı modüler yapıya geçilmeli.

**Yapılacak:**

1. Skill tanımlama yapısı:

```typescript
// server/agent/skills/skill-registry.ts (yeni)

interface AgentSkill {
  id: string;
  name: string;
  description: string;
  targetRoles: string[];
  schedule: 'realtime' | 'hourly' | 'daily' | 'weekly';
  autonomyLevel: 'full_auto' | 'suggest_approve' | 'info_only';
  dataSources: string[];
  
  // Kural bazlı analiz (her zaman çalışır — bedava)
  analyze: (context: SkillContext) => Promise<SkillInsight[]>;
  
  // AI zenginleştirme (opsiyonel — sadece insight varsa)
  aiEnrich?: (insights: SkillInsight[]) => Promise<EnrichedInsight[]>;
  
  // Aksiyon üretme
  generateActions: (insights: SkillInsight[]) => PendingAction[];
}

interface SkillContext {
  userId: string;
  role: string;
  branchId?: number;
  timeRange: { start: Date; end: Date };
}

interface SkillInsight {
  type: string;
  severity: 'critical' | 'warning' | 'info' | 'positive';
  message: string;
  data: any;
  requiresAI: boolean; // AI zenginleştirme gerekli mi?
}
```

2. Skill Registry:
```typescript
const SKILL_REGISTRY: AgentSkill[] = [
  dailyCoachSkill,        // Barista günlük koçluk
  teamTrackerSkill,       // Supervisor ekip takibi
  stockAssistantSkill,    // Stok yönetimi
  customerWatcherSkill,   // Müşteri memnuniyeti
  productionDirectorSkill,// Fabrika üretim
  foodSafetySkill,        // Gıda güvenliği
  trainingOptimizerSkill, // Eğitim optimizasyonu
  performanceCoachSkill,  // Performans koçluğu
];
```

3. Agent Engine'i skill-based çalışacak şekilde güncelle:
```typescript
// Mevcut runAgentAnalysis yerine:
async function runSkillsForUser(userId: string, role: string) {
  const applicableSkills = SKILL_REGISTRY.filter(
    skill => skill.targetRoles.includes(role)
  );
  
  for (const skill of applicableSkills) {
    // Adım 1: Kural bazlı analiz
    const insights = await skill.analyze(context);
    
    if (insights.length === 0) continue; // Sorun yoksa atla
    
    // Adım 2: AI zenginleştirme (sadece gerekiyorsa)
    const aiInsights = insights.filter(i => i.requiresAI);
    let enriched = insights;
    if (aiInsights.length > 0 && skill.aiEnrich) {
      enriched = await skill.aiEnrich(aiInsights);
    }
    
    // Adım 3: Aksiyon üret
    const actions = skill.generateActions(enriched);
    
    // Adım 4: pending_actions'a kaydet
    for (const action of actions) {
      await savePendingAction(action);
    }
  }
}
```

// Skill schedule'ları:
// 'realtime' → her API çağrısında (dashboard yüklenirken)
// 'hourly'   → saatte bir (09:00-18:00 arası)
// 'daily'    → günde bir (sabah 07:00 veya 08:00)
// 'weekly'   → haftada bir (Pazartesi 09:00 veya Cuma 17:00)

function startSkillScheduler() {
  // Saatlik skill'ler
  setInterval(async () => {
    const hour = new Date().getHours();
    if (hour >= 7 && hour <= 20) { // Çalışma saatleri
      await runHourlySkills();
    }
  }, 60 * 60 * 1000); // 1 saat

  // Günlük skill'ler (sabah 07:00)
  scheduleDaily('07:00', runDailySkills);
  
  // Haftalık skill'ler (Pazartesi 09:00)
  scheduleWeekly('monday', '09:00', runWeeklySkills);
}

async function runHourlySkills() {
  const hourlySkills = SKILL_REGISTRY.filter(s => s.schedule === 'hourly');
  const activeUsers = await getActiveUsers(); // Bugün giriş yapmış kullanıcılar
  
  for (const user of activeUsers) {
    const userSkills = hourlySkills.filter(s => s.targetRoles.includes(user.role));
    for (const skill of userSkills) {
      await runSkillForUser(skill, user);
    }
  }
}

2. AI zenginleştirme fonksiyonu:
```typescript
// server/agent/skills/ai-enrichment.ts (yeni)

async function enrichInsightsWithAI(
  insights: SkillInsight[],
  userContext: UserContext
): Promise<EnrichedInsight[]> {
  
  // Token budget kontrolü
  const budget = await checkTokenBudget(userContext.userId);
  if (!budget.hasRemaining) {
    return insights.map(i => ({ ...i, aiMessage: null }));
  }
  
  // Sadece requiresAI=true olan insight'lar için
  const aiInsights = insights.filter(i => i.requiresAI);
  if (aiInsights.length === 0) return insights;
  
  // Batch prompt (tüm insight'ları tek çağrıda)
  const prompt = buildEnrichmentPrompt(aiInsights, userContext);
  
  const response = await callLlm({
    model: 'gpt-4o-mini', // Varsayılan ucuz model
    messages: [
      { role: 'system', content: DOBODY_SYSTEM_PROMPT },
      { role: 'user', content: prompt }
    ],
    max_tokens: 500,
    temperature: 0.7
  });
  
  // Token kullanımı güncelle
  await incrementTokenUsage(userContext.userId, response.usage.total_tokens);
  
  return parseAIResponse(response, insights);
}
```

**Yapılacak:**

1. Skill aksiyon sonuçlarını bildirime dönüştür:
```typescript
// full_auto skill'ler → doğrudan bildirim gönder
// suggest_approve skill'ler → pending_action oluştur + bildirim

async function processSkillActions(actions: PendingAction[], skill: AgentSkill) {
  for (const action of actions) {
    if (skill.autonomyLevel === 'full_auto') {
      // Doğrudan bildirim gönder
      await createNotification({
        userId: action.targetUserId,
        type: 'agent_suggestion',
        title: `💬 Mr. Dobody: ${action.title}`,
        message: action.description,
        link: action.deepLink
      });
      // Log
      await logAgentAction(action, 'auto_executed');
      
    } else if (skill.autonomyLevel === 'suggest_approve') {
      // Pending action oluştur
      await savePendingAction(action);
      // Onay bekleyen kişiye bildirim
      await createNotification({
        userId: action.approverUserId,
        type: 'agent_pending',
        title: `🤖 Mr. Dobody öneri bekliyor`,
        message: action.title,
        link: '/agent-center'
      });
    }
  }
}
```
