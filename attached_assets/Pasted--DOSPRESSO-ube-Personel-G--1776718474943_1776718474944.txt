─── BÖLÜM 2: PERSONEL DAĞILIMI ───

SQL:
SELECT 
  b.name as branch,
  u.role,
  COUNT(*) as count,
  COUNT(*) FILTER (WHERE u.is_active = true) as active,
  COUNT(*) FILTER (WHERE u.account_status = 'approved') as approved
FROM users u
LEFT JOIN branches b ON b.id = u.branch_id
GROUP BY b.name, u.role
ORDER BY b.name, u.role;

SORULAR:

─── BÖLÜM 3: ŞUBE GÖREV SİSTEMİ (CLAIM/HAVUZ) ───

Tablolar: branch_task_categories, branch_recurring_tasks, 
         branch_task_instances, branchRecurringTaskOverrides

SQL:
-- Kategoriler
SELECT id, name, is_active FROM branch_task_categories ORDER BY id;

-- Aktif recurring tasks (şablon)
SELECT 
  rt.id, rt.name, rt.frequency, rt.category_id,
  bc.name as category,
  COUNT(ti.id) as instance_count_30d
FROM branch_recurring_tasks rt
LEFT JOIN branch_task_categories bc ON bc.id = rt.category_id  
LEFT JOIN branch_task_instances ti ON ti.recurring_task_id = rt.id 
  AND ti.created_at > NOW() - INTERVAL '30 days'
WHERE rt.is_active = true
GROUP BY rt.id, bc.name
ORDER BY rt.frequency, rt.name;

-- Son 7 gün claim istatistiği
SELECT 
  COUNT(*) as toplam,
  COUNT(*) FILTER (WHERE claimed_by IS NOT NULL) as claim_edilen,
  COUNT(*) FILTER (WHERE completed_at IS NOT NULL) as tamamlanan,
  COUNT(*) FILTER (WHERE claimed_by IS NULL AND completed_at IS NULL) as havuzda_bekleyen
FROM branch_task_instances
WHERE created_at > NOW() - INTERVAL '7 days';

SORULAR:
- Şube görevi claim sistemi GERÇEKTEN kullanılıyor mu (son 7 günde)?
- Kaç farklı görev şablonu var? (günlük/haftalık/aylık dağılımı)
- Havuzda bekleyen (claim edilmemiş) görev oranı nedir?
- En çok hangi kategoride görev var?

─── BÖLÜM 4: KİŞİSEL GÖREVLER (ATANAN) ───

Tablolar: shiftTasks, roleTaskTemplates, roleTaskCompletions, 
         eventTriggeredTasks, checklists

SQL:
-- Vardiya görevleri
SELECT 
  COUNT(*) as total,
  COUNT(*) FILTER (WHERE status = 'completed') as completed,
  COUNT(*) FILTER (WHERE status = 'pending') as pending,
  COUNT(*) FILTER (WHERE status = 'in_progress') as in_progress
FROM shift_tasks
WHERE created_at > NOW() - INTERVAL '7 days';

-- Rol bazlı görev şablonları
SELECT role, COUNT(*) as template_count 
FROM role_task_templates 
WHERE is_active = true 
GROUP BY role 
ORDER BY template_count DESC;

-- Checklist kullanımı
SELECT 
  c.title,
  c.branch_id,
  b.name as branch_name,
  COUNT(cc.id) as completion_count_7d
FROM checklists c
LEFT JOIN branches b ON b.id = c.branch_id
LEFT JOIN checklist_completions cc ON cc.checklist_id = c.id 
  AND cc.completed_at > NOW() - INTERVAL '7 days'
WHERE c.is_active = true
GROUP BY c.id, b.name
ORDER BY completion_count_7d DESC
LIMIT 20;

SORULAR:
- Shift_tasks kullanılıyor mu? Günlük ortalama kaç görev?
- Role task templates hangi rollerde var, hangi rollerde yok?
- Checklist'ler hangi şubelerde aktif?
- 4 ayrı görev sistemi (shift_tasks, branch_tasks, role_tasks, checklists) 
  birbiriyle nasıl ilişkili? Kopuk mu, entegre mi?

─── BÖLÜM 5: KİOSK KULLANIMI (ŞUAN) ───

SQL:
-- Kiosk giriş/çıkış
SELECT 
  b.name as branch,
  DATE(sa.check_in_time) as date,
  COUNT(*) as giris_sayisi,
  COUNT(DISTINCT sa.user_id) as farkli_kullanici
FROM shift_attendance sa
JOIN branches b ON b.id = sa.branch_id
WHERE sa.source IN ('kiosk', 'hq_kiosk', 'factory_kiosk')
  AND sa.check_in_time > NOW() - INTERVAL '14 days'
GROUP BY b.name, DATE(sa.check_in_time)
ORDER BY date DESC, giris_sayisi DESC
LIMIT 30;

-- Quiz zorunluluğu
SELECT 
  a.id, a.title, a.has_quiz, a.is_mandatory, a.audience,
  COUNT(aq.id) as question_count,
  COUNT(DISTINCT aack.user_id) as acknowledged_users
FROM announcements a
LEFT JOIN announcement_quiz_questions aq ON aq.announcement_id = a.id
LEFT JOIN announcement_acknowledgments aack ON aack.announcement_id = a.id
WHERE a.is_active = true
GROUP BY a.id
ORDER BY a.created_at DESC
LIMIT 20;

SORULAR:
