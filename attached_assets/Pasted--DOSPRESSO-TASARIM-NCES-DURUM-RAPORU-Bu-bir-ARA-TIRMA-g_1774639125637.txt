### 2B — Canlı API Testi (17 rol)
```bash
# Cookie jar temizle
rm -f /tmp/cookies_*.txt

# Test fonksiyonu
test_role() {
  local user=$1
  local expected_mc=$2
  local role_name=$3
  
  # Login
  HTTP_CODE=$(curl -s -o /tmp/login_resp.json -w '%{http_code}' \
    -c /tmp/cookies_${user}.txt \
    -H 'Content-Type: application/json' \
    -d "{\"username\":\"${user}\",\"password\":\"0000\"}" \
    http://localhost:5000/api/login)
  
  if [ "$HTTP_CODE" != "200" ]; then
    echo "❌ ${role_name} (${user}): LOGIN FAILED (${HTTP_CODE})"
    return
  fi
  
  ROLE=$(cat /tmp/login_resp.json | python3 -c "import sys,json; print(json.load(sys.stdin).get('role','?'))" 2>/dev/null)
  
  # Dashboard data
  DASH_CODE=$(curl -s -o /dev/null -w '%{http_code}' \
    -b /tmp/cookies_${user}.txt \
    http://localhost:5000/api/me/dashboard-data)
  
  # Widget config
  WIDGET_CODE=$(curl -s -o /dev/null -w '%{http_code}' \
    -b /tmp/cookies_${user}.txt \
    http://localhost:5000/api/me/widget-config)
  
  WIDGET_COUNT=$(curl -s -b /tmp/cookies_${user}.txt \
    http://localhost:5000/api/me/widget-config | python3 -c "import sys,json; d=json.load(sys.stdin); print(len(d) if isinstance(d,list) else '?')" 2>/dev/null)
  
  echo "✅ ${role_name} (${user}): role=${ROLE}, dashboard=${DASH_CODE}, widgets=${WIDGET_CODE} (${WIDGET_COUNT} adet), expected_mc=${expected_mc}"
}

echo "=== 17 ROL LOGIN + DASHBOARD TESTİ ==="
test_role "ali" "MissionControlHQ" "CEO"
test_role "utku" "MissionControlHQ" "CGO"
test_role "admin" "MissionControlHQ" "Admin"
test_role "yavuz" "MissionControlCoach" "Coach"
# Trainer kullanıcı adını araştır:
echo "--- Trainer kullanıcı ---"
curl -s -H 'Content-Type: application/json' -d '{"username":"admin","password":"0000"}' -c /tmp/cookies_admin2.txt http://localhost:5000/api/login > /dev/null
curl -s -b /tmp/cookies_admin2.txt "http://localhost:5000/api/users?role=trainer" | python3 -c "import sys,json; [print(f'  {u[\"username\"]} - {u[\"firstName\"]} {u[\"lastName\"]}') for u in json.load(sys.stdin)]" 2>/dev/null || echo "  Trainer bulunamadı"
test_role "mehmet" "MissionControlSupervisor" "Supervisor"
# Muhasebe kullanıcı adını araştır:
echo "--- Muhasebe kullanıcı ---"
curl -s -b /tmp/cookies_admin2.txt "http://localhost:5000/api/users?role=muhasebe_ik" | python3 -c "import sys,json; [print(f'  {u[\"username\"]} - {u[\"firstName\"]} {u[\"lastName\"]}') for u in json.load(sys.stdin)]" 2>/dev/null || echo "  Muhasebe bulunamadı"
test_role "umit" "MissionControlFabrika" "Üretim Şefi"
# Yatırımcı:
echo "--- Yatırımcı kullanıcı ---"
curl -s -b /tmp/cookies_admin2.txt "http://localhost:5000/api/users?role=yatirimci_branch" | python3 -c "import sys,json; [print(f'  {u[\"username\"]} - {u[\"firstName\"]} {u[\"lastName\"]}') for u in json.load(sys.stdin)]" 2>/dev/null || echo "  Yatırımcı bulunamadı"

echo ""
echo "=== KRİTİK API ENDPOINT TESTLERİ ==="
curl -s -o /dev/null -w "dashboard/coach: %{http_code}\n" -b /tmp/cookies_yavuz.txt http://localhost:5000/api/dashboard/coach
curl -s -o /dev/null -w "dashboard/executive: %{http_code}\n" -b /tmp/cookies_utku.txt http://localhost:5000/api/dashboard/executive
curl -s -o /dev/null -w "hq-summary: %{http_code}\n" -b /tmp/cookies_ali.txt http://localhost:5000/api/hq-summary
```

```bash
import sys,json
tasks = json.load(sys.stdin)
if isinstance(tasks, list):
  active = [t for t in tasks if t.get('isActive', t.get('is_active', False))]
  print(f'Toplam: {len(tasks)}, Aktif: {len(active)}')
  for t in active:
    print(f'  - {t.get(\"title\",\"?\")} | branchId={t.get(\"branchId\",t.get(\"branch_id\",\"NULL\"))} | freq={t.get(\"frequency\",\"?\")}')
```
