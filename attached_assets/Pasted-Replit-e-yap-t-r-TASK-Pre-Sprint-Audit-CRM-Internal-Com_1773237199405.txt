### 1. Database Tables Inventory

```bash
echo "=== CRM & Communication Tables ==="
psql -c "
SELECT table_name, 
  (SELECT COUNT(*) FROM information_schema.columns WHERE table_name=t.table_name) as columns

AND (table_name ILIKE '%feedback%' 
  OR table_name ILIKE '%complaint%'
```

### 3. Task / Assignment System Analysis

```bash
echo "=== tasks table structure ==="
psql -c "SELECT column_name, data_type FROM information_schema.columns WHERE table_name='tasks' ORDER BY ordinal_position;" 2>/dev/null

echo ""
echo "=== Task assignment patterns (who assigns to whom) ==="
psql -c "
SELECT 
  creator.role as assigned_by_role,
  assignee.role as assigned_to_role,
  COUNT(*) as count
FROM tasks t
JOIN users creator ON t.created_by = creator.id
JOIN users assignee ON t.assigned_to::integer = assignee.id
GROUP BY creator.role, assignee.role
ORDER BY count DESC
LIMIT 15;" 2>/dev/null

echo ""
echo "=== Task statuses ==="
psql -c "SELECT status, COUNT(*) FROM tasks GROUP BY status ORDER BY COUNT(*) DESC;" 2>/dev/null

echo ""
echo "=== HQ-to-branch task flow ==="
psql -c "
SELECT 
  creator.role as from_role,
  assignee.role as to_role,
  t.priority,
  t.status,
  t.title
FROM tasks t
JOIN users creator ON t.created_by = creator.id
JOIN users assignee ON t.assigned_to::integer = assignee.id
WHERE creator.role IN ('admin','ceo','cgo','coach','trainer')
ORDER BY t.created_at DESC
LIMIT 10;" 2>/dev/null
```

```bash
echo ""
```

```bash
echo ""
echo "=== equipment_faults — who reports, who resolves ==="
psql -c "
SELECT 
  reporter.role as reported_by,
  resolver.role as resolved_by,
  COUNT(*)
FROM equipment_faults ef
JOIN users reporter ON ef.reported_by = reporter.id
LEFT JOIN users resolver ON ef.resolved_by = resolver.id
GROUP BY reporter.role, resolver.role
ORDER BY COUNT(*) DESC
LIMIT 10;" 2>/dev/null
```

### 6. Notification System — What Types Exist

```bash
echo "=== Notification types ==="
psql -c "SELECT type, COUNT(*) FROM notifications GROUP BY type ORDER BY COUNT(*) DESC LIMIT 20;" 2>/dev/null

echo ""
echo "=== Notifications per role (last 7 days) ==="
psql -c "
SELECT u.role, COUNT(n.id) as notification_count
FROM notifications n
JOIN users u ON n.user_id = u.id
WHERE n.created_at > NOW() - INTERVAL '7 days'
GROUP BY u.role
ORDER BY notification_count DESC;" 2>/dev/null
```
