```typescript
// client/src/components/mission-control/dashboard-section-config.ts (YENİ DOSYA)

export interface DashboardSectionConfig {
  key: string;
  title: string;
  visibleTo: string[];  // roller listesi — boş array = herkese görünür
  defaultOpen: boolean;
}

// Her section'ın rol bazlı görünürlük config'i
export const DASHBOARD_SECTIONS: DashboardSectionConfig[] = [
  {
    key: 'ai_briefing',
    title: 'AI Briefing',
    visibleTo: [],  // herkese
    defaultOpen: true,
  },
  {
    key: 'kpi_strip',
    title: 'KPI Özet',
    visibleTo: [],  // herkese
    defaultOpen: true,
  },
  {
    key: 'todays_tasks',
    title: 'Bugünün Görevleri',
    visibleTo: [],  // herkese
    defaultOpen: true,
  },
  {
    key: 'alerts',
    title: 'Uyarılar',
    visibleTo: ['ceo', 'cgo', 'admin', 'coach', 'trainer', 'kalite_kontrol'],
    defaultOpen: true,
  },
  {
    key: 'branch_health',
    title: 'Şube Sağlığı',
    visibleTo: ['ceo', 'cgo', 'admin', 'coach', 'trainer'],
    defaultOpen: true,
  },
  {
    key: 'trend_chart',
    title: '6 Aylık Trend',
    visibleTo: ['ceo', 'cgo', 'admin'],
    defaultOpen: false,
  },
  {
    key: 'branch_grid',
    title: 'Şube Durumu',
    visibleTo: ['ceo', 'cgo', 'admin', 'coach'],
    defaultOpen: false,
  },
  {
    key: 'ik_summary',
    title: 'İK Özet',
    visibleTo: ['ceo', 'admin', 'muhasebe_ik', 'muhasebe'],
    defaultOpen: false,
  },
  {
    key: 'factory_qc',
    title: 'Fabrika & Kalite',
    visibleTo: ['ceo', 'admin', 'kalite_kontrol', 'gida_muhendisi'],
    defaultOpen: false,
  },
  {
    key: 'quick_actions',
    title: 'Hızlı İşlemler',
    visibleTo: [],  // herkese
    defaultOpen: true,
  },
];

// Rol bazlı filtreleme helper
export function getVisibleSections(role: string): string[] {
  return DASHBOARD_SECTIONS
    .filter(s => s.visibleTo.length === 0 || s.visibleTo.includes(role))
    .map(s => s.key);
}

export function isSectionVisible(sectionKey: string, role: string): boolean {
  const section = DASHBOARD_SECTIONS.find(s => s.key === sectionKey);
  if (!section) return true;  // bilinmeyen section → göster
  return section.visibleTo.length === 0 || section.visibleTo.includes(role);
}

export function getSectionDefaultOpen(sectionKey: string): boolean {
  const section = DASHBOARD_SECTIONS.find(s => s.key === sectionKey);
  return section?.defaultOpen ?? false;
}
```

```typescript
// Yatırımcı Dashboard — Şube finansal performansı odaklı
// Read-only, basit, finans odaklı

export default function MissionControlYatirimci() {
  const { user } = useAuth();
  const branchId = user?.branchId;
  
  // Branch summary API (mevcut)
  const { data: branchData } = useQuery({
    queryKey: ['/api/branch-summary', branchId],
    enabled: !!branchId,
  });
  
  // Branch dashboard alerts (mevcut)
  const { data: dashData } = useQuery({
    queryKey: ['/api/dashboard/branch', branchId],
    enabled: !!branchId,
  });
  
  // AI Briefing (mevcut)
  const { data: briefing } = useQuery({
    queryKey: ['/api/me/dashboard-briefing'],
    staleTime: 30 * 60 * 1000,
  });

  return (
    <div className="space-y-4 p-3 md:p-4">
      {/* Selamlama */}
      <div className="flex items-center justify-between">
        <div>
          <h2 className="text-2xl font-bold">
            {getGreeting()}, {user?.firstName}
          </h2>
          <p className="text-sm text-muted-foreground">
            {branchData?.branchName} — Yatırımcı Paneli
          </p>
        </div>
        <Badge variant="outline">Yatırımcı</Badge>
      </div>

      {/* AI Briefing (varsa) */}
      {briefing?.summary && (
        <CollapsibleSection title="AI Briefing" defaultOpen={true}>
          <p className="text-sm">{briefing.summary}</p>
        </CollapsibleSection>
      )}

      {/* KPI Strip — yatırımcı odaklı */}
      <UnifiedKPI variant="compact" items={[
        { label: 'Şube Skoru', value: branchData?.healthScore || '—', color: getScoreColor(branchData?.healthScore), suffix: '/100' },
        { label: 'Aktif Personel', value: branchData?.staffCount || 0, color: 'info' },
        { label: 'Müşteri Puanı', value: branchData?.customerRating || '—', suffix: '/5', color: 'info' },
        { label: 'Açık Arıza', value: branchData?.openFaults || 0, color: branchData?.openFaults > 0 ? 'danger' : 'success' },
      ]} />

      {/* Şube Sağlık Özeti */}
      <CollapsibleSection
        title="Şube Sağlık Durumu"
        badge={branchData?.healthScore ? `${branchData.healthScore}/100` : '—'}
        badgeVariant={getScoreBadgeVariant(branchData?.healthScore)}
        defaultOpen={true}
      >
        {/* Sağlık bileşenleri bar chart veya liste */}
        <div className="space-y-2 text-sm">
          {branchData?.scoreBreakdown?.map((item: any) => (
            <div key={item.label} className="flex justify-between">
              <span>{item.label}</span>
              <span className={cn(
                'font-medium',
                item.score >= 80 ? 'text-emerald-600' : item.score >= 60 ? 'text-amber-600' : 'text-red-600'
              )}>
                {item.score}/100
              </span>
            </div>
          ))}
        </div>
      </CollapsibleSection>

      {/* Uyarılar (varsa) */}
      {dashData?.alerts?.length > 0 && (
        <CollapsibleSection
          title="Dikkat Gereken Konular"
          badge={dashData.alerts.length}
          badgeVariant="warning"
          defaultOpen={true}
        >
          <AlertPanel alerts={dashData.alerts} />
        </CollapsibleSection>
      )}

      {/* Personel Durumu */}
      <CollapsibleSection title="Personel" defaultOpen={false}>
        {/* Aktif/izinli/checklist durumu */}
      </CollapsibleSection>

      {/* Quick Actions — yatırımcıya özel */}
      <div className="grid grid-cols-2 gap-2">
        <Link href="/crm" className="...">CRM & Talepler</Link>
        <Link href="/raporlar" className="...">Raporlar</Link>
      </div>
    </div>
  );
}
```
