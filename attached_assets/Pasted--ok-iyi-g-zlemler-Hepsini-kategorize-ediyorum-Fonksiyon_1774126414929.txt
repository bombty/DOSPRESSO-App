Map each guidance gap type to a navigation target:

```typescript
// Guidance gap → navigation mapping
const GUIDANCE_ACTIONS: Record<string, { path: string; label: string }> = {
  // Personnel gaps
  'no-manager': { path: '/admin/kullanicilar', label: 'Kullanıcı Yönetimi' },
  'no-supervisor': { path: '/admin/kullanicilar', label: 'Kullanıcı Yönetimi' },
  'insufficient-staff': { path: '/admin/kullanicilar', label: 'Kullanıcı Yönetimi' },
  'yetersiz-personel': { path: '/admin/kullanicilar', label: 'Kullanıcı Yönetimi' },
  'yonetici-yok': { path: '/admin/kullanicilar', label: 'Kullanıcı Yönetimi' },
  
  // Checklist gaps
  'no-checklist': { path: '/yonetim/checklist-yonetimi', label: 'Checklist Yönetimi' },
  'checklist-eksik': { path: '/yonetim/checklist-yonetimi', label: 'Checklist Yönetimi' },
  
  // Shift/Schedule gaps
  'no-shifts': { path: '/vardiya-planlama', label: 'Vardiya Planlama' },
  'vardiya-eksik': { path: '/vardiya-planlama', label: 'Vardiya Planlama' },
  
  // Training gaps
  'training-incomplete': { path: '/akademi-hq', label: 'Akademi Yönetimi' },
  'egitim-eksik': { path: '/akademi-hq', label: 'Akademi Yönetimi' },
  
  // Factory gaps
  'fabrika-no-targets': { path: '/fabrika/uretim-planlama', label: 'Üretim Planlama' },
  'fabrika-no-shifts': { path: '/fabrika/vardiya-planlama', label: 'Fabrika Vardiya' },
  'fabrika-onboarding': { path: '/admin/kullanicilar', label: 'Personel Yönetimi' },
  
  // Settings gaps
  'no-sla': { path: '/admin', label: 'Admin Ayarları' },
  'settings-incomplete': { path: '/admin', label: 'Admin Ayarları' },
  
  // Default
  'default': { path: '/admin', label: 'Yönetim Paneli' },
};

// In the guidance widget, update the Düzelt button:
function getActionForGap(gap: any): { path: string; label: string } {
  // Try to match by gap.id first
  const idMatch = Object.entries(GUIDANCE_ACTIONS).find(([key]) => 
    gap.id?.toLowerCase().includes(key) || gap.category?.toLowerCase().includes(key)
  );
  if (idMatch) return idMatch[1];
  
  // Try to match by gap.title or description keywords
  const text = (gap.title + ' ' + gap.description).toLowerCase();
  if (text.includes('personel') || text.includes('yönetici') || text.includes('müdür')) 
    return GUIDANCE_ACTIONS['no-manager'];
  if (text.includes('checklist') || text.includes('kontrol listesi'))
    return GUIDANCE_ACTIONS['no-checklist'];
  if (text.includes('vardiya') || text.includes('shift'))
    return GUIDANCE_ACTIONS['no-shifts'];
  if (text.includes('eğitim') || text.includes('akademi'))
    return GUIDANCE_ACTIONS['training-incomplete'];
  if (text.includes('fabrika') || text.includes('üretim'))
    return GUIDANCE_ACTIONS['fabrika-no-targets'];
  
  return GUIDANCE_ACTIONS['default'];
}

// Düzelt button onClick:
<Button variant="outline" size="sm" onClick={() => {
  const action = getActionForGap(gap);
  navigate(action.path);
}}>
  Düzelt →
</Button>
```

## FIX 3: Mr. Dobody Guidance — Collapse/Toggle System

The guidance widget shows 37+ items in a long list. Add collapse functionality:

```typescript
// In guidance-widget.tsx:

// Group gaps by category
const groupedGaps = useMemo(() => {
  const groups: Record<string, typeof gaps> = {};
  gaps.forEach(gap => {
    const category = gap.category || 'other';
    if (!groups[category]) groups[category] = [];
    groups[category].push(gap);
  });
  return groups;
}, [gaps]);

// Collapsible sections
const [expandedGroups, setExpandedGroups] = useState<Set<string>>(new Set());

function toggleGroup(category: string) {
  setExpandedGroups(prev => {
    const next = new Set(prev);
    if (next.has(category)) next.delete(category);
    else next.add(category);
    return next;
  });
}

// Render:
<div className="space-y-2">
  {/* Summary bar — always visible */}
  <div className="flex items-center justify-between p-2 bg-muted/30 rounded-lg">
    <div className="flex items-center gap-2">
      <span className="text-sm font-medium">Mr. Dobody — Yapılması Gerekenler</span>
      <Badge variant="secondary">{gaps.length} eksik</Badge>
    </div>
    <div className="flex gap-2">
      {criticalCount > 0 && <Badge variant="destructive">{criticalCount} kritik</Badge>}
      <Button variant="ghost" size="sm" onClick={() => setShowAll(!showAll)}>
        {showAll ? 'Daralt' : 'Tümünü Gör'}
      </Button>
    </div>
  </div>
  
  {/* Collapsed view — only show top 3 critical items */}
  {!showAll && (
    <div className="space-y-1.5">
      {gaps.filter(g => g.severity === 'critical').slice(0, 3).map(gap => (
        <GuidanceItem key={gap.id} gap={gap} compact />
      ))}
      {gaps.length > 3 && (
        <Button variant="ghost" size="sm" className="w-full text-xs text-muted-foreground"
          onClick={() => setShowAll(true)}>
          +{gaps.length - 3} daha fazla göster
        </Button>
      )}
    </div>
  )}
  
  {/* Expanded view — grouped by category */}
  {showAll && (
    <div className="space-y-2">
      {Object.entries(groupedGaps).map(([category, items]) => (
        <div key={category} className="border border-border rounded-lg overflow-hidden">
          <button className="w-full flex items-center justify-between p-2 bg-muted/20 hover:bg-muted/40 text-xs"
            onClick={() => toggleGroup(category)}>
            <span className="font-medium capitalize">{getCategoryLabel(category)}</span>
            <div className="flex items-center gap-2">
              <Badge variant="secondary" className="text-[10px]">{items.length}</Badge>
              <ChevronDown className={`h-3 w-3 transition-transform ${expandedGroups.has(category) ? 'rotate-180' : ''}`} />
            </div>
          </button>
          {expandedGroups.has(category) && (
            <div className="p-1.5 space-y-1">
              {items.map(gap => <GuidanceItem key={gap.id} gap={gap} />)}
            </div>
          )}
        </div>
      ))}
    </div>
  )}
</div>
```

Category label helper:
```typescript
function getCategoryLabel(category: string): string {
  const labels: Record<string, string> = {
    'personnel': 'Personel',
    'checklist': 'Checklist',
    'settings': 'Ayarlar',
    'data': 'Veri',
    'training': 'Eğitim',
    'configuration': 'Yapılandırma',
    'other': 'Diğer',
  };
  return labels[category] || category;
}
```
