#### A3: Bulk module check endpoint (performance optimization)
The current /api/module-flags/check endpoint checks ONE module at a time. With 15+ guards on routes, this means 15+ API calls on app load.

Create a new endpoint: `GET /api/module-flags/my-flags`
- Uses isAuthenticated (not admin-only)
- Returns ALL effective flags for the current user (based on their branchId and role)
- Response: `{ flags: { checklist: true, akademi: false, crm: true, ... } }`
- This single call replaces all individual /check calls

Update the frontend hook `use-module-flags.ts`:
```typescript
// New hook that fetches all flags at once
export function useMyModuleFlags(): { 
  flags: Record<string, boolean>; 
  isLoading: boolean;
  isModuleEnabled: (key: string) => boolean;
} {
  const { data, isLoading } = useQuery<{ flags: Record<string, boolean> }>({
    queryKey: ["/api/module-flags/my-flags"],
    staleTime: 60 * 1000,
  });
  return {
    flags: data?.flags ?? {},
    isLoading,
    isModuleEnabled: (key: string) => data?.flags?.[key] ?? true,
  };
}

// Keep the old useModuleEnabled hook working but make it use the bulk data
export function useModuleEnabled(moduleKey: string) {
  const { isModuleEnabled, isLoading } = useMyModuleFlags();
  return { isEnabled: isModuleEnabled(moduleKey), isLoading, isError: false };
}
```

The /api/module-flags/my-flags endpoint implementation:
```typescript
router.get("/api/module-flags/my-flags", isAuthenticated, async (req, res) => {
  const user = req.user as Express.User;
  const branchId = (user as any)?.branchId ?? null;
  const userRole = (user as any)?.role ?? null;
  
  // Get all non-deleted flags
  const allFlags = await db.select().from(moduleFlags).where(isNull(moduleFlags.deletedAt));
  
  // For each unique moduleKey, compute effective state using isModuleEnabled
  const uniqueKeys = [...new Set(allFlags.map(f => f.moduleKey))];
  const flags: Record<string, boolean> = {};
  for (const key of uniqueKeys) {
    flags[key] = await isModuleEnabled(key, branchId, "ui", userRole);
  }
  
  res.json({ flags });
});
```

#### B2: Integrate module flags into score calculation

```typescript
import { isModuleEnabled } from "../services/module-flag-service";

// Inside the score calculation function:
async function calculateCompositeScore(branchId: number, userId?: string) {
  const components: { key: string; score: number; weight: number }[] = [];
  
  // Only include checklist score if module is enabled
  if (await isModuleEnabled("checklist", branchId, "data")) {
    const checklistScore = await getChecklistScore(branchId, userId);
    components.push({ key: "checklist", score: checklistScore, weight: 20 });
  }
  
  // Only include akademi score if module is enabled
  if (await isModuleEnabled("akademi", branchId, "data")) {
    const trainingScore = await getTrainingScore(branchId, userId);
    components.push({ key: "akademi", score: trainingScore, weight: 20 });
  }
  
  // ... same for pdks, gorevler, crm
  // NOTE: pdks has behavior "ui_hidden_data_continues" — context="data" returns true even when UI is hidden
  // So pdks score is ALWAYS included (which is correct)
  
  // Recalculate weights proportionally based on active components
  const totalWeight = components.reduce((sum, c) => sum + c.weight, 0);
  const normalizedScore = components.reduce((sum, c) => sum + (c.score * c.weight / totalWeight), 0);
  
  return { score: normalizedScore, components, totalWeight };
}
```

IMPORTANT: 
- Use context="data" for score calculations (not "ui") — this ensures pdks/vardiya data is always included even when UI is hidden
- If ALL score components are disabled (unlikely but possible), return score=0 with a note
- The existing score function might use a completely different pattern — ADAPT to the existing code structure, don't rewrite it
