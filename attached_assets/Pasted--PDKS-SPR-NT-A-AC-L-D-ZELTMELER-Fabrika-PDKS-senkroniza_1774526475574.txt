## A1.1: Fabrika kiosk check-in endpoint'ine PDKS insert ekle

**Pattern:** Şube kiosk'un PDKS yazma akışını referans al — aynı mantığı fabrika'ya ekle.

```typescript
// Fabrika check-in endpoint'inde (factory-shift-routes.ts veya benzeri):
// MEVCUT: Sadece factory_shift_sessions'a insert
// EKLENECEK: pdks_records'a da "giris" kaydı yaz

// Session oluşturulduktan HEMEN SONRA:
try {
  await db.insert(pdksRecords).values({
    userId: userId.toString(),
    branchId: factoryBranchId,  // Fabrika'nın branch_id'si
    recordDate: new Date().toISOString().split('T')[0],
    recordTime: new Date().toTimeString().split(' ')[0],  // "HH:MM:SS"
    recordType: 'giris',
    source: 'kiosk',
  });
} catch (pdksError) {
  // PDKS hatası session'ı engellememeli — log yaz, devam et
  console.error('Fabrika PDKS giriş kaydı hatası:', pdksError);
}
```

## A1.5: Geriye dönük PDKS migration (80 eksik kayıt)

```sql
-- Fabrika session'larından eksik PDKS giriş kayıtları oluştur
INSERT INTO pdks_records (user_id, branch_id, record_date, record_time, record_type, source)
SELECT 
  fss.user_id::text,
  (SELECT id FROM branches WHERE name ILIKE '%fabrika%' OR branch_type = 'factory' LIMIT 1),
  fss.check_in_time::date,
  fss.check_in_time::time,
  'giris',
  'migration_fix'
FROM factory_shift_sessions fss
WHERE NOT EXISTS (
  SELECT 1 FROM pdks_records p 
  WHERE p.user_id = fss.user_id::text 
  AND p.record_date = fss.check_in_time::date 
  AND p.record_type = 'giris'
);

-- Fabrika session'larından eksik PDKS çıkış kayıtları oluştur (completed + auto_closed)
INSERT INTO pdks_records (user_id, branch_id, record_date, record_time, record_type, source)
SELECT 
  fss.user_id::text,
  (SELECT id FROM branches WHERE name ILIKE '%fabrika%' OR branch_type = 'factory' LIMIT 1),
  fss.check_in_time::date,
  COALESCE(fss.check_out_time::time, '20:30:00'),
  'cikis',
  'migration_fix'
FROM factory_shift_sessions fss
WHERE fss.status IN ('completed', 'auto_closed')
AND NOT EXISTS (
  SELECT 1 FROM pdks_records p 
  WHERE p.user_id = fss.user_id::text 
  AND p.record_date = fss.check_in_time::date 
  AND p.record_type = 'cikis'
);
```

**⚠ Migration çalıştırmadan ÖNCE:**
- Fabrika branch_id'sini doğrula
- Kaç kayıt oluşacağını SELECT ile kontrol et (INSERT yapmadan)
- Onay al

```typescript
// agent-scheduler.ts veya index.ts'e ekle:

// Her gün 22:00 TR — şube açık session'ları kapat
schedule('branch-auto-close', '0 22 * * *', async () => {
  const openSessions = await db.select()
    .from(branchShiftSessions)
    .where(eq(branchShiftSessions.status, 'active'));
  
  for (const session of openSessions) {
    // Session kapat
    await db.update(branchShiftSessions)
      .set({
        status: 'auto_closed',
        checkOutTime: new Date(),
        netWorkMinutes: calculateMinutes(session.checkInTime, new Date()),
      })
      .where(eq(branchShiftSessions.id, session.id));
    
    // PDKS çıkış yaz
    try {
      await db.insert(pdksRecords).values({
        userId: session.userId.toString(),
        branchId: session.branchId,
        recordDate: session.checkInTime.toISOString().split('T')[0],
        recordTime: '22:00:00',
        recordType: 'cikis',
        source: 'auto_close',
      });
    } catch (e) {
      console.error('Şube auto-close PDKS hatası:', e);
    }
  }
  
  if (openSessions.length > 0) {
    console.log(`[Branch Auto-Close] ${openSessions.length} session kapatıldı`);
    // İsteğe bağlı: Supervisor'lara bildirim gönder
  }
});
```

## A3.3: Geç kalma tespiti — kiosk giriş anında

```typescript
// Tüm kiosk'larda (şube/fabrika/HQ) check-in anında:

// 1. Bu personelin bugünkü planlanan vardiya saatini bul
const plannedShift = await db.select()
  .from(shifts)
  .where(and(
    eq(shifts.userId, userId),
    eq(shifts.date, today),
  ))
  .limit(1);

// 2. Veya lokasyon ayarlarından varsayılan başlangıç saatini al
const kioskSettings = await db.select()
  .from(branchKioskSettings)
  .where(eq(branchKioskSettings.branchId, branchId))
  .limit(1);

const plannedStart = plannedShift?.[0]?.startTime || kioskSettings?.[0]?.shiftStartTime || '08:00';
const tolerance = kioskSettings?.[0]?.lateToleranceMinutes || 15;

// 3. Geç kalma hesapla
const actualTime = new Date();
const plannedTime = parseTime(plannedStart);  // "08:00" → Date
const diffMinutes = (actualTime - plannedTime) / 60000;

if (diffMinutes > tolerance) {
  // Geç kalma kaydı yaz
  await db.insert(attendancePenalties).values({
    userId: userId,
    branchId: branchId,
    penaltyDate: today,
    penaltyType: 'late_arrival',
    plannedTime: plannedStart,
    actualTime: actualTime.toTimeString().split(' ')[0],
    lateMinutes: Math.round(diffMinutes),
    toleranceMinutes: tolerance,
    excessMinutes: Math.round(diffMinutes - tolerance),  // Tolerans sonrası net ihlal
  });
  
  // İsteğe bağlı: Supervisor bildirim (ileri aşamada)
}
```

**⚠ Bu mantık mevcut check-in akışına EKLENIR — mevcut session + PDKS yazma akışı değişmez.**
