```typescript
// server/lib/export-service.ts

async function runExport(jobId: string, options: ExportOptions) {
  const zip = new JSZip();
  
  // Manifest
  zip.file('manifest.json', JSON.stringify({
    version: '1.0',
    exportDate: new Date().toISOString(),
    scope: options.scope,
    appVersion: process.env.APP_VERSION || '16.0',
    tableCount: 321,
    userCount: await db.select({ count: count() }).from(users)
  }));
  
  // Tablolar — kategori bazlı export
  for (const category of EXPORT_CATEGORIES) {
    for (const table of category.tables) {
      const data = await db.select().from(table.schema);
      zip.file(`data/${category.name}/${table.name}.json`, JSON.stringify(data, null, 2));
      updateProgress(jobId, ...);
    }
  }
  
  // Media (opsiyonel)
  if (options.includeMedia) {
    // Upload dosyalarını ekle
  }
  
  // Checksum
  const content = await zip.generateAsync({ type: 'nodebuffer' });
  const checksum = crypto.createHash('sha256').update(content).digest('hex');
  zip.file('checksum.sha256', checksum);
  
  // Dosyayı kaydet
  const finalContent = await zip.generateAsync({ type: 'nodebuffer' });
  fs.writeFileSync(`/tmp/export-${jobId}.zip`, finalContent);
}
```

```
POST /api/admin/import
  Body: multipart/form-data (zip dosyası + seçenekler)
  {
    file: export.zip,
    mode: 'full_replace' | 'merge' | 'config_only',
    password?: string // şifreli export ise
  }
→ { jobId: string, status: 'validating' }

GET /api/admin/import/:jobId/status
→ { status: 'validating' | 'importing' | 'completed' | 'failed',
```

2. Import modları:

**A) Full Replace:** Tüm mevcut veriyi sil, import'tan yükle (temiz kurulum)
