#### Step 3: Create Professional Certificate Renderer

Create `client/src/components/certificate-renderer.tsx`:

This component renders a professional certificate as an HTML element that can be printed/saved as PDF.

**Design specifications (matching DOSPRESSO brand):**
- Size: A4 landscape (297mm x 210mm) or 1120x790px for screen
- Background: White with subtle navy border frame
- Top: DOSPRESSO logo (centered, ~120px wide)
- Below logo: Gold/navy decorative line separator
- Title: Certificate name in navy (#122549), serif font, 28px
- Subtitle: "DOSPRESSO Donut & Coffee" in smaller text
- Body: "Bu belge, [PERSON NAME]'ın [description] belgeler." 
- Person name: Large, bold, navy
- Date: Issue date in Turkish format
- Bottom left: "Sertifika No: CERT-2026-XXXX"
- Bottom center: DOSPRESSO seal/stamp graphic (create with SVG — circular badge with logo)
- Bottom right: Signature line with "DOSPRESSO Eğitim Müdürü" or custom signer
- Accent: Red (#cc1f1f) used sparingly for decorative elements
- Overall feel: Clean, professional, corporate — like a university diploma

```typescript
interface CertificateProps {
  template: CertificateTemplate;
  recipientName: string;
  issueDate: string;
  certificateId: string;
  moduleName?: string; // For module completion certs
  signerName?: string;
  signerTitle?: string;
}

export function CertificateRenderer({ ... }: CertificateProps) {
  return (

      {/* DOSPRESSO Logo — use the actual logo file */}
      <div className="flex justify-center mb-4">

      </div>
      
      {/* Decorative separator */}
      <div className="flex justify-center mb-6">
        <div style={{ width: '200px', height: '2px', background: 'linear-gradient(90deg, transparent, #d4af37, transparent)' }} />
      </div>

      <h1 style={{ textAlign: 'center', fontSize: '28px', fontWeight: 'bold', marginBottom: '8px' }}>
```

#### Step 4: Add Print/PDF functionality

Add a print button that opens the certificate in a new window for printing:

```typescript
function handlePrintCertificate(certData: CertificateProps) {
  const printWindow = window.open('', '_blank');
  if (!printWindow) return;
  
  // Render certificate HTML into the new window
  printWindow.document.write(`
    <html>
    <head><title>DOSPRESSO Sertifika</title>
    <style>
      @media print { body { margin: 0; } .no-print { display: none; } }
      body { display: flex; justify-content: center; align-items: center; min-height: 100vh; background: #f0f0f0; }
      .cert-wrapper { background: white; }
    </style>
    </head>
    <body>
      <button class="no-print" onclick="window.print()" style="position:fixed;top:10px;right:10px;padding:8px 16px;background:#122549;color:white;border:none;border-radius:4px;cursor:pointer;">Yazdır / PDF</button>
      <div class="cert-wrapper">
        ${renderCertificateHTML(certData)}
      </div>
    </body>
    </html>
  `);
}
```
