## Kiosk Session DB Migration — In-Memory Map → PostgreSQL

Kiosk sessions are stored in an in-memory Map in server/localAuth.ts (line 461). When the server restarts, all active kiosk sessions are lost — factory/branch kiosk users must re-login. This migration moves sessions to PostgreSQL so they persist across restarts.

### STEP 2: Rewrite the 4 kiosk session functions

In server/localAuth.ts, replace the in-memory Map with DB operations:

```typescript
// REMOVE: const kioskSessions = new Map<string, {...}>();

// createKioskSession — same signature, now writes to DB
export async function createKioskSession(userId: string): Promise<string> {
  const token = crypto.randomUUID();
  const expiresAt = new Date(Date.now() + 8 * 60 * 60 * 1000); // 8 hours
  
  await db.insert(kioskSessions).values({
    token,
    userId,
    expiresAt,
  });
  
  return token;
}

// validateKioskSession — same signature, now reads from DB
export async function validateKioskSession(token: string): Promise<{ userId: string; stationId?: number } | null> {
  if (!token) return null;
  
  const [session] = await db
    .select()
    .from(kioskSessions)
    .where(eq(kioskSessions.token, token))
    .limit(1);
  
  if (!session) return null;
  
  // Check expiry
  if (new Date(session.expiresAt) < new Date()) {
    // Expired — delete and return null
    await db.delete(kioskSessions).where(eq(kioskSessions.token, token));
    return null;
  }
  
  return { userId: session.userId, stationId: session.stationId ?? undefined };
}

// updateKioskStation — same signature, now updates DB
export async function updateKioskStation(token: string, stationId: number): Promise<void> {
  await db
    .update(kioskSessions)
    .set({ stationId })
    .where(eq(kioskSessions.token, token));
}

// deleteKioskSession — same signature, now deletes from DB
export async function deleteKioskSession(token: string): Promise<void> {
  await db.delete(kioskSessions).where(eq(kioskSessions.token, token));
}

### STEP 6: Add in-memory cache for hot sessions (performance optimization)

DB lookup on every request (17 endpoints) could be slow. Add a thin cache layer:

```typescript
const sessionCache = new Map<string, { userId: string; stationId?: number; cachedAt: number }>();
const CACHE_TTL = 30 * 1000; // 30 seconds cache

export async function validateKioskSession(token: string): Promise<{ userId: string; stationId?: number } | null> {
  if (!token) return null;
  
  // Check cache first
  const cached = sessionCache.get(token);
  if (cached && Date.now() - cached.cachedAt < CACHE_TTL) {
    return { userId: cached.userId, stationId: cached.stationId };
  }
  
  // Cache miss — query DB
  const [session] = await db.select().from(kioskSessions).where(eq(kioskSessions.token, token)).limit(1);
  
  if (!session || new Date(session.expiresAt) < new Date()) {
    sessionCache.delete(token);
    if (session) await db.delete(kioskSessions).where(eq(kioskSessions.token, token));
    return null;
  }
  
  // Update cache
  sessionCache.set(token, { userId: session.userId, stationId: session.stationId ?? undefined, cachedAt: Date.now() });
  
  return { userId: session.userId, stationId: session.stationId ?? undefined };
}

// Clear cache on delete/update
export async function deleteKioskSession(token: string): Promise<void> {
  sessionCache.delete(token);
  await db.delete(kioskSessions).where(eq(kioskSessions.token, token));
}

export async function updateKioskStation(token: string, stationId: number): Promise<void> {
  sessionCache.delete(token); // Invalidate cache
  await db.update(kioskSessions).set({ stationId }).where(eq(kioskSessions.token, token));
}
```
