### 1A: Break Time Auto-Calculation

When creating/generating shifts, auto-calculate break times:

```typescript
// Break scheduling algorithm:
function calculateBreakTime(
  shiftStart: string, // "08:00"
  shiftEnd: string,   // "17:00"
  existingBreaks: { start: string; end: string }[], // other staff breaks
  breakDuration: number = 60 // minutes
): { breakStart: string; breakEnd: string } {
  
  const startMinutes = timeToMinutes(shiftStart);
  const endMinutes = timeToMinutes(shiftEnd);
  const shiftDuration = endMinutes - startMinutes;
  
  // Break window: 3.5h to 5h after shift start
  const earliestBreak = startMinutes + 210; // 3.5 hours
  const latestBreak = startMinutes + 300;   // 5 hours
  
  // If shift is < 6 hours, no mandatory break
  if (shiftDuration < 360) return { breakStart: '', breakEnd: '' };
  
  // Find first available slot with 10-minute gap from other breaks
  for (let candidate = earliestBreak; candidate <= latestBreak; candidate += 10) {
    const candidateEnd = candidate + breakDuration;
    
    // Check: no overlap with existing breaks + 10min gap
    const hasConflict = existingBreaks.some(existing => {
      const existStart = timeToMinutes(existing.start);
      const existEnd = timeToMinutes(existing.end);
      // Must have 10min gap before or after
      return !(candidateEnd + 10 <= existStart || candidate >= existEnd + 10);
    });
    
    if (!hasConflict) {
      return {
        breakStart: minutesToTime(candidate),
        breakEnd: minutesToTime(candidateEnd),
      };
    }
  }
  
  // Fallback: schedule at midpoint of shift
  const mid = Math.floor((startMinutes + endMinutes) / 2) - 30;
  return {
    breakStart: minutesToTime(mid),
    breakEnd: minutesToTime(mid + breakDuration),
  };
}

function timeToMinutes(time: string): number {
  const [h, m] = time.split(':').map(Number);
  return h * 60 + m;
}

function minutesToTime(minutes: number): string {
  const h = Math.floor(minutes / 60).toString().padStart(2, '0');
  const m = (minutes % 60).toString().padStart(2, '0');
  return `${h}:${m}`;
}
```

### 1B: Mola Crosscheck Rules

Add validation to the shift creation/update:

```typescript
// Validation rules for break scheduling:
interface BreakValidation {
  valid: boolean;
  errors: string[];
  warnings: string[];
}

function validateBreakSchedule(
  shifts: Array<{ userId: string; startTime: string; endTime: string; breakStart: string; breakEnd: string; isOff: boolean }>,
  branchSize: 'small' | 'medium' | 'large' // small: ≤6 staff, medium: 7-12, large: 13+
): BreakValidation {
  const errors: string[] = [];
  const warnings: string[] = [];
  
  const maxSimultaneous = branchSize === 'small' ? 1 : branchSize === 'medium' ? 1 : 2;
  const minGapMinutes = 10;
  
  const activeShifts = shifts.filter(s => !s.isOff && s.breakStart);
  
  // Sort by break start
  const sorted = [...activeShifts].sort((a, b) => 
    timeToMinutes(a.breakStart) - timeToMinutes(b.breakStart)
  );
  
  // Check simultaneous breaks
  for (let i = 0; i < sorted.length; i++) {
    let simultaneous = 1;
    const iStart = timeToMinutes(sorted[i].breakStart);
    const iEnd = timeToMinutes(sorted[i].breakEnd);
    
    for (let j = i + 1; j < sorted.length; j++) {
      const jStart = timeToMinutes(sorted[j].breakStart);
      const jEnd = timeToMinutes(sorted[j].breakEnd);
      
      // Overlap check
      if (jStart < iEnd && jEnd > iStart) {
        simultaneous++;
      }
    }
    
    if (simultaneous > maxSimultaneous) {
      errors.push(`Saat ${sorted[i].breakStart}'da ${simultaneous} kişi aynı anda molada — max ${maxSimultaneous}`);
    }
  }
  
  // Check 10-minute gap between consecutive breaks
  for (let i = 0; i < sorted.length - 1; i++) {
    const currentEnd = timeToMinutes(sorted[i].breakEnd);
    const nextStart = timeToMinutes(sorted[i + 1].breakStart);
    const gap = nextStart - currentEnd;
    
    if (gap < minGapMinutes && gap >= 0) {
      warnings.push(`${sorted[i].breakStart} ve ${sorted[i+1].breakStart} molaları arası ${gap}dk — min 10dk olmalı`);
    }
  }
  
  // Check: during each break, minimum staff on floor
  const minOnFloor = branchSize === 'small' ? 1 : 2;
  // For each 10-minute slot between 08:00-18:00, count working (not on break) staff
  for (let slot = 480; slot <= 1080; slot += 10) { // 08:00 to 18:00
    const working = activeShifts.filter(s => {
      const sStart = timeToMinutes(s.startTime);
      const sEnd = timeToMinutes(s.endTime);
      const bStart = timeToMinutes(s.breakStart);
      const bEnd = timeToMinutes(s.breakEnd);
      const onShift = slot >= sStart && slot < sEnd;
      const onBreak = slot >= bStart && slot < bEnd;
      return onShift && !onBreak;
    }).length;
    
    if (working < minOnFloor) {
      errors.push(`Saat ${minutesToTime(slot)}'da şubede sadece ${working} kişi — min ${minOnFloor} olmalı`);
    }
  }
  
  return { valid: errors.length === 0, errors, warnings };
}
```

### 2C: Plan Generation Algorithm

```typescript
function generateWeeklyPlan(
  staff: any[],
  weekStartDate: string,
  leaves: any[],
  prevShifts: any[],
  branchId: number
): ShiftPlan[] {
  const plan: ShiftPlan[] = [];
  const staffCount = staff.length;
  
  // Branch size determines coverage rules
  const branchSize = staffCount <= 6 ? 'small' : staffCount <= 12 ? 'medium' : 'large';
  
  // Track weekly hours per person
  const weeklyHours: Record<string, number> = {};
  staff.forEach(s => { weeklyHours[s.id] = 0; });
  
  // Track OFF days per person
  const offDays: Record<string, number[]> = {};
  staff.forEach(s => { offDays[s.id] = []; });
  
  // Determine which day each person was OFF last week (for rotation)
  const lastWeekOff: Record<string, number> = {};
  prevShifts.forEach(s => {
    if (s.isOff) {
      const day = new Date(s.startTime).getDay();
      lastWeekOff[s.userId] = day;
    }
  });
  
  // For each day of the week (0=Mon to 6=Sun)
  for (let dayOffset = 0; dayOffset < 7; dayOffset++) {
    const date = new Date(weekStartDate);
    date.setDate(date.getDate() + dayOffset);
    const dayOfWeek = date.getDay(); // 0=Sun, 1=Mon, ...
    const isBusyDay = [0, 5, 6].includes(dayOfWeek); // Sun, Fri, Sat
    const isQuietDay = [1, 2].includes(dayOfWeek); // Mon, Tue
    
    // Staff available today (not on leave, not on holiday)
    const availableToday = staff.filter(s => {
      const onLeave = leaves.some(l => {
        const lStart = new Date(l.startDate);
        const lEnd = new Date(l.endDate);
        return date >= lStart && date <= lEnd && l.userId === s.id;
      });
      return !onLeave;
    });
    
    // Determine OFF assignments for today
    // Rule: each person gets 1-2 OFF days/week
    // Rotate: if person was OFF on Tuesday last week, give them Wednesday this week
    const needOff = isQuietDay ? Math.ceil(staffCount * 0.3) : (isBusyDay ? Math.max(1, Math.floor(staffCount * 0.1)) : Math.ceil(staffCount * 0.2));
    
    // Sort by who needs OFF most (fewest OFF days so far this week)
    const offCandidates = availableToday
      .filter(s => offDays[s.id].length < 2) // Max 2 OFF/week
      .filter(s => s.role !== 'mudur' || !isBusyDay) // Manager works busy days
      .sort((a, b) => offDays[a.id].length - offDays[b.id].length);
    
    const todayOff = offCandidates.slice(0, needOff).map(s => s.id);
    todayOff.forEach(id => offDays[id].push(dayOffset));
    
    // Working staff today
    const workingToday = availableToday.filter(s => !todayOff.includes(s.id));
    
    // Assign shift times
    const dayBreaks: { start: string; end: string }[] = [];
    
    for (let i = 0; i < workingToday.length; i++) {
      const person = workingToday[i];
      const isFT = person.employmentType === 'tam_zamanli' || !person.employmentType;
      const targetDailyHours = isFT ? 9 : 6; // FT: 9h shift (8h work + 1h break)
      
      let startTime: string;
      let endTime: string;
      
      // Opening shift (first 2 people)
      if (i < 2) {
        startTime = i === 0 ? '07:30' : '08:00';
        endTime = minutesToTime(timeToMinutes(startTime) + targetDailyHours * 60);
      }
      // Closing shift (last 2-3 people)
      else if (i >= workingToday.length - (isBusyDay ? 3 : 2)) {
        endTime = '17:30';
        startTime = minutesToTime(timeToMinutes(endTime) - targetDailyHours * 60);
      }
      // Middle shifts
      else {
        startTime = '08:00';
        endTime = minutesToTime(timeToMinutes(startTime) + targetDailyHours * 60);
      }
      
      // Calculate break
      const breakInfo = calculateBreakTime(startTime, endTime, dayBreaks, 60);
      dayBreaks.push({ start: breakInfo.breakStart, end: breakInfo.breakEnd });
      
      // Track hours (shift hours minus break = net work)
      const shiftMinutes = timeToMinutes(endTime) - timeToMinutes(startTime);
      const netWorkMinutes = shiftMinutes - 60; // 1h break
      weeklyHours[person.id] += netWorkMinutes / 60;
      
      plan.push({
        userId: person.id,
        branchId,
        date: date.toISOString().split('T')[0],
        startTime,
        endTime,
        breakStartTime: breakInfo.breakStart,
        breakEndTime: breakInfo.breakEnd,
        breakDurationMinutes: 60,
        isOff: false,
        dayOfWeek: dayOfWeek,
        shiftType: i < 2 ? 'opening' : (i >= workingToday.length - 2 ? 'closing' : 'standard'),
      });
    }
    
    // Create OFF records
    todayOff.forEach(userId => {
      plan.push({
        userId,
        branchId,
        date: date.toISOString().split('T')[0],
        startTime: null,
        endTime: null,
        breakStartTime: null,
        breakEndTime: null,
        breakDurationMinutes: 0,
        isOff: true,
        dayOfWeek: dayOfWeek,
        shiftType: 'off',
      });
    });
  }
  
  // === POST-PROCESSING ===
  // Check each FT person has ~45h, adjust if needed
  for (const person of staff) {
    const isFT = person.employmentType === 'tam_zamanli' || !person.employmentType;
    const targetHours = isFT ? 45 : 25;
    const actualHours = weeklyHours[person.id] || 0;
    
    if (Math.abs(actualHours - targetHours) > 2) {
      // Adjust: if under, extend a shift. If over, shorten one.
      // This is a simplified adjustment — AI can suggest better options
    }
  }
  
  return plan;
}

                <div className={`rounded px-1 py-0.5 ${
                  shift.shiftType === 'opening' ? 'bg-blue-500/10 text-blue-400' :
                  shift.shiftType === 'closing' ? 'bg-orange-500/10 text-orange-400' :
                  'bg-green-500/10 text-green-400'
                }`}>
                  <p className="font-mono text-[10px]">{shift.startTime}-{shift.endTime}</p>
                  <p className="text-[8px] text-muted-foreground">Mola: {shift.breakStartTime}</p>
                </div>
              </td>
            );
          })}
          <td className="p-2 text-center font-mono text-[11px]">
            <span className={weeklyHours > 45 ? 'text-red-400' : weeklyHours < 40 ? 'text-amber-400' : 'text-green-400'}>
              {weeklyHours}h
            </span>
          </td>
        </tr>
      ))}
    </tbody>
  </table>
</div>

// Summary bar
<div className="flex gap-4 mt-4 p-3 bg-muted/30 rounded-lg text-xs">
  <span>Toplam personel: {staff.length}</span>
  <span>Çalışan: {workingCount}</span>
  <span>OFF: {offCount}</span>
  <span className={warnings.length > 0 ? 'text-amber-400' : 'text-green-400'}>
    Uyarı: {warnings.length}
  </span>
</div>

// AI Generate button
<Button onClick={generateAIPlan} className="bg-primary">
  AI ile Haftalık Plan Oluştur
</Button>
```
