## 1.1: Periyodik Snapshot Tablosu (⚠ DANIŞMA)

```sql
-- Her ayın sonunda tüm şubelerin KPI'larını donduran tablo
CREATE TABLE IF NOT EXISTS branch_monthly_snapshots (
  id SERIAL PRIMARY KEY,
  branch_id INTEGER NOT NULL REFERENCES branches(id),
  snapshot_month INTEGER NOT NULL,    -- 1-12
  snapshot_year INTEGER NOT NULL,     -- 2026
  
  -- Personel metrikleri
  staff_count INTEGER DEFAULT 0,
  new_hires INTEGER DEFAULT 0,
  terminations INTEGER DEFAULT 0,
  turnover_rate NUMERIC(5,2) DEFAULT 0,
  attendance_rate NUMERIC(5,2) DEFAULT 0,
  late_count INTEGER DEFAULT 0,
  leave_days_used INTEGER DEFAULT 0,
  
  -- Operasyonel metrikler
  task_total INTEGER DEFAULT 0,
  task_completed INTEGER DEFAULT 0,
  task_completion_rate NUMERIC(5,2) DEFAULT 0,
  checklist_completion_rate NUMERIC(5,2) DEFAULT 0,
  
  -- Müşteri metrikleri
  customer_complaints INTEGER DEFAULT 0,
  customer_avg_rating NUMERIC(3,2) DEFAULT 0,
  sla_breaches INTEGER DEFAULT 0,
  tickets_total INTEGER DEFAULT 0,
  tickets_resolved INTEGER DEFAULT 0,
  avg_resolution_hours NUMERIC(8,2) DEFAULT 0,
  
  -- Ekipman metrikleri
  equipment_faults INTEGER DEFAULT 0,
  repeat_faults INTEGER DEFAULT 0,
  avg_repair_hours NUMERIC(8,2) DEFAULT 0,
  
  -- Eğitim metrikleri
  training_completions INTEGER DEFAULT 0,
  avg_quiz_score NUMERIC(5,2) DEFAULT 0,
  certificates_issued INTEGER DEFAULT 0,
  
  -- Finansal (varsa)
  cost_payroll NUMERIC(12,2) DEFAULT 0,
  cost_supplies NUMERIC(12,2) DEFAULT 0,
  cost_total NUMERIC(12,2) DEFAULT 0,
  revenue_total NUMERIC(12,2) DEFAULT 0,
  net_profit NUMERIC(12,2) DEFAULT 0,
  
  -- Skor
  overall_health_score NUMERIC(5,2) DEFAULT 0,  -- 0-100 arası hesaplanan
  
  created_at TIMESTAMP DEFAULT NOW(),
  UNIQUE(branch_id, snapshot_month, snapshot_year)
);

CREATE INDEX IF NOT EXISTS idx_branch_snapshot ON branch_monthly_snapshots(branch_id, snapshot_year, snapshot_month);

-- Fabrika aylık snapshot
CREATE TABLE IF NOT EXISTS factory_monthly_snapshots (
  id SERIAL PRIMARY KEY,
  snapshot_month INTEGER NOT NULL,
  snapshot_year INTEGER NOT NULL,
  
  production_total INTEGER DEFAULT 0,
  production_target INTEGER DEFAULT 0,
  production_rate NUMERIC(5,2) DEFAULT 0,
  waste_rate NUMERIC(5,2) DEFAULT 0,
  qc_pass_rate NUMERIC(5,2) DEFAULT 0,
  active_stations INTEGER DEFAULT 0,
  shipments_count INTEGER DEFAULT 0,
  staff_count INTEGER DEFAULT 0,
  avg_worker_score NUMERIC(5,2) DEFAULT 0,
  
  created_at TIMESTAMP DEFAULT NOW(),
  UNIQUE(snapshot_month, snapshot_year)
);
```

## 1.2: Snapshot Hesaplama Servisi

`server/services/monthly-snapshot-service.ts` oluştur:

```typescript
// Her ayın 1'inde gece 02:00'da çalışır
// Önceki ayın tüm verilerini toplar ve snapshot tablosuna yazar

export async function calculateMonthlySnapshots(month: number, year: number) {
  const branches = await getActiveBranches();
  
  for (const branch of branches) {
    const snapshot = {
      branchId: branch.id,
      snapshotMonth: month,
      snapshotYear: year,
      
      // Personel — users tablosundan
      staffCount: await countActiveStaff(branch.id),
      newHires: await countNewHires(branch.id, month, year),
      terminations: await countTerminations(branch.id, month, year),
      // turnoverRate hesapla
      
      // Devam — shift_attendance (veri yoksa 0 döner)
      attendanceRate: await calcAttendanceRate(branch.id, month, year),
      lateCount: await countLateArrivals(branch.id, month, year),
      
      // Görevler — tasks tablosundan
      taskTotal: await countTasks(branch.id, month, year),
      taskCompleted: await countCompletedTasks(branch.id, month, year),
      
      // Checklist — checklist_completions tablosundan
      checklistCompletionRate: await calcChecklistRate(branch.id, month, year),
      
      // CRM — support_tickets tablosundan
      customerComplaints: await countComplaints(branch.id, month, year),
      customerAvgRating: await calcAvgRating(branch.id, month, year),
      slaBreaches: await countSLABreaches(branch.id, month, year),
      
      // Ekipman — equipment_faults tablosundan
      equipmentFaults: await countFaults(branch.id, month, year),
      
      // Eğitim — training_completions tablosundan
      trainingCompletions: await countTrainingCompletions(branch.id, month, year),
      avgQuizScore: await calcAvgQuizScore(branch.id, month, year),
      
      // Finansal — branch_financial_summary tablosundan (Sprint 7)
      // costPayroll, costSupplies, costTotal, revenueTotal, netProfit
      
      // Genel sağlık skoru (ağırlıklı ortalama)
      // overallHealthScore hesapla (devam %20, görev %15, müşteri %20, ekipman %15, eğitim %15, SLA %15)
    };
    
    // Upsert
    await upsertSnapshot(snapshot);
  }
}
```

**Scheduler'a ekle:** Her ayın 1'i, 02:00 → calculateMonthlySnapshots(öncekiAy, yıl)

## 2.10: Tarih filtresi detaylı tasarım

### DateRangeFilter bileşeni

```
┌────────────────────────────────────────────────────────────────┐
│ [Bugün] [Bu hafta] [Bu ay] [Son 3 ay] [Son 6 ay] [Son 1 yıl] [Özel ▾] │
└────────────────────────────────────────────────────────────────┘

"Özel" tıklanınca:
┌─────────────────────────────────────┐
│  Başlangıç: [01/01/2026]           │
│  Bitiş:     [24/03/2026]           │
│  [Uygula]                          │
└─────────────────────────────────────┘
```

### Tüm dashboard endpoint'leri bu parametreleri destekler:
```
?period=today|this_week|this_month|last_month|last_quarter|last_6months|last_year|custom
?startDate=2026-01-01&endDate=2026-03-24  (period=custom ise)
```

### Periyodik raporlama seçenekleri (raporlar sayfası):
```
Rapor Tipi: [Aylık ▾] [Haftalık ▾] [Yıllık ▾]
Dönem: [Mart 2026 ▾]  veya  [Hafta 12 ▾]  veya  [2025 ▾]

→ Seçilen periyoda göre branch_monthly_snapshots tablosundan veri çeker
→ Karşılaştırma: "Bu ay vs geçen ay" veya "Bu çeyrek vs geçen çeyrek"
```

### Trend grafiklerinde tarih:
- Seçilen periyoda göre x ekseni otomatik ayarlanır
- Bugün → saatlik (09:00, 10:00, ...)
- Bu hafta → günlük (Pzt, Sal, ...)
- Bu ay → haftalık (H1, H2, H3, H4)
- Son 3-6 ay → aylık (Oca, Şub, Mar, ...)
- Son 1 yıl → aylık
- Özel → tarih aralığına göre otomatik (gün/hafta/ay)

## 2.11: Hiyerarşik bildirim filtresi

### Bildirim seviye matrisi oluştur

Mevcut bildirim oluşturma noktalarında seviye kontrolü ekle:

```typescript
// Bildirim seviyeleri
const NOTIFICATION_LEVELS = {
  'operational': ['supervisor', 'mudur'],           // gecikme, checklist, günlük operasyon
  'branch_alert': ['mudur', 'coach'],               // şube bazlı anormallik
  'regional': ['coach', 'trainer', 'cgo'],          // bölge bazlı trend
  'strategic': ['ceo', 'cgo', 'admin'],             // stratejik insight, P&L
  'personal': ['self'],                             // kendi görev, eğitim, izin
  'hr_admin': ['muhasebe_ik', 'ceo', 'cgo'],        // bordro, belge, personel değişikliği
  'factory': ['fabrika_mudur', 'ceo'],               // üretim, kalite, sevkiyat
};

// Bildirim oluşturma öncesi kontrol
function shouldReceiveNotification(notificationType: string, userRole: string): boolean {
  const level = getNotificationLevel(notificationType);
  return NOTIFICATION_LEVELS[level]?.includes(userRole) ?? false;
}
```

### Mevcut bildirim oluşturma noktalarını güncelle

Tüm `createNotification` çağrılarında:
1. Bildirim tipine göre seviye belirle
2. Sadece o seviyedeki rollere gönder
3. CEO'ya operasyonel bildirim gitmesin — sadece haftalık Mr. Dobody özeti gitsin

**⚠ Bu büyük bir değişiklik. Mevcut createNotification fonksiyonuna level parametresi ekle. Eski çağrıları kırmamak için default level = 'operational' olsun.**
