### Ana fonksiyon:

```typescript
export async function analyzeAllBranches(): Promise<BranchInsight[]> {
  const branches = await getActiveBranches();
  const insights: BranchInsight[] = [];
  
  for (const branch of branches) {
    const data = await collectBranchData(branch.id, 30); // 30 gün
    
    for (const rule of CORRELATION_RULES) {
      if (rule.condition(data)) {
        insights.push({
          branchId: branch.id,
          branchName: branch.name,
          ruleId: rule.id,
          ruleName: rule.name,
          severity: rule.severity,
          recommendation: rule.recommendation(data),
          data: data, // ham veri (detay için)
          createdAt: new Date()
        });
      }
    }
  }
  
  return insights;
}
```

// Trigger: Haftalık (Pazartesi 08:00)
// Hedef: CEO, CGO, Coach

action: async () => {
  const insights = await analyzeAllBranches();
  
  if (insights.length === 0) return;
  
  // Severity'ye göre grupla
  const critical = insights.filter(i => i.severity === 'critical');
  const warnings = insights.filter(i => i.severity === 'warning');
  const info = insights.filter(i => i.severity === 'info');
  
  // CEO/CGO'ya özet gönder
  const hqUsers = await getHQLeaders(); // ceo, cgo rolleri
  
  for (const user of hqUsers) {
    const summary = [
      critical.length > 0 ? `🔴 ${critical.length} kritik bulgu` : null,
      warnings.length > 0 ? `🟡 ${warnings.length} uyarı` : null,
      info.length > 0 ? `🟢 ${info.length} olumlu gelişme` : null,
    ].filter(Boolean).join(' | ');
    
    await createAgentAction({
      userId: user.id,
      skillId: 'cross-module-insight',
      type: 'weekly_report',
      title: 'Haftalık şube insight raporu',
      message: summary + '\n\n' + insights.slice(0, 5).map(i => `• ${i.recommendation}`).join('\n'),
      deepLink: '/raporlar/insight',
      priority: critical.length > 0 ? 'high' : 'medium',
      expiresAt: addDays(new Date(), 7)
    });
  }
  
  // Coach'lara kendi şubelerine özel insight
  const coaches = await getCoaches();
  for (const coach of coaches) {
    const coachInsights = insights.filter(i => 
      coach.branchIds?.includes(i.branchId)
    );
    if (coachInsights.length > 0) {
      await createAgentAction({
        userId: coach.id,
        skillId: 'cross-module-insight',
        type: 'coach_insight',
        title: `Şubeleriniz için ${coachInsights.length} insight`,
        message: coachInsights.map(i => `• ${i.recommendation}`).join('\n'),
        deepLink: '/raporlar/insight',
        priority: 'medium',
        expiresAt: addDays(new Date(), 7)
      });
    }
  }
}

## B0: Veri kaynağı analizi

**⚠ ÖN ARAŞTIRMADA kontrol et — gelir verisi var mı?**

Muhtemelen POS/satış verisi YOK (DOSPRESSO platformunda POS entegrasyonu olmayabilir). Bu durumda P&L raporu şu şekilde çalışır:

### Senaryo 1: Gelir verisi VAR
- Gelir = POS/satış tablosundan
- Gider = bordro + satınalma + sabit gider
- Kar = Gelir - Gider

### Senaryo 2: Gelir verisi YOK (büyük ihtimalle bu)
- Gelir = Manuel giriş (admin tarafından aylık) VEYA tahmini (şube kapasitesi × doluluk)
- Gider = bordro + satınalma + sabit gider (bunlar mevcut)
- P&L = sınırlı ama gider analizi yapılabilir

**⚠ DANIŞMA: Gelir verisi yoksa P&L yerine "Gider Analizi + Maliyet Karşılaştırma" raporu oluştur. Bu da çok değerli.**

```typescript
export async function calculateBranchFinancials(branchId: number, month: number, year: number) {
  // 1. Bordro maliyeti (payroll_records'tan)
  const payrollCost = await db.select({ total: sum(payrollRecords.netSalary) })
    .from(payrollRecords)
    .where(and(
      eq(payrollRecords.branchId, branchId),
      eq(payrollRecords.month, month),
      eq(payrollRecords.year, year)
    ));
  
  // 2. Satınalma maliyeti (branch_orders veya cari_transactions'tan)
  const supplyCost = await db.select({ total: sum(branchOrders.totalAmount) })
    .from(branchOrders)
    .where(and(
      eq(branchOrders.branchId, branchId),
      // tarih filtresi
    ));
  
  // 3. Bakım maliyeti (equipment_maintenance_logs'tan — varsa maliyet kolonu)
  const maintenanceCost = await getMaintenanceCost(branchId, month, year);
  
  // 4. Sabit giderler (branch ayarlarından veya manuel giriş)
  const fixedCosts = await getBranchFixedCosts(branchId);
  
  // 5. Gelir (varsa POS, yoksa manual/estimated)
  const revenue = await getBranchRevenue(branchId, month, year);
  
  // Hesapla
  const totalCost = payrollCost + supplyCost + maintenanceCost + fixedCosts.rent + fixedCosts.utilities + fixedCosts.other;
  const netProfit = revenue - totalCost;
  const profitMargin = revenue > 0 ? (netProfit / revenue * 100) : 0;
  
  // Kaydet (upsert)
  await db.insert(branchFinancialSummary).values({
    branchId, periodMonth: month, periodYear: year,
    costPayroll: payrollCost,
    costSupplies: supplyCost,
    costMaintenance: maintenanceCost,
    costRent: fixedCosts.rent,
    costUtilities: fixedCosts.utilities,
    costOther: fixedCosts.other,
    revenueTotal: revenue,
    totalCost, netProfit, profitMargin,
    staffCount: await getBranchStaffCount(branchId),
    costPerEmployee: payrollCost / await getBranchStaffCount(branchId),
    calculatedAt: new Date()
  }).onConflictDoUpdate({
    target: [branchFinancialSummary.branchId, branchFinancialSummary.periodMonth, branchFinancialSummary.periodYear],
    set: { /* aynı alanlar */ }
  });
}
```

// Trigger: Aylık (her ayın 5'i — bordro hesaplandıktan sonra)
// Hedef: CEO, CGO

action: async () => {
  const currentMonth = new Date().getMonth(); // önceki ay
  const currentYear = new Date().getFullYear();
  
  // Tüm şubeler için hesapla
  const branches = await getActiveBranches();
  for (const branch of branches) {
    await calculateBranchFinancials(branch.id, currentMonth, currentYear);
  }
  
  // Analiz
  const summaries = await getAllBranchFinancials(currentMonth, currentYear);
  const unprofitable = summaries.filter(s => s.netProfit < 0);
  const highCost = summaries.filter(s => s.costPerEmployee > threshold);
  
  // CEO'ya rapor
  if (unprofitable.length > 0 || highCost.length > 0) {
    await createAgentAction({
      userId: getCEOId(),
      skillId: 'financial-insight',
      type: 'monthly_financial',
      title: `Aylık finansal rapor hazır`,
      message: [
        unprofitable.length > 0 ? `${unprofitable.length} şube zararda: ${unprofitable.map(s => s.branchName).join(', ')}` : null,
        highCost.length > 0 ? `${highCost.length} şubede personel maliyeti yüksek` : null,
        `En kârlı: ${summaries.sort((a,b) => b.netProfit - a.netProfit)[0]?.branchName}`
      ].filter(Boolean).join('\n'),
      deepLink: '/raporlar/sube-finansal',
      priority: unprofitable.length > 0 ? 'high' : 'medium'
    });
  }
}
