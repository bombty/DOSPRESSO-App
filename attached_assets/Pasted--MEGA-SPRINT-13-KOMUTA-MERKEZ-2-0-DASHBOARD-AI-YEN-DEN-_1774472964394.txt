function groupAlerts(alerts: Alert[]): GroupedAlert[] {
  const groups = new Map<string, Alert[]>();
  alerts.forEach(alert => {
    const key = alert.branchId || alert.category || 'general';
    if (!groups.has(key)) groups.set(key, []);
    groups.get(key)!.push(alert);
  });
  
  return Array.from(groups.entries()).map(([key, items]) => ({
    groupKey: key,
    groupLabel: items[0].branchName || items[0].category,
    count: items.length,
    highestSeverity: getHighestSeverity(items),
    alerts: items,
    isExpanded: false  // varsayılan kapalı
  }));
}

// Bildirim oluşturma öncesi kontrol fonksiyonu
async function shouldSendNotification(userId: string, type: string): Promise<boolean> {
  // Son 24 saatte aynı tip bildirim gönderildi mi?
  const existing = await db.select()
    .from(notifications)
    .where(
      and(
        eq(notifications.userId, userId),
        eq(notifications.type, type),
        gt(notifications.createdAt, sql`NOW() - INTERVAL '24 hours'`)
      )
    )
    .limit(1);
  
  return existing.length === 0;  // yoksa gönder, varsa gönderme
}

## 2.2: Dashboard section'larını collapsible yap

**Prensip:** Her section kartı collapsible olsun — başlık her zaman görünür, içerik toggle ile açılır/kapanır. Varsayılan: en önemli 3 section açık, geri kalan kapalı.

```typescript
// Yeni shared bileşen: CollapsibleSection
interface CollapsibleSectionProps {
  title: string;
  icon?: React.ReactNode;
  badge?: string | number;  // "3 uyarı" veya "İyi" gibi
  badgeColor?: 'success' | 'warning' | 'danger' | 'info';
  defaultOpen?: boolean;
  children: React.ReactNode;
}

function CollapsibleSection({ title, icon, badge, badgeColor, defaultOpen = false, children }: CollapsibleSectionProps) {
  const [isOpen, setIsOpen] = useState(defaultOpen);
  return (
    <div className="border border-border/40 rounded-lg overflow-hidden">
      <button onClick={() => setIsOpen(!isOpen)}
        className="w-full flex items-center justify-between px-4 py-3 hover:bg-muted/50 transition-colors">
        <div className="flex items-center gap-2">
          {icon}
          <span className="font-semibold text-sm">{title}</span>
          {badge && <span className={cn("text-xs px-2 py-0.5 rounded-full", badgeColorClass)}>{badge}</span>}
        </div>
        <ChevronDown className={cn("w-4 h-4 transition-transform", isOpen && "rotate-180")} />
      </button>
      {isOpen && <div className="px-4 pb-4">{children}</div>}
    </div>
  );
}
```

  maxTokens?: number;
  jsonMode?: boolean;  // JSON response istiyorsa
}

interface AICompletionResponse {
  content: string;
  model: string;
  provider: string;
  tokensUsed: number;
}

// ai_settings tablosundan oku — her çağrıda CACHE'lenmiş config kullanır
let cachedSettings: any = null;
let cacheTime = 0;

async function getAISettings() {
  if (cachedSettings && Date.now() - cacheTime < 60000) return cachedSettings;  // 1 dk cache
  const [settings] = await db.select().from(aiSettings).limit(1);
  cachedSettings = settings;
  cacheTime = Date.now();
  return settings;
}

export async function aiComplete(req: AICompletionRequest): Promise<AICompletionResponse> {
  const settings = await getAISettings();
  
  if (!settings?.isActive) {
    throw new Error('AI servisi aktif değil');
  }

  const provider = settings.provider;  // 'openai' | 'anthropic' | 'gemini'
  const model = settings.chatModel;    // 'gpt-4o-mini' | 'claude-sonnet-4-20250514' | 'gemini-2.5-pro'
  const temperature = req.temperature ?? settings.temperature ?? 0.3;
  const maxTokens = req.maxTokens ?? settings.maxTokens ?? 1000;

  try {
    let content: string;
    let tokensUsed: number;

    if (provider === 'openai') {
      const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });
      const response = await openai.chat.completions.create({
        model: model,
        messages: [
          { role: 'system', content: req.systemPrompt },
          { role: 'user', content: req.userPrompt },
        ],
        temperature,
        max_tokens: maxTokens,
        ...(req.jsonMode ? { response_format: { type: 'json_object' } } : {}),
      });
      content = response.choices[0]?.message?.content || '';
      tokensUsed = response.usage?.total_tokens || 0;

    } else if (provider === 'anthropic') {
      // Claude API
      const response = await fetch('https://api.anthropic.com/v1/messages', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'x-api-key': process.env.ANTHROPIC_API_KEY || '',
          'anthropic-version': '2023-06-01',
        },
        body: JSON.stringify({
          model: model,
          max_tokens: maxTokens,
          system: req.systemPrompt,
          messages: [{ role: 'user', content: req.userPrompt }],
        }),
      });
      const data = await response.json();
      content = data.content?.[0]?.text || '';
      tokensUsed = (data.usage?.input_tokens || 0) + (data.usage?.output_tokens || 0);

    } else if (provider === 'gemini') {
      // Google Gemini API
      const response = await fetch(
        `https://generativelanguage.googleapis.com/v1beta/models/${model}:generateContent?key=${process.env.GEMINI_API_KEY}`,
        {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({
            systemInstruction: { parts: [{ text: req.systemPrompt }] },
            contents: [{ parts: [{ text: req.userPrompt }] }],
            generationConfig: { temperature, maxOutputTokens: maxTokens },
          }),
        }
      );
      const data = await response.json();
      content = data.candidates?.[0]?.content?.parts?.[0]?.text || '';
      tokensUsed = data.usageMetadata?.totalTokenCount || 0;

    } else {
      throw new Error(`Bilinmeyen AI provider: ${provider}`);
    }

    // Usage log kaydet
    await logAIUsage(provider, model, tokensUsed);

    return { content, model, provider, tokensUsed };
  } catch (error) {
    console.error(`AI ${provider} hatası:`, error);
    throw error;
  }
}

// Kısa yardımcı — JSON parse ile
export async function aiCompleteJSON<T>(req: AICompletionRequest): Promise<T> {
  const response = await aiComplete({ ...req, jsonMode: true });
  return JSON.parse(response.content);
}
```

// server/routes/dashboard-data-routes.ts'e ekle:
router.get('/api/dashboard/briefing', requireAuth, async (req, res) => {
  const user = req.user as AuthUser;
  
  // Önce dashboard verisini topla (mevcut API'lerden)
  let dashboardData;
  if (isHQRole(user.role)) {
    dashboardData = await getExecutiveDashboard(user);
  } else if (['coach', 'trainer'].includes(user.role)) {
    dashboardData = await getCoachDashboard(user);
  } else if (['supervisor', 'mudur'].includes(user.role)) {
    dashboardData = await getBranchDashboard(user.branchId);
  } else {
    dashboardData = await getBaristaDashboard(user);
  }

  // AI briefing üret (cache: 1 saat)
  const cacheKey = `briefing:${user.id}:${new Date().toISOString().split('T')[0]}`;
  let briefing = await getFromCache(cacheKey);
  
  if (!briefing) {
    briefing = await generateDashboardBriefing(user.id, user.role, dashboardData);
    await setCache(cacheKey, briefing, 3600);  // 1 saat cache
  }

  res.json(briefing);
});
```
