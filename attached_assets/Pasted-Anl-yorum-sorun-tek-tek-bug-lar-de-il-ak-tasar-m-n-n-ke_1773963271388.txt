  const userId = req.query.userId as string;
  if (!userId) return res.status(400).json({ error: "userId gerekli" });
  
  const today = new Date();
  today.setHours(0, 0, 0, 0);
  
  // Total production today
  const production = await db.select({ 
    total: sql`COALESCE(sum(quantity), 0)` 
  }).from(factoryProductionOutputs)
    .where(and(
      eq(factoryProductionOutputs.userId, userId),
      gte(factoryProductionOutputs.createdAt, today)
    ));
  
  // Total break minutes today
  const breaks = await db.select({ 
    total: sql`COALESCE(sum(duration_minutes), 0)` 
  }).from(factoryBreakLogs)
    .where(and(
      eq(factoryBreakLogs.userId, userId),
      gte(factoryBreakLogs.startedAt, today)
    ));
  
  // Shift start time
  const shift = await db.select().from(factoryShiftSessions)
    .where(and(
      eq(factoryShiftSessions.userId, userId),
      eq(factoryShiftSessions.status, 'active')
    ))
    .orderBy(desc(factoryShiftSessions.startedAt))
    .limit(1);
  
  const shiftStart = shift[0]?.startedAt;
  const shiftMinutes = shiftStart ? Math.floor((Date.now() - new Date(shiftStart).getTime()) / 60000) : 0;
  const shiftHours = `${Math.floor(shiftMinutes / 60)}:${String(shiftMinutes % 60).padStart(2, '0')}`;
  
  res.json({
    produced: Number(production[0]?.total || 0),
    breakMinutes: Number(breaks[0]?.total || 0),
    shiftHours,
    shiftStartTime: shiftStart,
  });
});
