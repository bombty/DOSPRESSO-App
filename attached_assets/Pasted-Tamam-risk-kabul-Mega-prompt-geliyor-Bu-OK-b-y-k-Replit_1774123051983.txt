```typescript
interface PayrollInput {
  userId: string;
  month: number; // 1-12
  year: number;
  baseSalaryGross: number; // Brüt maaş
  employmentType: 'tam_zamanli' | 'yari_zamanli' | 'yevmiye';
  // From PDKS
  totalWorkedMinutes: number;
  expectedMinutes: number;
  overtimeMinutes: number;
  deficitMinutes: number;
  lateCount: number;
  earlyLeaveCount: number;
  forgottenCheckoutCount: number;
  absentDays: number;
  // Multipliers
  offDayWorkedMinutes: number; // 1.5x
  holidayWorkedMinutes: number; // 2.0x
  // Extras
  salesBonus: number;
  cashBonus: number;
  performanceBonus: number;
}

interface PayrollResult {
  grossSalary: number;
  overtimePay: number;
  offDayPay: number;
  holidayPay: number;
  totalGross: number;
  sgkEmployee: number; // %14
  unemploymentEmployee: number; // %1
  incomeTax: number; // kümülatif dilim
  stampTax: number; // %0.759
  agi: number;
  totalDeductions: number;
  totalBonuses: number;
  netSalary: number;
  deficitDeduction: number;
  // Breakdown for export
  breakdown: {
    baseDailyGross: number;
    baseHourlyGross: number;
    overtimeHourlyRate: number;
    workedDays: number;
    expectedDays: number;
  };
}

export function calculateMonthlyPayroll(input: PayrollInput, cumulativeTaxBase: number = 0): PayrollResult {
  // Turkish labor law calculation
  const dailyGross = input.baseSalaryGross / 30; // Always 30 days
  const hourlyGross = dailyGross / 7.5;
  
  // Base salary (always full month if employed full month)
  let grossSalary = input.baseSalaryGross;
  
  // Deficit deduction (unauthorized absence)
  const deficitDeduction = input.absentDays > 0 
    ? (input.absentDays * dailyGross) 
    : 0;
  grossSalary -= deficitDeduction;
  
  // Overtime pay
  const overtimeHours = input.overtimeMinutes / 60;
  const overtimePay = overtimeHours * hourlyGross * 1.5;
  
  // OFF day work pay
  const offDayHours = input.offDayWorkedMinutes / 60;
  const offDayPay = offDayHours * hourlyGross * 1.5;
  
  // Holiday work pay
  const holidayHours = input.holidayWorkedMinutes / 60;
  const holidayPay = holidayHours * hourlyGross * 2.0;
  
  // Total gross
  const totalBonuses = input.salesBonus + input.cashBonus + input.performanceBonus;
  const totalGross = grossSalary + overtimePay + offDayPay + holidayPay + totalBonuses;
  
  // SGK (Social Security)
  const sgkEmployee = totalGross * 0.14;
  const unemploymentEmployee = totalGross * 0.01;
  const sgkMatrah = totalGross - sgkEmployee - unemploymentEmployee;
  
  // Income tax (cumulative brackets)
  const newCumulative = cumulativeTaxBase + sgkMatrah;
  const incomeTax = calculateIncomeTax(sgkMatrah, cumulativeTaxBase);
  
  // Stamp tax
  const stampTax = totalGross * 0.00759;
  
  // AGI (Minimum Living Allowance) — simplified, depends on marital status
  const agi = 0; // To be configured per employee
  
  // Total deductions
  const totalDeductions = sgkEmployee + unemploymentEmployee + incomeTax + stampTax;
  
  // Net salary
  const netSalary = totalGross - totalDeductions + agi;
  
  return {
    grossSalary,
    overtimePay,
    offDayPay,
    holidayPay,
    totalGross,
    sgkEmployee,
    unemploymentEmployee,
    incomeTax,
    stampTax,
    agi,
    totalDeductions,
    totalBonuses,
    netSalary,
    deficitDeduction,
    breakdown: {
      baseDailyGross: dailyGross,
      baseHourlyGross: hourlyGross,
      overtimeHourlyRate: hourlyGross * 1.5,
      workedDays: Math.round(input.totalWorkedMinutes / 60 / 7.5),
      expectedDays: Math.round(input.expectedMinutes / 60 / 7.5),
    },
  };
}

function calculateIncomeTax(taxableIncome: number, cumulativeBase: number): number {
  // 2026 Turkish income tax brackets (approximate — should be configurable)
  const brackets = [
    { limit: 110000, rate: 0.15 },
    { limit: 230000, rate: 0.20 },
    { limit: 870000, rate: 0.27 },
    { limit: 3000000, rate: 0.35 },
    { limit: Infinity, rate: 0.40 },
  ];
  
  let remaining = taxableIncome;
  let tax = 0;
  let cumPos = cumulativeBase;
  
  for (const bracket of brackets) {
    if (cumPos >= bracket.limit) continue;
    const spaceInBracket = bracket.limit - cumPos;
    const taxableInBracket = Math.min(remaining, spaceInBracket);
    tax += taxableInBracket * bracket.rate;
    remaining -= taxableInBracket;
    cumPos += taxableInBracket;
    if (remaining <= 0) break;
  }
  
  return Math.round(tax * 100) / 100;
}
```
