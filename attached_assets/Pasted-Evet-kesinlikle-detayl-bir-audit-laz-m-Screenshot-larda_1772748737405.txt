```bash
# 4. Her sayfanın navigasyon pattern'i
# Hangi sayfalar hangi pattern kullanıyor?
for f in client/src/pages/*.tsx; do
  name=$(basename "$f" .tsx)
  has_tabs=$(grep -c "TabsTrigger" "$f" 2>/dev/null || echo 0)

  if [ "$has_tabs" -gt 0 ] || [ "$has_sidenav" -gt 0 ]; then
    echo "$name: tabs=$has_tabs, sidenav=$has_sidenav"
  fi
done

# 5. Breadcrumb tutarlılığı
grep -rn "Breadcrumb\|breadcrumb\|useBreadcrumb" client/src/pages/ --include="*.tsx" -l | wc -l
```
