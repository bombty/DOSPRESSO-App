```sql

UPDATE users 
SET dashboard_preferences = jsonb_set(
  COALESCE(dashboard_preferences, '{}')::jsonb, 
  '{mode}', 
  '"mission-control"'
)
WHERE dashboard_preferences IS NOT NULL 
AND dashboard_preferences::text LIKE '%classic%';
```
