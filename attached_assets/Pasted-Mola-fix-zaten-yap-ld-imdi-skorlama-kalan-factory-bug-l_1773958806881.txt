```typescript
// server/services/factory-scoring-service.ts

interface FactoryWorkerScore {
  userId: string;
  periodStart: Date;
  periodEnd: Date;
  productionScore: number;    // 0-100
  wasteScore: number;         // 0-100
  qualityScore: number;       // 0-100
  attendanceScore: number;    // 0-100
  breakScore: number;         // 0-100
  totalScore: number;         // 0-100 weighted average
  details: {
    totalProduced: number;
    targetProduction: number;
    wasteCount: number;
    wasteRate: number;
    qualityChecks: number;
    qualityPassRate: number;
    totalShiftMinutes: number;
    expectedShiftMinutes: number;
    lateCount: number;
    totalBreakMinutes: number;
    allowedBreakMinutes: number;
  };
}

// Weights for scoring
const SCORING_WEIGHTS = {
  production: 0.35,    // Üretim verimi
  waste: 0.10,         // Fire/hurda oranı (düşük ağırlık — henüz az veri)
  quality: 0.15,       // Kalite skoru
  attendance: 0.25,    // Devam düzeni
  break: 0.15,         // Mola disiplini
};

export async function calculateWorkerScore(
  userId: string, 
  periodStart: Date, 
  periodEnd: Date
): Promise<FactoryWorkerScore> {
  
  // 1. PRODUCTION SCORE (35%)
  // Query: factory_production_outputs for this user in period
  // Compare against factory_station_benchmarks target
  // Score = min(100, (actual_output / target_output) * 100)
  // If user worked multiple stations, weight by time spent at each
  
  // 2. WASTE SCORE (10%)  
  // Query: waste/reject data from production runs
  // Score = max(0, 100 - (waste_rate * penalty_factor))
  // If no waste data exists, default to 80 (neutral)
  
  // 3. QUALITY SCORE (15%)
  // Query: factory_quality_checks for this user
  // Score = (passed_checks / total_checks) * 100
  // If no quality data, default to 75 (neutral)
  
  // 4. ATTENDANCE SCORE (25%)
  // Query: factory_shift_sessions for this user
  // Calculate: actual worked minutes vs expected shift minutes
  // Deduct for late arrivals: -5 points per late day
  // Score = min(100, (actual_minutes / expected_minutes) * 100) - late_penalty
  // Also check PDKS records if available
  
  // 5. BREAK SCORE (15%)
  // Query: factory_break_logs for this user
  // Calculate: total break minutes vs allowed break minutes
  // Standard: 30 min break per 8 hour shift
  // Score = max(0, 100 - excess_break_penalty)
  // excess_break_penalty = max(0, (total_break - allowed_break) / allowed_break * 50)
  
  // 6. TOTAL SCORE (weighted)
  // total = production * 0.35 + waste * 0.10 + quality * 0.15 + attendance * 0.25 + break * 0.15
  
  // Return complete score object
}

export async function calculateAllWorkerScores(
  periodStart: Date, 
  periodEnd: Date
): Promise<FactoryWorkerScore[]> {
  // Get all active factory workers
  const factoryBranch = await db.select().from(branches)
    .where(like(branches.name, '%Fabrika%')).limit(1);
  
  if (!factoryBranch.length) return [];
  
  const workers = await db.select().from(users)
    .where(and(
      eq(users.branchId, factoryBranch[0].id),
      eq(users.isActive, true),
      inArray(users.role, ['fabrika_operator', 'stajyer', 'supervisor', 'supervisor_buddy'])
    ));
  
  const scores: FactoryWorkerScore[] = [];
  for (const worker of workers) {
    try {
      const score = await calculateWorkerScore(worker.id, periodStart, periodEnd);
      scores.push(score);
    } catch (err) {
      console.error(`[Factory Scoring] Error for ${worker.firstName}:`, err);
    }
  }
  
  return scores;
}

export async function saveWorkerScores(scores: FactoryWorkerScore[]): Promise<void> {
  for (const score of scores) {
    // Upsert into factory_worker_scores
    // Check if record exists for this user + period
    const existing = await db.select().from(factoryWorkerScores)
      .where(and(
        eq(factoryWorkerScores.userId, score.userId),
        eq(factoryWorkerScores.periodStart, score.periodStart)
      )).limit(1);
    
    if (existing.length > 0) {
      await db.update(factoryWorkerScores)
        .set({
          productionScore: score.productionScore,
          wasteScore: score.wasteScore,
          qualityScore: score.qualityScore,
          attendanceScore: score.attendanceScore,
          breakScore: score.breakScore,
          totalScore: score.totalScore,
          updatedAt: new Date(),
        })
        .where(eq(factoryWorkerScores.id, existing[0].id));
    } else {
      await db.insert(factoryWorkerScores).values({
        userId: score.userId,
        periodStart: score.periodStart,
        periodEnd: score.periodEnd,
        productionScore: score.productionScore,
        wasteScore: score.wasteScore,
        qualityScore: score.qualityScore,
        attendanceScore: score.attendanceScore,
        breakScore: score.breakScore,
        totalScore: score.totalScore,
      });
    }
  }
}
```

```typescript
async function backfillFactoryScores() {
  const nullScores = await db.select().from(factoryWorkerScores)
    .where(isNull(factoryWorkerScores.totalScore))
    .limit(50);
  
  for (const record of nullScores) {
    try {
      const score = await calculateWorkerScore(
        record.userId, 
        new Date(record.periodStart), 
        new Date(record.periodEnd)
      );
      await db.update(factoryWorkerScores)
        .set({
          productionScore: score.productionScore,
          wasteScore: score.wasteScore,
          qualityScore: score.qualityScore,
          attendanceScore: score.attendanceScore,
          breakScore: score.breakScore,
          totalScore: score.totalScore,
          updatedAt: new Date(),
        })
        .where(eq(factoryWorkerScores.id, record.id));
    } catch (err) {
      console.error(`[Factory Scoring] Backfill error:`, err);
    }
  }
  
  if (nullScores.length > 0) {
    console.log(`[Factory Scoring] Backfilled ${nullScores.length} NULL scores`);
  }
```
