### BÖLÜM 3: Backend API Sağlık Kontrolü

```bash
# 1. Tüm API route dosyaları
ls server/routes/*.ts 2>/dev/null

# 2. Toplam endpoint sayısı
grep -rn "router\.\(get\|post\|put\|patch\|delete\)" server/routes/ --include="*.ts" | wc -l

# 3. Her route dosyasında kaç endpoint var?
for f in server/routes/*.ts; do
  count=$(grep -c "router\.\(get\|post\|put\|patch\|delete\)" "$f" 2>/dev/null)
  name=$(basename "$f" .ts)
  echo "$name: $count endpoint"
done | sort -t: -k2 -rn

# 4. Hata dönen endpoint'ler (500 error)
grep -rn "500\|Internal Server\|catch.*error\|throw.*Error" server/routes/ --include="*.ts" | head -20
```

### BÖLÜM 4: Veritabanı Durum Kontrolü

```sql
-- 1. Toplam tablo sayısı
SELECT COUNT(*) FROM information_schema.tables WHERE table_schema = 'public';

-- 2. En büyük 20 tablo (kayıt sayısı)
SELECT schemaname, tablename, n_live_tup 
FROM pg_stat_user_tables 
ORDER BY n_live_tup DESC LIMIT 20;

-- 3. Boş tablolar (0 kayıt)
SELECT tablename FROM pg_stat_user_tables WHERE n_live_tup = 0 ORDER BY tablename;

-- 4. Aktif kullanıcı ve rol dağılımı
SELECT role, COUNT(*) FROM users WHERE is_active = true GROUP BY role ORDER BY count DESC;

-- 5. Aktif şube sayısı
SELECT COUNT(*) FROM branches WHERE is_active = true;

-- 6. Son 24 saatteki aktivite
SELECT 'notifications' as tablo, COUNT(*) FROM notifications WHERE created_at > NOW() - INTERVAL '24 hours'
UNION ALL SELECT 'tasks', COUNT(*) FROM tasks WHERE created_at > NOW() - INTERVAL '24 hours'
UNION ALL SELECT 'training_completions', COUNT(*) FROM training_completions WHERE completed_at > NOW() - INTERVAL '24 hours'
UNION ALL SELECT 'customer_feedback', COUNT(*) FROM customer_feedback WHERE created_at > NOW() - INTERVAL '24 hours';
```
