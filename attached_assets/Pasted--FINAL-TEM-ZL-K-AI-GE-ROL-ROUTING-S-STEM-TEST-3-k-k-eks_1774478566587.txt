## B1: filterSuggestionsForRole fonksiyonu

`server/lib/dobody-suggestions.ts`'e ekle:

```typescript
/**
 * Rol bazlı suggestion filtreleme
 * CEO: sadece konsolide + kritik
 * CGO: konsolide + tüm uyarılar
 * Coach: kendi şubeleri (zaten getCoachSuggestions ile ayrı)
 * Kalite: sadece checklist/hijyen/kalite
 * Muhasebe: sadece İK/bordro ilgili
 * Diğer HQ: tümü
 */
export function filterSuggestionsForRole(
  suggestions: any[],
  role: string
): any[] {
  switch (role) {
    case 'ceo':
      // CEO sadece konsolide özetler ve kritik uyarılar görür
      return suggestions.filter(s =>
        s.consolidated === true ||
        s.severity === 'critical' ||
        s.type === 'strategic'
      );

    case 'cgo':
      // CGO konsolide + tüm uyarılar (operasyonel detay hariç)
      return suggestions.filter(s =>
        s.consolidated === true ||
        s.severity === 'critical' ||
        s.severity === 'warning' ||
        s.type !== 'operational_detail'
      );

    case 'kalite_kontrol':
    case 'gida_muhendisi':
      // Sadece kalite ile ilgili
      return suggestions.filter(s =>
        s.type === 'checklist_inactive' ||
        s.type === 'consolidated-inactive-checklist' ||
        s.type === 'quality' ||
        s.type === 'hygiene' ||
        s.type === 'compliance' ||
        s.type === 'consolidated-compliance-aging' ||
        s.consolidated === true
      );

    case 'muhasebe_ik':
    case 'muhasebe':
      // Sadece İK/personel ile ilgili
      return suggestions.filter(s =>
        s.type === 'attendance' ||
        s.type === 'leave' ||
        s.type === 'payroll' ||
        s.type === 'personnel' ||
        s.consolidated === true
      );

    case 'marketing':
      // Sadece müşteri/kampanya ile ilgili
      return suggestions.filter(s =>
        s.type === 'customer' ||
        s.type === 'campaign' ||
        s.type === 'feedback' ||
        s.consolidated === true
      );

    case 'satinalma':
      // Sadece tedarik/stok ile ilgili
      return suggestions.filter(s =>
        s.type === 'supply' ||
        s.type === 'stock' ||
        s.type === 'procurement' ||
        s.consolidated === true
      );

    default:
      // Admin, Coach, Trainer, diğerleri: tümünü görsün
      return suggestions;
  }
}
```

## B2: getHQSuggestions'a rol filtresi entegre et

```typescript
// MEVCUT: getHQSuggestions() → tüm öneriler
// YENİ: getHQSuggestions(role) → rol filtreli öneriler

// Fonksiyon imzasını genişlet:
export async function getHQSuggestions(role?: string): Promise<Suggestion[]> {
  // ... mevcut suggestion toplama kodu ...
  
  let suggestions = [...allSuggestions];
  
  // Rol filtresi uygula
  if (role) {
    suggestions = filterSuggestionsForRole(suggestions, role);
  }
  
  return suggestions;
}
```
