### T005: Compact Status Badges on Mobile

Find all Badge usages with status text:
```bash
grep -rn "<Badge" client/src/pages/ --include="*.tsx" | grep -i "bekle\|orta\|yüksek\|düşük\|kritik\|tamamlan\|devam\|acil\|kapalı\|aktif\|çözül" | head -30
```

Create a reusable helper component `client/src/components/compact-badge.tsx`:
```typescript
import { Badge } from "@/components/ui/badge";

interface CompactBadgeProps {
  label: string;
  shortLabel?: string; // Mobile abbreviated version
  variant?: "default" | "secondary" | "destructive" | "outline";
  className?: string;
  color?: string; // Optional custom color class
}

export function CompactBadge({ label, shortLabel, variant = "default", className, color }: CompactBadgeProps) {
  return (
    <Badge variant={variant} className={`${color || ""} text-[10px] px-1.5 py-0.5 md:text-xs md:px-2.5 md:py-0.5 ${className || ""}`}>
      {shortLabel ? (
        <>
          <span className="md:hidden">{shortLabel}</span>
          <span className="hidden md:inline">{label}</span>
        </>
      ) : (
        label
      )}
    </Badge>
  );
}
```
