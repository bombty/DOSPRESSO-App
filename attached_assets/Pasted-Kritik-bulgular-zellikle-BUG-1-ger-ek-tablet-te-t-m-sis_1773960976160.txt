function kioskFetch(url: string, options: RequestInit = {}): Promise<Response> {
  const token = localStorage.getItem("factory-kiosk-token");

    ...options,
    credentials: 'include',
    headers: {
      "Content-Type": "application/json",

      ...(options.headers || {}),

}

// Close kiosk sessions older than 12 hours
async function cleanupStaleFactoryKioskSessions() {
  const twelveHoursAgo = new Date(Date.now() - 12 * 60 * 60 * 1000);
  
  const stale = await db.update(factoryKioskSessions)
    .set({ 
      status: 'completed', 
      endedAt: new Date(),
      notes: 'Otomatik kapatıldı — 12 saatten eski'
    })
    .where(and(
      eq(factoryKioskSessions.status, 'active'),
      lt(factoryKioskSessions.startedAt, twelveHoursAgo)
    ))
    .returning({ id: factoryKioskSessions.id });
  
  if (stale.length > 0) {
    console.log(`[Factory Kiosk] Cleaned ${stale.length} stale sessions`);
  }
}
