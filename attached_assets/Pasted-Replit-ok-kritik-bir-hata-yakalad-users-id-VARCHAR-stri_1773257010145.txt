```typescript
async function checkCrmTicketSla() {
  const overdue = await db.select().from(crmTickets)
    .where(and(
      inArray(crmTickets.status, ['acik', 'atandi', 'isleniyor']),
      lte(crmTickets.slaDeadline, new Date()),
      eq(crmTickets.slaBreached, false),
      eq(crmTickets.isActive, true)
    ));

  for (const ticket of overdue) {
    await db.update(crmTickets)
      .set({ slaBreached: true })
      .where(eq(crmTickets.id, ticket.id));

    // Find escalation role from SLA rules
    const rule = await db.select().from(crmSlaRules)
      .where(eq(crmSlaRules.category, ticket.category)).limit(1);
    
    const escalationRole = rule[0]?.escalationRole || 'cgo';
    const escalationUser = await db.select().from(users)
      .where(and(eq(users.role, escalationRole), eq(users.isActive, true))).limit(1);

    if (escalationUser[0]) {
      await createNotification({
        userId: escalationUser[0].id,
        title: `⚠️ SLA İhlali: ${ticket.subject}`,
        message: `Ticket #${ticket.ticketNumber} — ${ticket.category} kategorisinde SLA süresi aşıldı.`,
        type: 'sla_breach',
        priority: 'high'
      });
    }

    await db.insert(crmTicketHistory).values({
      ticketId: ticket.id,
      action: 'sla_breached',
      newValue: `SLA aşıldı. Escalation: ${escalationRole}`,
    });
  }
}
```
