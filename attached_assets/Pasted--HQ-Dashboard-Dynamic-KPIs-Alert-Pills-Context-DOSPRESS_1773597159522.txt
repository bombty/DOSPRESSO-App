### 4A: Add API queries

```tsx
// Add these queries at the top of the component:

// HQ Summary KPIs
const { data: summary } = useQuery({
  queryKey: ['/api/hq/summary'],
  queryFn: () => fetch('/api/hq/summary').then(r => r.json()),
  refetchInterval: 5 * 60 * 1000, // refresh every 5 minutes
  staleTime: 2 * 60 * 1000,
});

// Active delegations (already exists from delegation sprint)
const { data: activeDelegations = [] } = useQuery({
```

### 4C: Replace hardcoded Alert Pills

Find the DashboardAlertPills usage and make it dynamic:

```tsx
// Build pills dynamically from real data:
const alertPills = useMemo(() => {
  const pills = [];
  
  if ((summary?.slaBreaches ?? 0) > 0) {
    pills.push({ 
      label: `${summary.slaBreaches} SLA İhlali`, 
      variant: 'red' as const, 
      dot: true 
    });
  }
  
  if ((summary?.openTickets ?? 0) > 0) {
    pills.push({ 
      label: `${summary.openTickets} Açık Ticket`, 
      variant: 'orange' as const, 
      dot: true 
    });
  }
  
  if ((summary?.activeBranches ?? 0) > 0) {
    pills.push({ 
      label: `${summary.activeBranches} Aktif Şube`, 
      variant: 'green' as const, 
      dot: true 
    });
  }
  
  // Add active delegation alerts
  activeDelegations.forEach(del => {
    pills.push({
      label: `${del.moduleName} devredildi`,
      variant: 'blue' as const,
      dot: true
    });
  });
  
  // Fallback: at least one pill if everything is 0
  if (pills.length === 0) {
    pills.push({ label: 'Sistem normal', variant: 'green' as const, dot: true });
  }
  
  return pills;
}, [summary, activeDelegations]);
```

Then pass `alertPills` to `<DashboardAlertPills pills={alertPills} />`.
