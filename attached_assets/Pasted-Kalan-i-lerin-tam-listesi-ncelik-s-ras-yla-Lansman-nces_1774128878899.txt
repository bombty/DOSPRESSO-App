```typescript
// Gap ID patterns → navigation targets
const GAP_NAVIGATION: Record<string, string> = {
  // Personnel
  'yonetici-yok': '/admin/kullanicilar',
  'yetersiz-personel': '/admin/kullanicilar',
  'no-manager': '/admin/kullanicilar',
  'insufficient-staff': '/admin/kullanicilar',
  
  // Checklist
  'checklist': '/yonetim/checklist-yonetimi',
  'acilis-kapalis': '/yonetim/checklist-yonetimi',
  
  // Shift/Schedule
  'vardiya': '/vardiya-planlama',
  'shift': '/vardiya-planlama',
  'no-shifts': '/vardiya-planlama',
  
  // Training
  'egitim': '/akademi-hq',
  'onboarding': '/akademi-hq',
  'training': '/akademi-hq',
  
  // Factory
  'fabrika': '/fabrika/dashboard',
  'uretim': '/fabrika/uretim-planlama',
  'factory': '/fabrika/dashboard',
  
  // Settings
  'sla': '/admin',
  'ayar': '/admin',
  'settings': '/admin',
};

function getNavigationForGap(gap: any): string {
  // 1. Check deepLink from backend
  if (gap.deepLink || gap.actionUrl || gap.action_url) {
    return gap.deepLink || gap.actionUrl || gap.action_url;
  }
  
  // 2. Match by gap ID or title keywords
  const searchText = ((gap.id || '') + ' ' + (gap.title || '') + ' ' + (gap.description || '')).toLowerCase();
  
  for (const [keyword, path] of Object.entries(GAP_NAVIGATION)) {
    if (searchText.includes(keyword)) return path;
  }
  
  // 3. Default
  return '/admin';
}
```

Update the "Düzelt" button:

<Button variant="outline" size="sm" onClick={() => navigate(getNavigationForGap(gap))}>

router.get("/api/some-stub", isAuthenticated, (req, res) => {
