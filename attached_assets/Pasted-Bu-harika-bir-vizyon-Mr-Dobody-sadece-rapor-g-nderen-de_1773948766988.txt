// GET /api/agent/guidance — returns guidance items for current user's role
router.get("/api/agent/guidance", isAuthenticated, async (req, res) => {
  try {
    const user = req.user as any;
    const allGaps = await detectSystemGaps();
    
    // Filter by user's role and branch
    const myGuidance = allGaps.filter(gap => {
      // Role match
      if (!gap.targetRoles.includes(user.role)) return false;
      // Branch match (if branch-specific)
      if (gap.targetBranchId && user.branchId && gap.targetBranchId !== user.branchId) {
        // HQ roles (coach, trainer, admin) see ALL branches
        if (!['admin', 'ceo', 'cgo', 'coach', 'trainer', 'muhasebe_ik', 'kalite_kontrol', 'gida_muhendisi'].includes(user.role)) {
          return false;
        }
      }
      return true;
    });
    
    // Sort by severity
    const severityOrder = { critical: 0, high: 1, medium: 2, low: 3 };
    myGuidance.sort((a, b) => severityOrder[a.severity] - severityOrder[b.severity]);
    
    // Group by category
    const grouped = {
      critical: myGuidance.filter(g => g.severity === 'critical'),
      high: myGuidance.filter(g => g.severity === 'high'),
      medium: myGuidance.filter(g => g.severity === 'medium'),
      low: myGuidance.filter(g => g.severity === 'low'),
    };
    
    res.json({
      totalGaps: myGuidance.length,
      criticalCount: grouped.critical.length,
      items: myGuidance,
      grouped,
      lastChecked: new Date().toISOString(),
    });
  } catch (error) {
    console.error("[Guidance] Error:", error);
    res.json({ totalGaps: 0, items: [], grouped: { critical: [], high: [], medium: [], low: [] } });
  }
});

// Daily at 06:00 — detect system gaps and create notifications for critical ones
async function runDailyGapDetection() {
  try {
    const gaps = await detectSystemGaps();
    const criticalGaps = gaps.filter(g => g.severity === 'critical');
    
    if (criticalGaps.length > 0) {
      // Create ONE summary notification per target role
      const roleGaps = new Map<string, CompletenessItem[]>();
      for (const gap of criticalGaps) {
        for (const role of gap.targetRoles) {
          if (!roleGaps.has(role)) roleGaps.set(role, []);
          roleGaps.get(role)!.push(gap);
        }
      }
      
      for (const [role, items] of roleGaps) {
        // Find users with this role
        const targetUsers = await db.select().from(usersTable)
          .where(and(eq(usersTable.role, role), eq(usersTable.isActive, true)));
        
        for (const targetUser of targetUsers) {
          // Dedup: don't send if same summary sent in last 24h
          await createNotificationWithDedup({
            userId: targetUser.id,
            type: 'agent_guidance',
            title: `Mr. Dobody: ${items.length} kritik eksiklik tespit edildi`,
            message: items.map(i => `• ${i.title}`).join('\n'),
            link: '/agent-merkezi',
          });
        }
      }
    }
    
    console.log(`[Agent Guidance] Daily check: ${gaps.length} total gaps, ${criticalGaps.length} critical`);
  } catch (error) {
    console.error("[Agent Guidance] Daily check failed:", error);
  }
}
