```bash
echo "════════════════════════════════════════════════════════"

echo "════════════════════════════════════════════════════════"

echo ""
echo "=== 0A: Sunucu durumu ==="
curl -s http://localhost:5000/api/health 2>/dev/null | head -5 || echo "Health endpoint yok, port kontrolü:"
curl -s -o /dev/null -w "%{http_code}" http://localhost:5000/ 2>/dev/null

echo "
SELECT id, username, first_name, last_name, role, branch_id,
  CASE WHEN branch_id IS NULL THEN 'HQ' ELSE (SELECT name FROM branches WHERE id = branch_id) END as branch_name
FROM users 
WHERE is_active = true 
AND role IN ('admin','ceo','cgo','coach','trainer','muhasebe_ik','fabrika_mudur','fabrika_operator','supervisor','mudur','barista','stajyer','yatirimci_branch','gida_muhendisi','kalite_kontrol','satinalma','sube_kiosk')
ORDER BY 
  CASE role 
    WHEN 'admin' THEN 1 WHEN 'ceo' THEN 2 WHEN 'cgo' THEN 3 
    WHEN 'coach' THEN 4 WHEN 'trainer' THEN 5 WHEN 'muhasebe_ik' THEN 6
    WHEN 'fabrika_mudur' THEN 7 WHEN 'gida_muhendisi' THEN 8
    WHEN 'mudur' THEN 9 WHEN 'supervisor' THEN 10
    WHEN 'barista' THEN 11 WHEN 'stajyer' THEN 12
    ELSE 20 END;
" | psql

echo ""
echo "=== 0C: Mevcut bildirim sayıları (baseline) ==="
echo "
SELECT u.role, u.first_name, COUNT(n.id) as bildirim_sayisi,
  COUNT(CASE WHEN n.created_at > NOW() - INTERVAL '24 hours' THEN 1 END) as son_24_saat
FROM users u 
LEFT JOIN notifications n ON n.user_id = u.id
WHERE u.is_active = true
AND u.role IN ('ceo','cgo','fabrika_mudur','gida_muhendisi','coach','trainer','muhasebe_ik','supervisor','mudur')
GROUP BY u.role, u.first_name
ORDER BY bildirim_sayisi DESC;
" | psql

echo ""
echo "=== 0D: Cookie/session hazırlığı ==="
# Her test rolü için login yapıp cookie kaydet
ROLES_TO_TEST="admin ceo cgo coach trainer muhasebe_ik fabrika_mudur supervisor mudur barista stajyer yatirimci_branch gida_muhendisi kalite_kontrol fabrika_operator satinalma"

for ROLE in $ROLES_TO_TEST; do
  USER=$(echo "SELECT username FROM users WHERE role='$ROLE' AND is_active=true LIMIT 1;" | psql -t | tr -d ' ')
  if [ -n "$USER" ]; then
    RESULT=$(curl -s -c /tmp/cookie_${ROLE}.txt -X POST http://localhost:5000/api/login \
      -H "Content-Type: application/json" \
      -d "{\"username\":\"$USER\",\"password\":\"0000\"}" \
      -w "\n%{http_code}" 2>/dev/null)
    HTTP_CODE=$(echo "$RESULT" | tail -1)
    echo "  $ROLE ($USER): HTTP $HTTP_CODE"
  else
    echo "  $ROLE: KULLANICI YOK — atlanıyor"
  fi
done
```

### 2.2 Canlı Bildirim Tetikleme Testi

```bash
echo ""
echo "════════════════════════════════════════"
echo "TEST GRUBU 2.2: CANLI BİLDİRİM TETİKLEME"
echo "════════════════════════════════════════"

echo "=== T2.2.1: Üretim kaydı gir → bildirim oluştu mu? ==="
echo "--- Bildirim öncesi sayı ---"
BEFORE=$(echo "SELECT COUNT(*) FROM notifications WHERE type LIKE 'factory%' AND created_at > NOW() - INTERVAL '5 minutes';" | psql -t | tr -d ' ')

echo ""
echo "--- Kiosk ile üretim kaydı simülasyonu ---"

  SESSION_ID=$(echo "$ACTIVE_SESSION" | cut -d'|' -f1 | tr -d ' ')
  echo "Session $SESSION_ID ile üretim kaydı deneniyor..."
  
  PROD_RESULT=$(curl -s -b /tmp/cookie_fabrika_operator.txt \
    -X POST http://localhost:5000/api/factory/kiosk/log-production \
    -H "Content-Type: application/json" \

    2>/dev/null)

  echo "⚠ Aktif kiosk session yok — üretim kaydı simülasyonu atlanıyor"
  echo "  NOT: Bu normal olabilir, kiosk oturumu başlatılmamışsa."
fi

echo ""
echo "--- Bildirim sonrası sayı ---"
```
