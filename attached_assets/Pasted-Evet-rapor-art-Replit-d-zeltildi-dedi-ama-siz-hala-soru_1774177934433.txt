```bash
echo "══════════════════════════════════════════"
echo "CANLI MENÜ TESTİ — GERÇEK DURUM"
echo "══════════════════════════════════════════"

echo "=== 1. SUNUCU ÇALIŞIYOR MU? ==="
curl -s -o /dev/null -w "Health: %{http_code}\n" http://localhost:5000/api/health

echo ""
echo "=== 2. ADMIN GİRİŞ + MENÜ ==="
# Admin login
ADMIN_COOKIE=$(curl -s -X POST http://localhost:5000/api/login \
  -H "Content-Type: application/json" \
  -d '{"username":"utku","password":"0000"}' \
  -c - 2>/dev/null | grep connect.sid | awk '{print $NF}')

echo "Cookie alındı: ${ADMIN_COOKIE:0:10}..."

# Admin menü öğelerini çek
echo "--- Admin menü öğeleri ---"
curl -s http://localhost:5000/api/me/menu -b "connect.sid=$ADMIN_COOKIE" 2>/dev/null | node -e "
  const data = require('fs').readFileSync('/dev/stdin','utf8');
  try {
    const menu = JSON.parse(data);
    if (Array.isArray(menu)) {
      menu.forEach(section => {
        console.log('\\n[' + (section.label || section.title || 'SECTION') + ']');
        if (section.items) {
          section.items.forEach(item => {
            console.log('  ' + (item.label || item.title) + ' → ' + (item.path || item.href || 'NO PATH'));
          });
        }
      });
    } else {
      console.log(JSON.stringify(menu, null, 2).substring(0, 2000));
    }
  } catch(e) { console.log('RAW:', data.substring(0, 1000)); }
"

echo ""
echo "=== 3. AYARLAR SPESİFİK KONTROL ==="
echo "--- Ayarlar menüde var mı? ---"
curl -s http://localhost:5000/api/me/menu -b "connect.sid=$ADMIN_COOKIE" 2>/dev/null | grep -i "ayarlar\|settings\|admin-panel"

echo "--- Ayarlar sayfası route var mı? ---"
grep -n "ayarlar\|settings\|admin-panel\|admin-settings" client/src/App.tsx | head -10

echo "--- Ayarlar sayfası dosyası var mı? ---"
find client/src/pages -name "*ayarlar*" -o -name "*settings*" -o -name "*admin-panel*" | sort

echo ""
echo "=== 4. KIRIK LİNK TESPİTİ ==="
echo "--- Menüdeki TÜM path'leri bul ---"
curl -s http://localhost:5000/api/me/menu -b "connect.sid=$ADMIN_COOKIE" 2>/dev/null | node -e "
  const data = require('fs').readFileSync('/dev/stdin','utf8');
  try {
    const menu = JSON.parse(data);
    const paths = [];
    const extract = (items) => {
      if (!items) return;
      items.forEach(item => {
        if (item.path) paths.push(item.path);
        if (item.href) paths.push(item.href);
        if (item.items) extract(item.items);
      });
    };
    if (Array.isArray(menu)) menu.forEach(s => { if(s.path) paths.push(s.path); extract(s.items); });
    paths.forEach(p => console.log(p));
  } catch(e) { console.log('PARSE ERROR'); }
" > /tmp/menu_paths.txt

echo "Toplam menü path sayısı: $(wc -l < /tmp/menu_paths.txt)"

echo ""
echo "--- Her path için App.tsx'te route var mı? ---"
while read path; do
  # Path'ten parametreleri çıkar
  search_path=$(echo "$path" | sed 's|/[0-9]*$||' | sed 's|/$||')
  found=$(grep -c "$search_path" client/src/App.tsx 2>/dev/null)
  if [ "$found" -eq "0" ]; then
    # Daha kısa arama dene
    short=$(echo "$search_path" | sed 's|^/||' | cut -d'/' -f1)
    found2=$(grep -c "$short" client/src/App.tsx 2>/dev/null)
    if [ "$found2" -eq "0" ]; then
      echo "  KIRIK: $path — App.tsx'te route YOK"
    fi
  fi
done < /tmp/menu_paths.txt

echo ""
echo "--- Her path için sayfa dosyası var mı? ---"
while read path; do
  pagename=$(echo "$path" | sed 's|^/||' | tr '/' '-')
  found=$(find client/src/pages -name "*${pagename}*" 2>/dev/null | head -1)
  if [ -z "$found" ]; then
    short=$(echo "$path" | sed 's|^/||' | cut -d'/' -f1)
    found2=$(find client/src/pages -name "*${short}*" 2>/dev/null | head -1)
    if [ -z "$found2" ]; then
      echo "  SAYFA YOK: $path"
    fi
  fi
done < /tmp/menu_paths.txt

echo ""
echo "=== 5. DİĞER ROLLER TESTİ ==="
for username in coach supervisor stajyer; do
  echo "--- $username menü sayısı ---"
  COOKIE=$(curl -s -X POST http://localhost:5000/api/login \
    -H "Content-Type: application/json" \
    -d "{\"username\":\"$username\",\"password\":\"0000\"}" \
    -c - 2>/dev/null | grep connect.sid | awk '{print $NF}')
  
  if [ -z "$COOKIE" ]; then
    # Farklı kullanıcı adı dene
    real_user=$(echo "SELECT username FROM users WHERE role='$username' AND is_active=true LIMIT 1;" | psql -t 2>/dev/null | tr -d ' ')
    if [ -n "$real_user" ]; then
      COOKIE=$(curl -s -X POST http://localhost:5000/api/login \
        -H "Content-Type: application/json" \
        -d "{\"username\":\"$real_user\",\"password\":\"0000\"}" \
        -c - 2>/dev/null | grep connect.sid | awk '{print $NF}')
      echo "  (kullanıcı: $real_user)"
    fi
  fi
  
  if [ -n "$COOKIE" ]; then
    count=$(curl -s http://localhost:5000/api/me/menu -b "connect.sid=$COOKIE" 2>/dev/null | node -e "
      const d=require('fs').readFileSync('/dev/stdin','utf8');
      try{const m=JSON.parse(d);let c=0;m.forEach(s=>{if(s.items)c+=s.items.length});console.log(c)}catch(e){console.log('?')}
    ")
    echo "  Menü öğesi sayısı: $count"
  else
    echo "  GİRİŞ BAŞARISIZ"
  fi
done

echo ""
echo "=== 6. SIDEBAR_ALLOWED_ITEMS DEĞİŞİKLİĞİ UYGULANMIŞ MI? ==="
echo "--- admin whitelist'te settings var mı? ---"
grep -A30 "admin:" server/menu-service.ts | grep "settings\|admin-panel\|users" | head -5

echo "--- alwaysVisible bug düzeltilmiş mi? ---"
grep -n "alwaysVisible" server/menu-service.ts | head -5

echo "--- Profil linki eklendi mi? ---"
grep -n "my-profile\|Profilim\|profil" server/menu-service.ts | head -5

echo "══════════════════════════════════════════"
echo "RAPORLA:"
echo "1. Menü düzeltmesi gerçekten uygulanmış mı?"
echo "2. Admin kaç menü öğesi görüyor — Ayarlar var mı?"
echo "3. Kırık linkler — menüde olup sayfası olmayan"
echo "4. Diğer roller menü durumu"
echo "5. Sorun devam ediyorsa KÖK NEDEN ne?"
echo "══════════════════════════════════════════"
```
