  "POST /api/factory/kiosk/start-shift" \

  "POST /api/factory/kiosk/end-shift" \

  url=$(echo $ep | cut -d' ' -f2)
  status=$(curl -s -o /dev/null -w "%{http_code}" -X $method http://localhost:5000$url -H "Content-Type: application/json" -d '{}')

# Test with kiosk token
echo "--- With kiosk token ---"
# First get a token

    "POST /api/factory/kiosk/start-shift" \
    "POST /api/factory/kiosk/log-break" \
